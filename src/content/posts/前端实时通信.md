## 概述

前端实时通信是现代Web应用中不可或缺的技术，它允许服务器主动向客户端推送数据，实现真正的实时交互。从即时通讯、在线协作到实时数据监控，实时通信技术为Web应用带来了全新的体验。本文将详细介绍WebSocket、Server-Sent Events、轮询等各种实时通信方案及其应用场景。

## 核心概念

### 实时通信方案对比

| 方案 | 优点 | 缺点 | 适用场景 |
|------|------|------|----------|
| WebSocket | 双向通信，低延迟 | 实现复杂，需要特殊协议 | 即时通讯、在线游戏 |
| SSE | 简单，自动重连 | 单向通信，只能文本 | 股票行情、新闻推送 |
| 长轮询 | 兼容性好 | 服务器资源消耗大 | 旧浏览器兼容 |
| 短轮询 | 实现简单 | 延迟高，服务器压力大 | 简单状态检查 |

### WebSocket协议

**1. 握手过程**
- 客户端发送HTTP升级请求
- 服务器返回101状态码
- 建立TCP长连接

**2. 数据传输**
- 帧格式传输数据
- 支持文本和二进制数据
- 低延迟双向通信

**3. 连接管理**
- 心跳检测
- 自动重连机制
- 连接状态监控

### 1. WebSocket基础实现

```javascript
// WebSocket管理器
class WebSocketManager {
    constructor(url, options = {}) {
        this.url = url;
        this.options = {
            reconnectInterval: options.reconnectInterval || 3000,
            maxReconnectAttempts: options.maxReconnectAttempts || 5,
            heartbeatInterval: options.heartbeatInterval || 30000,
            protocols: options.protocols || [],
            onMessage: options.onMessage || (() => {}),
            onOpen: options.onOpen || (() => {}),
            onClose: options.onClose || (() => {}),
            onError: options.onError || (() => {}),
            ...options
        };
        
        this.ws = null;
        this.reconnectAttempts = 0;
        this.heartbeatTimer = null;
        this.messageQueue = [];
        this.isManualClose = false;
        
        this.connect();
    }
    
    connect() {
        try {
            this.ws = new WebSocket(this.url, this.options.protocols);
            this.setupEventListeners();
        } catch (error) {
            console.error('WebSocket连接失败:', error);
            this.scheduleReconnect();
        }
    }
    
    setupEventListeners() {
        this.ws.onopen = (event) => {
            console.log('WebSocket连接已建立');
            this.reconnectAttempts = 0;
            this.startHeartbeat();
            this.flushMessageQueue();
            this.options.onOpen(event);
        };
        
        this.ws.onmessage = (event) => {
            this.handleMessage(event);
        };
        
        this.ws.onclose = (event) => {
            console.log('WebSocket连接已关闭:', event.code, event.reason);
            this.stopHeartbeat();
            
            if (!this.isManualClose) {
                this.scheduleReconnect();
            }
            
            this.options.onClose(event);
        };
        
        this.ws.onerror = (error) => {
            console.error('WebSocket错误:', error);
            this.options.onError(error);
        };
    }
    
    handleMessage(event) {
        try {
            // 尝试解析JSON消息
            const data = JSON.parse(event.data);
            
            // 处理心跳响应
            if (data.type === 'pong') {
                this.handlePong();
                return;
            }
            
            this.options.onMessage(data, event);
        } catch (error) {
            // 非JSON消息，直接传递原始数据
            this.options.onMessage(event.data, event);
        }
    }
    
    send(data) {
        if (this.ws && this.ws.readyState === WebSocket.OPEN) {
            const message = typeof data === 'string' ? data : JSON.stringify(data);
            this.ws.send(message);
        } else {
            // 连接未建立，加入队列
            this.messageQueue.push(data);
            console.warn('WebSocket未连接，消息已加入队列');
        }
    }
    
    flushMessageQueue() {
        while (this.messageQueue.length > 0 && this.ws.readyState === WebSocket.OPEN) {
            const message = this.messageQueue.shift();
            this.send(message);
        }
    }
    
    startHeartbeat() {
        this.stopHeartbeat();
        this.heartbeatTimer = setInterval(() => {
            this.send({ type: 'ping', timestamp: Date.now() });
        }, this.options.heartbeatInterval);
    }
    
    stopHeartbeat() {
        if (this.heartbeatTimer) {
            clearInterval(this.heartbeatTimer);
            this.heartbeatTimer = null;
        }
    }
    
    handlePong() {
        console.log('收到心跳响应');
        // 可以在这里更新连接状态
    }
    
    scheduleReconnect() {
        if (this.reconnectAttempts >= this.options.maxReconnectAttempts) {
            console.error('达到最大重连次数，停止重连');
            return;
        }
        
        this.reconnectAttempts++;
        const delay = this.options.reconnectInterval * Math.pow(2, this.reconnectAttempts - 1);
        
        console.log(`将在${delay}ms后尝试第${this.reconnectAttempts}次重连`);
        
        setTimeout(() => {
            this.connect();
        }, delay);
    }
    
    close() {
        this.isManualClose = true;
        this.stopHeartbeat();
        if (this.ws) {
            this.ws.close();
        }
    }
    
    getState() {
        if (!this.ws) return 'DISCONNECTED';
        
        switch (this.ws.readyState) {
            case WebSocket.CONNECTING:
                return 'CONNECTING';
            case WebSocket.OPEN:
                return 'OPEN';
            case WebSocket.CLOSING:
                return 'CLOSING';
            case WebSocket.CLOSED:
                return 'CLOSED';
            default:
                return 'UNKNOWN';
        }
    }
}

// 使用示例
```

### 2. Server-Sent Events (SSE)

```javascript
// SSE管理器
class SSEManager {
    constructor(url, options = {}) {
        this.url = url;
        this.options = {
            withCredentials: options.withCredentials || false,
            reconnectInterval: options.reconnectInterval || 3000,
            onMessage: options.onMessage || (() => {}),
            onOpen: options.onOpen || (() => {}),
            onError: options.onError || (() => {}),
            ...options
        };
        
        this.eventSource = null;
        this.reconnectTimer = null;
        this.isManualClose = false;
        
        this.connect();
    }
    
    connect() {
        try {
            this.eventSource = new EventSource(this.url, {
                withCredentials: this.options.withCredentials
            });
            
            this.setupEventListeners();
        } catch (error) {
            console.error('SSE连接失败:', error);
            this.scheduleReconnect();
        }
    }
    
    setupEventListeners() {
        this.eventSource.onopen = (event) => {
            console.log('SSE连接已建立');
            this.reconnectTimer = null;
            this.options.onOpen(event);
        };
        
        this.eventSource.onmessage = (event) => {
            this.handleMessage(event);
        };
        
        this.eventSource.onerror = (error) => {
            console.error('SSE错误:', error);
            this.options.onError(error);
            
            if (!this.isManualClose) {
                this.scheduleReconnect();
            }
        };
        
        // 监听自定义事件
        if (this.options.eventTypes) {
            this.options.eventTypes.forEach(eventType => {
                this.eventSource.addEventListener(eventType, (event) => {
                    this.handleCustomEvent(eventType, event);
                });
            });
        }
    }
    
    handleMessage(event) {
        try {
            const data = JSON.parse(event.data);
            this.options.onMessage(data, event);
        } catch (error) {
            this.options.onMessage(event.data, event);
        }
    }
    
    handleCustomEvent(eventType, event) {
        try {
            const data = JSON.parse(event.data);
            if (this.options[eventType]) {
                this.options[eventType](data, event);
            }
        } catch (error) {
            console.error('处理自定义事件失败:', error);
        }
    }
    
    scheduleReconnect() {
        if (this.reconnectTimer) return;
        
        console.log(`将在${this.options.reconnectInterval}ms后重连`);
        
        this.reconnectTimer = setTimeout(() => {
            this.close();
            this.connect();
        }, this.options.reconnectInterval);
    }
    
    close() {
        this.isManualClose = true;
        if (this.reconnectTimer) {
            clearTimeout(this.reconnectTimer);
            this.reconnectTimer = null;
        }
        if (this.eventSource) {
            this.eventSource.close();
            this.eventSource = null;
        }
    }
    
    getState() {
        if (!this.eventSource) return 'CLOSED';
        
        switch (this.eventSource.readyState) {
            case EventSource.CONNECTING:
                return 'CONNECTING';
            case EventSource.OPEN:
                return 'OPEN';
            case EventSource.CLOSED:
                return 'CLOSED';
            default:
                return 'UNKNOWN';
        }
    }
}
```

```javascript
// 轮询管理器
class PollingManager {
    constructor(options = {}) {
        this.options = {
            url: options.url,
            interval: options.interval || 5000,
            method: options.method || 'GET',
            headers: options.headers || {},
            body: options.body || null,
            onSuccess: options.onSuccess || (() => {}),
            onError: options.onError || (() => {}),
            enabled: options.enabled !== false,
            ...options
        };
        
        this.timer = null;
        this.isPolling = false;
        
        if (this.options.enabled) {
            this.start();
        }
    }
    
    async poll() {
        if (!this.options.enabled || this.isPolling) return;
        
        this.isPolling = true;
        
        try {
            const response = await fetch(this.options.url, {
                method: this.options.method,
                headers: this.options.headers,
                body: this.options.body
            });
            
            if (!response.ok) {
                throw new Error(`HTTP error! status: ${response.status}`);
            }
            
            const data = await response.json();
            await this.options.onSuccess(data);
            
        } catch (error) {
            console.error('轮询请求失败:', error);
            await this.options.onError(error);
        } finally {
            this.isPolling = false;
        }
    }
    
    start() {
        if (this.timer) return;
        
        this.options.enabled = true;
        this.poll(); // 立即执行一次
        this.timer = setInterval(() => {
            this.poll();
        }, this.options.interval);
        
        console.log(`轮询已启动，间隔: ${this.options.interval}ms`);
    }
    
    stop() {
        this.options.enabled = false;
        
        if (this.timer) {
            clearInterval(this.timer);
            this.timer = null;
        }
        
        console.log('轮询已停止');
    }
    
    updateInterval(newInterval) {
        this.options.interval = newInterval;
        
        if (this.options.enabled) {
            this.stop();
            this.start();
        }
    }
    
    updateOptions(newOptions) {
        Object.assign(this.options, newOptions);
        
        if (this.options.enabled && !this.timer) {
            this.start();
        } else if (!this.options.enabled && this.timer) {
            this.stop();
        }
    }
}

// 使用示例：检查订单状态
const orderPolling = new PollingManager({
    url: '/api/orders/12345/status',
    interval: 3000, // 3秒轮询一次
    method: 'GET',

        'Authorization': 'Bearer your-token'
    },
    onSuccess: (data) => {
        console.log('订单状态:', data);
        
        updateOrderStatus(data.status);
        
        // 如果订单已完成，停止轮询
        if (data.status === 'completed' || data.status === 'cancelled') {
            orderPolling.stop();
            showOrderComplete(data.status);
        }
    },
    onError: (error) => {
        console.error('获取订单状态失败:', error);
        showError('无法获取订单状态');
    }
});

function updateOrderStatus(status) {
    const statusElement = document.getElementById('order-status');
    statusElement.textContent = getStatusText(status);
    statusElement.className = `status ${status}`;
```

```javascript
// 即时通讯客户端
class ChatClient {
    constructor(config) {
        this.config = {
            wsUrl: config.wsUrl,
            apiBaseUrl: config.apiBaseUrl,
            userId: config.userId,
            roomId: config.roomId,
            token: config.token,
            onMessage: config.onMessage || (() => {}),
            onUserJoined: config.onUserJoined || (() => {}),
            onUserLeft: config.onUserLeft || (() => {}),
            ...config
        };
        
        this.ws = null;
        this.messageHistory = [];
        this.typingUsers = new Set();
        this.typingTimer = null;
        
        this.connect();
    }
    
    connect() {
        this.ws = new WebSocket(`${this.config.wsUrl}?userId=${this.config.userId}&token=${this.config.token}`);
        
        this.ws.onopen = () => {
            console.log('聊天服务连接成功');
            this.joinRoom();
        };
        
        this.ws.onmessage = (event) => {
            this.handleMessage(JSON.parse(event.data));
        };
        
        this.ws.onclose = () => {
            console.log('聊天服务连接断开');
            this.scheduleReconnect();
        };
        
        this.ws.onerror = (error) => {
            console.error('聊天服务错误:', error);
        };
    }
    
    handleMessage(data) {
        switch (data.type) {
            case 'message':
                this.handleChatMessage(data);
                break;
            case 'user_joined':
                this.handleUserJoined(data);
                break;
            case 'user_left':
                this.handleUserLeft(data);
                break;
            case 'typing':
                this.handleTyping(data);
                break;
            case 'history':
                this.handleHistory(data);
                break;
            case 'error':
                this.handleError(data);
                break;
        }
    }
    
    handleChatMessage(data) {
        const message = {
            id: data.messageId,
            userId: data.userId,
            username: data.username,
            content: data.content,
            timestamp: data.timestamp,
            isOwn: data.userId === this.config.userId
        };
        
        this.messageHistory.push(message);
        this.config.onMessage(message);
    }
    
    handleUserJoined(data) {
        console.log(`用户 ${data.username} 加入房间`);
        this.config.onUserJoined(data);
        this.showSystemMessage(`${data.username} 加入了聊天`);
    }
    
    handleUserLeft(data) {
        console.log(`用户 ${data.username} 离开房间`);
        this.config.onUserLeft(data);
        this.showSystemMessage(`${data.username} 离开了聊天`);
    }
    
    handleTyping(data) {
        if (data.userId === this.config.userId) return;
        
        this.typingUsers.add(data.username);
        this.updateTypingIndicator();
        
        // 清除之前的定时器
        if (this.typingTimer) {
            clearTimeout(this.typingTimer);
        }
        
        // 3秒后移除正在输入状态
        this.typingTimer = setTimeout(() => {
            this.typingUsers.delete(data.username);
            this.updateTypingIndicator();
        }, 3000);
    }
    
    handleHistory(data) {
        this.messageHistory = data.messages || [];
        this.config.onMessage(null, this.messageHistory);
    }
    
    handleError(data) {
        console.error('聊天错误:', data.message);
        this.showError(data.message);
    }
    
    joinRoom() {
        this.send({
            type: 'join_room',
            roomId: this.config.roomId
        });
    }
    
    sendMessage(content) {
        this.send({
            type: 'message',
            roomId: this.config.roomId,
            content: content,
            timestamp: Date.now()
        });
    }
    
    sendTyping() {
        this.send({
            type: 'typing',
            roomId: this.config.roomId
        });
    }
    
    send(data) {
        if (this.ws && this.ws.readyState === WebSocket.OPEN) {
            this.ws.send(JSON.stringify(data));
        } else {
            console.warn('WebSocket未连接，无法发送消息');
        }
    }
    
    updateTypingIndicator() {
        const indicator = document.getElementById('typing-indicator');
        
        if (this.typingUsers.size > 0) {
            const usernames = Array.from(this.typingUsers).join(', ');
            indicator.textContent = `${usernames} 正在输入...`;
            indicator.style.display = 'block';
        } else {
            indicator.style.display = 'none';
        }
    }
    
    showSystemMessage(content) {
        this.config.onMessage({
            type: 'system',
            content: content,
            timestamp: Date.now()
        });
    }
    
    showError(message) {
        alert('错误: ' + message);
    }
    
    getHistory() {
        return this.messageHistory;
    }
    
    disconnect() {
        if (this.ws) {
            this.ws.close();
        }
    }
}
```

```javascript
class CollaborativeEditor {
    constructor(config) {
        this.config = {
            wsUrl: config.wsUrl,
            documentId: config.documentId,
            userId: config.userId,
            token: config.token,
            onRemoteChange: config.onRemoteChange || (() => {}),

            ...config

        this.ws = null;
        this.pendingOperations = [];
        this.cursors = new Map();
        this.transformers = new OperationalTransformers();

    }
    
    connect() {

            this.requestDocument();
        };
        
        this.ws.onmessage = (event) => {
            this.handleMessage(JSON.parse(event.data));
        };
        
        this.ws.onclose = () => {
            console.log('协作编辑服务连接断开');
            this.scheduleReconnect();
        };
        
        this.ws.onerror = (error) => {
            console.error('协作编辑服务错误:', error);
        };
    }
    
    handleMessage(data) {
        switch (data.type) {
            case 'document':
                this.handleDocument(data);
                break;
            case 'operation':
                this.handleOperation(data);
                break;
            case 'cursor':
                this.handleCursor(data);
                break;
            case 'user_join':
                this.handleUserJoin(data);
                break;
            case 'user_leave':
                this.handleUserLeave(data);
                break;
            case 'ack':
                this.handleAck(data);
                break;
        }
    }
    
    handleDocument(data) {
        console.log('收到文档内容:', data.content);
        this.document = data.content;
        this.revision = data.revision;
        
        // 初始化编辑器
        this.initializeEditor(data.content);
    }
    
    handleOperation(data) {
        if (data.userId === this.config.userId) {
            // 本地操作的确认
            this.handleAck(data);
        } else {
            // 远程操作
            this.applyRemoteOperation(data.operation);
        }
    }
    
    handleCursor(data) {
        if (data.userId === this.config.userId) return;
        
        this.cursors.set(data.userId, {
            position: data.position,
            selection: data.selection,
            color: data.color,
            username: data.username
        });

    handleUserJoin(data) {

        this.config.onUserJoin(data);

        this.cursors.delete(data.userId);
        this.updateCursorDisplay();
        this.config.onUserLeave(data);
    }
    
    handleAck(data) {
        // 移除已确认的待处理操作
        this.pendingOperations = this.pendingOperations.filter(op => op.id !== data.operationId);
    }
    
    requestDocument() {
        this.send({
            type: 'get_document'
        });
    }
    
    applyLocalOperation(operation) {
        // 应用本地操作
        const result = this.transformers.apply(this.document, operation);
        this.document = result.document;
        
        // 发送到服务器
        const operationData = {
            id: this.generateOperationId(),
            operation: operation,
            revision: this.revision,
            timestamp: Date.now()
        };
        
        this.pendingOperations.push(operationData);
        this.send({
            type: 'operation',
            data: operationData
        });
        
        return result;
    }
    
    applyRemoteOperation(operation) {
        // 转换操作以解决冲突
        const transformed = this.transformers.transform(operation, this.pendingOperations);
        
        // 应用转换后的操作
        const result = this.transformers.apply(this.document, transformed);
        this.document = result.document;
        
        // 更新UI
        this.config.onRemoteChange(transformed);
    }

            position: position,
            selection: selection
        });

    }

            this.ws.send(JSON.stringify(data));
        }
    }

    }

// 操作转换器（简化版）
class OperationalTransformers {
    apply(document, operation) {
        // 应用操作到文档
        let result = document;
        
        if (operation.type === 'insert') {
            result = document.substring(0, operation.position) + 
                     operation.content + 
                     document.substring(operation.position);
        } else if (operation.type === 'delete') {
            result = document.substring(0, operation.position) + 
                     document.substring(operation.position + operation.length);
        }
        
        return { document: result };
    }
    
    transform(operation, pendingOperations) {
        // 转换操作以解决冲突
        let transformed = { ...operation };
        
        pendingOperations.forEach(pending => {
            if (pending.operation.position <= transformed.position) {
                if (pending.operation.type === 'insert') {
                    transformed.position += pending.operation.content.length;
                } else if (pending.operation.type === 'delete') {
                    transformed.position = Math.max(
                        transformed.position - pending.operation.length,
                        pending.operation.position
                    );
                }
            }
        });
        
        return transformed;
    }
}
```

```javascript
class ConnectionManager {

        }
        
        this.connectionState.isConnecting = true;
        
        try {
            await this.connectWithTimeout(url);
            this.onConnectionSuccess();
        } catch (error) {
            this.onConnectionFailure(error);
        }
    }
    
    connectWithTimeout(url) {
        return new Promise((resolve, reject) => {
            this.ws = new WebSocket(url);
            
            // 设置连接超时
            this.connectionTimeoutTimer = setTimeout(() => {
                this.ws.close();
                reject(new Error('连接超时'));

            this.ws.onopen = () => {
                clearTimeout(this.connectionTimeoutTimer);
                resolve();
            };
            
            this.ws.onerror = (error) => {
                clearTimeout(this.connectionTimeoutTimer);
                reject(error);
            };
        });
    }
    
    onConnectionSuccess() {
        this.connectionState.isConnected = true;
        this.connectionState.isConnecting = false;
        this.connectionState.lastConnectedTime = Date.now();
        this.reconnectAttempts = 0;
        
        console.log('连接成功');
        this.startHeartbeat();
    }
    
    onConnectionFailure(error) {
        this.connectionState.isConnected = false;
        this.connectionState.isConnecting = false;
        
        console.error('连接失败:', error);
        
        if (!this.isManualClose) {
            this.scheduleReconnect();
        }
    }
    
    startHeartbeat() {
        this.stopHeartbeat();
        
        this.heartbeatTimer = setInterval(() => {
            if (this.ws && this.ws.readyState === WebSocket.OPEN) {
                this.ws.send(JSON.stringify({ type: 'ping' }));
            }
        }, this.options.heartbeatInterval);
    }
    
    stopHeartbeat() {
        if (this.heartbeatTimer) {
            clearInterval(this.heartbeatTimer);
            this.heartbeatTimer = null;
        }
    }
    
    scheduleReconnect() {
        if (this.reconnectAttempts >= this.options.maxReconnectAttempts) {
            console.error('达到最大重连次数');
            return;
        }
        
        this.reconnectAttempts++;
        this.connectionState.reconnectCount++;
        
        // 指数退避
        const delay = this.options.reconnectInterval * Math.pow(2, this.reconnectAttempts - 1);
        
        console.log(`将在${delay}ms后尝试第${this.reconnectAttempts}次重连`);
        
        setTimeout(() => {
            this.connect(this.url);
        }, delay);
    }
    
    close() {
        this.isManualClose = true;
        this.stopHeartbeat();
        
        if (this.connectionTimeoutTimer) {
            clearTimeout(this.connectionTimeoutTimer);
        }
        
        if (this.ws) {
            this.ws.close();
        }
        
        this.connectionState.isConnected = false;
        this.connectionState.isConnecting = false;
    }
    
    getConnectionState() {
        return {
            ...this.connectionState,
            reconnectAttempts: this.reconnectAttempts
        };
    }
```

```javascript
            throw new Error(validation.error);

// 消息验证器
class MessageValidator {
    constructor() {
        this.schema = {
            type: 'object',
            properties: {
                type: { type: 'string' },
                data: { type: ['string', 'object'] },
                timestamp: { type: 'number' },
                userId: { type: 'string' },
                signature: { type: 'string' }
            },
            required: ['type', 'timestamp']
        };
    }
    
    validate(message) {
        // 基本结构验证
        if (!message || typeof message !== 'object') {
            return { isValid: false, error: '无效的消息格式' };
        }
        
        // 必填字段检查
        for (const field of this.schema.required) {
            if (!(field in message)) {
                return { isValid: false, error: `缺少必填字段: ${field}` };
            }
        }
        
        // 类型检查
        for (const [field, schema] of Object.entries(this.schema.properties)) {
            if (field in message) {
                const isValidType = Array.isArray(schema.type)
                    ? schema.type.includes(typeof message[field])
                    : typeof message[field] === schema.type;
                
                if (!isValidType) {
                    return { isValid: false, error: `字段类型错误: ${field}` };
                }
            }
        }
        
        // 时间戳验证
        const now = Date.now();
        const maxTimeDiff = 5 * 60 * 1000; // 5分钟
        
        if (Math.abs(now - message.timestamp) > maxTimeDiff) {
            return { isValid: false, error: '时间戳无效' };
        }
        
        return { isValid: true };
    }
}
```

```javascript
        if (this.options.prioritizeTypes.includes(priority)) {
            this.flush();
        }
    }
    
    async flush() {
        if (this.isProcessing || this.messageQueue.length === 0) {
            return;
        }
        
        this.isProcessing = true;
        
        try {
            const batchSize = Math.min(this.options.batchSize, this.messageQueue.length);
            const batch = this.messageQueue.splice(0, batchSize);
            
            // 批量发送消息
            await this.sendBatch(batch);
            
        } catch (error) {
            console.error('批量发送失败:', error);
            // 重新加入队列
            this.messageQueue.unshift(...batch);
        } finally {
            this.isProcessing = false;
        }
    }
    
    async sendBatch(batch) {
        // 这里应该实现实际的发送逻辑
        console.log(`发送 ${batch.length} 条消息`);
        
        // 模拟网络延迟
        await new Promise(resolve => setTimeout(resolve, 100));
    }
    
    generateMessageId() {
        return `${Date.now()}-${Math.random().toString(36).substr(2, 9)}`;
    }
    
    getQueueSize() {
        return this.messageQueue.length;
    }
    
    getQueueStats() {
        const stats = {
            total: this.messageQueue.length,
            byPriority: {}
        };
        
        this.messageQueue.forEach(msg => {
            stats.byPriority[msg.priority] = (stats.byPriority[msg.priority] || 0) + 1;
        });
        
        return stats;
    }
```
