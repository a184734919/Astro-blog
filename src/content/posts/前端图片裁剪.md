## 概述

前端图片裁剪是现代Web应用中常用的功能，广泛应用于头像上传、图片编辑、社交媒体等场景。通过在浏览器端直接裁剪图片，可以减少服务器负载、提升用户体验，并让用户有更多的控制权。本文将详细介绍图片裁剪的各种实现方式、技术细节和最佳实践。

## 核心概念

### 裁剪原理

**1. 视口坐标系**
- 原始图片的像素坐标系
- 显示区域的视口坐标系
- 裁剪框的相对坐标系

**2. 坐标转换**
- 图片原始坐标 → 视口坐标
- 视口坐标 → 裁剪坐标
- 裁剪坐标 → 输出坐标

**3. 比例计算**
- 宽高比控制
- 缩放比例计算
- 位置偏移计算

### 裁剪功能分类

**1. 基础裁剪**
- 固定比例裁剪
- 自由比例裁剪
- 预设尺寸裁剪

**2. 高级裁剪**
- 圆形裁剪
- 旋转裁剪
- 翻转裁剪

**3. 智能裁剪**
- 自动对焦裁剪
- 人脸识别裁剪
- 内容感知裁剪

```javascript
// 基础图片裁剪器
class BasicImageCropper {
    constructor(containerId, options = {}) {
        this.container = document.getElementById(containerId);
        this.options = {
            aspectRatio: options.aspectRatio || null, // null表示自由比例
            minWidth: options.minWidth || 50,
            minHeight: options.minHeight || 50,
            maxWidth: options.maxWidth || 800,
            maxHeight: options.maxHeight || 800,
            outputWidth: options.outputWidth || 400,
            outputHeight: options.outputHeight || 400,
            outputFormat: options.outputFormat || 'image/jpeg',
            outputQuality: options.outputQuality || 0.9,
            ...options
        };
        
        this.cropBox = { x: 0, y: 0, width: 200, height: 200 };
        this.imageData = null;
        this.isDragging = false;
        this.startX = 0;
        this.startY = 0;
        
        this.init();
    }
    
    init() {
        this.createUI();
        this.setupEventListeners();
    }
    
    createUI() {
        this.container.innerHTML = `
            <div class="cropper-wrapper">
                <div class="cropper-container">
                    <div class="cropper-image-wrapper">
                        <img class="cropper-image" alt="待裁剪图片">
                        <div class="crop-box">
                            <div class="crop-box-border"></div>
                            <div class="crop-handle crop-handle-nw"></div>
                            <div class="crop-handle crop-handle-ne"></div>
                            <div class="crop-handle crop-handle-sw"></div>
                            <div class="crop-handle crop-handle-se"></div>
                            <div class="crop-handle crop-handle-n"></div>
                            <div class="crop-handle crop-handle-s"></div>
                            <div class="crop-handle crop-handle-w"></div>
                            <div class="crop-handle crop-handle-e"></div>
                            <div class="crop-overlay"></div>
                        </div>
                    </div>
                </div>
                <div class="cropper-controls">
                    <div class="control-group">
                        <label>宽高比:</label>
                        <select class="aspect-ratio-select">
                            <option value="free">自由</option>
                            <option value="1">1:1</option>
                            <option value="1.33">4:3</option>
                            <option value="1.77">16:9</option>
                            <option value="0.75">3:4</option>
                        </select>
                    </div>
                    <div class="control-group">
                        <button class="btn-rotate-left">向左旋转</button>
                        <button class="btn-rotate-right">向右旋转</button>
                    </div>
                    <div class="control-group">
                        <button class="btn-reset">重置</button>
                        <button class="btn-crop">裁剪</button>
                    </div>
                </div>
                <div class="cropper-preview">
                    <h4>预览</h4>
                    <div class="preview-container"></div>
                </div>
            </div>
        `;
        
        this.elements = {
            image: this.container.querySelector('.cropper-image'),
            imageWrapper: this.container.querySelector('.cropper-image-wrapper'),
            cropBox: this.container.querySelector('.crop-box'),
            aspectRatioSelect: this.container.querySelector('.aspect-ratio-select'),
            previewContainer: this.container.querySelector('.preview-container'),
            rotateLeftBtn: this.container.querySelector('.btn-rotate-left'),
            rotateRightBtn: this.container.querySelector('.btn-rotate-right'),
            resetBtn: this.container.querySelector('.btn-reset'),
            cropBtn: this.container.querySelector('.btn-crop')
        };
    }
    
    setupEventListeners() {
        // 裁剪框拖拽
        this.elements.cropBox.addEventListener('mousedown', this.handleCropBoxMouseDown.bind(this));
        document.addEventListener('mousemove', this.handleMouseMove.bind(this));
        document.addEventListener('mouseup', this.handleMouseUp.bind(this));
        
        // 触摸事件
        this.elements.cropBox.addEventListener('touchstart', this.handleTouchStart.bind(this));
        document.addEventListener('touchmove', this.handleTouchMove.bind(this));
        document.addEventListener('touchend', this.handleTouchEnd.bind(this));
        
        // 控制按钮
        this.elements.aspectRatioSelect.addEventListener('change', this.handleAspectRatioChange.bind(this));
        this.elements.rotateLeftBtn.addEventListener('click', () => this.rotateImage(-90));
        this.elements.rotateRightBtn.addEventListener('click', () => this.rotateImage(90));
        this.elements.resetBtn.addEventListener('click', this.resetCrop.bind(this));
        this.elements.cropBtn.addEventListener('click', this.cropImage.bind(this));
    }
    
    loadImage(imageSource) {
        return new Promise((resolve, reject) => {
            const img = new Image();
            img.onload = () => {
                this.imageData = {
                    original: img,
                    width: img.naturalWidth,
                    height: img.naturalHeight,
                    aspectRatio: img.naturalWidth / img.naturalHeight
                };
                
                this.elements.image.src = imageSource;
                this.resetCrop();
                resolve();
            };
            img.onerror = () => reject(new Error('图片加载失败'));
            img.src = imageSource;
        });
    }
    
    handleCropBoxMouseDown(e) {
        if (e.target.classList.contains('crop-handle')) {
            this.isResizing = true;
            this.resizeHandle = e.target;
        } else {
            this.isDragging = true;
        }
        
        this.startX = e.clientX;
        this.startY = e.clientY;
        this.startCropBox = { ...this.cropBox };
        
        e.preventDefault();
    }
    
    handleMouseMove(e) {
        if (!this.isDragging && !this.isResizing) return;
        
        const deltaX = e.clientX - this.startX;
        const deltaY = e.clientY - this.startY;
        
        if (this.isDragging) {
            this.moveCropBox(deltaX, deltaY);
        } else if (this.isResizing) {
            this.resizeCropBox(deltaX, deltaY);
        }
        
        this.updateCropBoxUI();
        this.updatePreview();
    }
    
    handleMouseUp() {
        this.isDragging = false;
        this.isResizing = false;
        this.resizeHandle = null;
    }
    
    handleTouchStart(e) {
        const touch = e.touches[0];
        this.handleCropBoxMouseDown({
            target: e.target,
            clientX: touch.clientX,
            clientY: touch.clientY,
            preventDefault: () => e.preventDefault()
        });
    }
    
    handleTouchMove(e) {
        const touch = e.touches[0];
        this.handleMouseMove({
            clientX: touch.clientX,
            clientY: touch.clientY
        });
    }
    
    handleTouchEnd() {
        this.handleMouseUp();
    }
    
    moveCropBox(deltaX, deltaY) {
        const imageRect = this.elements.image.getBoundingClientRect();
        
        let newX = this.startCropBox.x + deltaX;
        let newY = this.startCropBox.y + deltaY;
        
        // 边界限制
        newX = Math.max(0, Math.min(newX, imageRect.width - this.cropBox.width));
        newY = Math.max(0, Math.min(newY, imageRect.height - this.cropBox.height));
        
        this.cropBox.x = newX;
        this.cropBox.y = newY;
    }
    
    resizeCropBox(deltaX, deltaY) {
        let newX = this.startCropBox.x;
        let newY = this.startCropBox.y;
        let newWidth = this.startCropBox.width;
        let newHeight = this.startCropBox.height;
        
        const handleClass = this.resizeHandle.className;
        
        if (handleClass.includes('e')) {
            newWidth += deltaX;
        } else if (handleClass.includes('w')) {
            newX += deltaX;
            newWidth -= deltaX;
        }
        
        if (handleClass.includes('s')) {
            newHeight += deltaY;
        } else if (handleClass.includes('n')) {
            newY += deltaY;
            newHeight -= deltaY;
        }
        
        // 尺寸限制
        newWidth = Math.max(this.options.minWidth, Math.min(newWidth, this.options.maxWidth));
        newHeight = Math.max(this.options.minHeight, Math.min(newHeight, this.options.maxHeight));
        
        // 宽高比限制
        if (this.options.aspectRatio) {
            if (handleClass.includes('e') || handleClass.includes('w')) {
                newHeight = newWidth / this.options.aspectRatio;
            } else {
                newWidth = newHeight * this.options.aspectRatio;
            }
        }
        
        this.cropBox = { x: newX, y: newY, width: newWidth, height: newHeight };
    }
    
    handleAspectRatioChange(e) {
        const value = e.target.value;
        
        if (value === 'free') {
            this.options.aspectRatio = null;
        } else {
            this.options.aspectRatio = parseFloat(value);
            // 调整当前裁剪框的宽高比
            if (this.cropBox.width > 0) {
                this.cropBox.height = this.cropBox.width / this.options.aspectRatio;
            }
        }
        
        this.updateCropBoxUI();
        this.updatePreview();
    }
    
    rotateImage(degrees) {
        if (!this.imageData) return;
        
        this.imageData.rotation = (this.imageData.rotation || 0) + degrees;
        
        const currentRotation = this.imageData.rotation;
        this.elements.image.style.transform = `rotate(${currentRotation}deg)`;
        
        // 调整容器尺寸
        this.adjustImageWrapper();
    }
    
    adjustImageWrapper() {
        const rotation = this.imageData.rotation || 0;
        const absRotation = Math.abs(rotation) % 180;
        const radians = absRotation * Math.PI / 180;
        
        const cos = Math.abs(Math.cos(radians));
        const sin = Math.abs(Math.sin(radians));
        
        const newWidth = this.imageData.width * cos + this.imageData.height * sin;
        const newHeight = this.imageData.width * sin + this.imageData.height * cos;
        
        const scale = Math.min(
            this.elements.imageWrapper.clientWidth / newWidth,
            this.elements.imageWrapper.clientHeight / newHeight,
            1
        );
        
        this.elements.image.style.width = `${this.imageData.width * scale}px`;
        this.elements.image.style.height = `${this.imageData.height * scale}px`;
        this.imageData.scale = scale;
    }
    
    updateCropBoxUI() {
        const { cropBox } = this;
        this.elements.cropBox.style.left = `${cropBox.x}px`;
        this.elements.cropBox.style.top = `${cropBox.y}px`;
        this.elements.cropBox.style.width = `${cropBox.width}px`;
        this.elements.cropBox.style.height = `${cropBox.height}px`;
    }
    
    updatePreview() {
        if (!this.imageData) return;
        
        const preview = this.generateCropPreview();
        this.elements.previewContainer.innerHTML = '';
        this.elements.previewContainer.appendChild(preview);
    }
    
    generateCropPreview() {
        const canvas = document.createElement('canvas');
        const ctx = canvas.getContext('2d');
        
        canvas.width = this.options.outputWidth;
        canvas.height = this.options.outputHeight;
        
        const img = this.elements.image;
        const rect = img.getBoundingClientRect();
        
        // 计算裁剪区域在原始图片中的位置
        const scaleX = this.imageData.width / rect.width;
        const scaleY = this.imageData.height / rect.height;
        
        const sourceX = this.cropBox.x * scaleX;
        const sourceY = this.cropBox.y * scaleY;
        const sourceWidth = this.cropBox.width * scaleX;
        const sourceHeight = this.cropBox.height * scaleY;
        
        ctx.drawImage(
            img,
            sourceX, sourceY,
            sourceWidth, sourceHeight,
            0, 0,
            canvas.width, canvas.height
        );
        
        return canvas;
    }
    
    cropImage() {
        const preview = this.generateCropPreview();
        const dataUrl = preview.toDataURL(this.options.outputFormat, this.options.outputQuality);
        
        // 创建下载链接
        const link = document.createElement('a');
        link.download = `cropped-image.${this.getExtension(this.options.outputFormat)}`;
        link.href = dataUrl;
        link.click();
        
        return dataUrl;
    }
    
    getExtension(mimeType) {
        const mimeMap = {
            'image/jpeg': 'jpg',
            'image/png': 'png',
            'image/webp': 'webp'
        };
        return mimeMap[mimeType] || 'jpg';
    }
    
    resetCrop() {
        if (!this.imageData) return;
        
        const imageRect = this.elements.image.getBoundingClientRect();
        const defaultSize = Math.min(imageRect.width, imageRect.height) * 0.8;
        
        this.cropBox = {
            x: (imageRect.width - defaultSize) / 2,
            y: (imageRect.height - defaultSize) / 2,
            width: defaultSize,
            height: this.options.aspectRatio ? defaultSize / this.options.aspectRatio : defaultSize
        };
        
        this.updateCropBoxUI();
        this.updatePreview();
    }
}

// 使用示例
const cropper = new BasicImageCropper('cropper-container', {
    aspectRatio: 1, // 1:1正方形
    outputWidth: 400,
    outputHeight: 400,
    outputFormat: 'image/jpeg',
    outputQuality: 0.9
});

// 加载图片
cropper.loadImage('https://example.com/image.jpg');
```

    moveCropCircle(deltaX, deltaY) {
        const containerRect = this.elements.imageContainer.getBoundingClientRect();
        
        let newX = this.startCropCircle.x + deltaX;
        let newY = this.startCropCircle.y + deltaY;
        
        // 边界限制
        newX = Math.max(this.cropCircle.radius, Math.min(newX, containerRect.width - this.cropCircle.radius));
        newY = Math.max(this.cropCircle.radius, Math.min(newY, containerRect.height - this.cropCircle.radius));
        
        this.cropCircle.x = newX;
        this.cropCircle.y = newY;
    }
    
    resizeCropCircle(deltaX, deltaY) {
        const delta = Math.max(deltaX, deltaY);
        let newRadius = this.startCropCircle.radius + delta / 2;
        
        // 尺寸限制
        newRadius = Math.max(this.options.minSize / 2, Math.min(newRadius, this.options.maxSize / 2));
        
        this.cropCircle.radius = newRadius;
    }
    
    handleZoom(e) {
        const scale = parseFloat(e.target.value);
        this.setZoom(scale);
    }
    
    adjustZoom(delta) {
        const currentScale = parseFloat(this.elements.zoomSlider.value);
        const newScale = Math.max(0.5, Math.min(3, currentScale + delta));
        this.elements.zoomSlider.value = newScale;
        this.setZoom(newScale);
    }
    
    setZoom(scale) {
        if (!this.imageData) return;
        
        this.imageData.scale = scale;
        this.elements.sourceImage.style.transform = `scale(${scale})`;
    }
    
    updatePreview() {
        if (!this.imageData) return;
        
        // 大预览
        const largePreview = this.generateAvatarPreview(200);
        this.elements.previewLarge.innerHTML = '';
        this.elements.previewLarge.appendChild(largePreview);
        
        // 小预览
        const smallPreview = this.generateAvatarPreview(50);
        this.elements.previewSmall.innerHTML = '';
        this.elements.previewSmall.appendChild(smallPreview);
    }
    
    generateAvatarPreview(size) {
        const canvas = document.createElement('canvas');
        const ctx = canvas.getContext('2d');
        
        canvas.width = size;
        canvas.height = size;
        
        // 创建圆形裁剪
        ctx.beginPath();
        ctx.arc(size / 2, size / 2, size / 2, 0, Math.PI * 2);
        ctx.closePath();
        ctx.clip();
        
        // 计算源图片的裁剪区域
        const img = this.elements.sourceImage;
        const rect = img.getBoundingClientRect();
        const containerRect = this.elements.imageContainer.getBoundingClientRect();
        
        const scaleX = this.imageData.width / rect.width;
        const scaleY = this.imageData.height / rect.height;
        
        const sourceX = (this.cropCircle.x - this.cropCircle.radius) * scaleX;
        const sourceY = (this.cropCircle.y - this.cropCircle.radius) * scaleY;
        const sourceSize = this.cropCircle.radius * 2 * scaleX;
        
        ctx.drawImage(
            img,
            sourceX, sourceY,
            sourceSize, sourceSize,
            0, 0,
            size, size
        );
        
        return canvas;
    }
    
    cropAvatar() {
        const preview = this.generateAvatarPreview(this.options.size);
        const dataUrl = preview.toDataURL('image/png', 1.0);
        
        // 创建下载链接
        const link = document.createElement('a');
        link.download = 'avatar.png';
        link.href = dataUrl;
        link.click();
        
        return dataUrl;
    }
}

                    let cropWidth, cropHeight, cropX, cropY;
                    
                    if (sourceRatio > targetRatio) {
                        // 源图片更宽，裁剪宽度
                        cropHeight = sourceHeight;
                        cropWidth = sourceHeight * targetRatio;
                        cropX = (sourceWidth - cropWidth) / 2;
                        cropY = 0;
                    } else {
                        // 源图片更高，裁剪高度
                        cropWidth = sourceWidth;
                        cropHeight = sourceWidth / targetRatio;
                        cropX = 0;
                        cropY = (sourceHeight - cropHeight) / 2;
                    }
                    
                    canvas.width = settings.width;
                    canvas.height = settings.height;
                    
                    ctx.drawImage(
                        img,
                        cropX, cropY,
                        cropWidth, cropHeight,
                        0, 0,
                        settings.width, settings.height
                    );
                    
                    const dataUrl = canvas.toDataURL(settings.format, settings.quality);
                    
                    resolve({
                        originalName: file.name,
                        dataUrl: dataUrl,
                        size: dataUrl.length,
                        format: settings.format
                    });
                };
                
                img.onerror = () => reject(new Error('图片加载失败'));
                img.src = e.target.result;
            };
            
            reader.onerror = () => reject(new Error('文件读取失败'));
            reader.readAsDataURL(file);
        });
    }
    
    addResultToGrid(result) {
        const resultItem = document.createElement('div');
        resultItem.className = 'result-item';
        
        resultItem.innerHTML = `
            <img src="${result.dataUrl}" alt="${result.originalName}">
            <div class="info">
                <div>${result.originalName}</div>
                <div>${this.formatFileSize(result.size)}</div>
            </div>
        `;
        
        this.elements.resultsGrid.appendChild(resultItem);
    }
    
    updateProgress(current, total) {
        const percentage = (current / total) * 100;
        this.elements.progressBar.style.width = `${percentage}%`;
        this.elements.progressText.textContent = `${current} / ${total}`;
    }
    
    downloadAll() {
        this.results.forEach((result, index) => {
            setTimeout(() => {
                const link = document.createElement('a');
                link.download = result.originalName;
                link.href = result.dataUrl;
                link.click();
            }, index * 200); // 间隔下载避免浏览器阻止
        });
    }
    
    async downloadAsZip() {
        // 需要JSZip库
        if (typeof JSZip === 'undefined') {
            alert('需要JSZip库支持');
            return;
        }
        
        const zip = new JSZip();
        
        this.results.forEach(result => {
            const base64Data = result.dataUrl.split(',')[1];
            zip.file(result.originalName, base64Data, { base64: true });
        });
        
        const content = await zip.generateAsync({ type: 'blob' });
        
        const link = document.createElement('a');
        link.download = 'cropped-images.zip';
        link.href = URL.createObjectURL(content);
        link.click();
    }
    
    formatFileSize(bytes) {
        if (bytes === 0) return '0 B';
        const k = 1024;
        const sizes = ['B', 'KB', 'MB', 'GB'];
        const i = Math.floor(Math.log(bytes) / Math.log(k));
        return Math.round((bytes / Math.pow(k, i)) * 100) / 100 + ' ' + sizes[i];
    }

// 移动端优化的图片裁剪
class MobileImageCropper {
    constructor(containerId, options = {}) {
        this.container = document.getElementById(containerId);
        this.options = {
            touchSensitivity: options.touchSensitivity || 1,
            enableGestures: options.enableGestures !== false,
            preventScrolling: options.preventScrolling !== false,
            ...options
        };
        
        this.touchState = {
            active: false,
            touches: [],
            lastDistance: 0
        };
        
        this.init();
    }
    
    setupTouchEvents() {
        const element = this.elements.cropBox;
        
        element.addEventListener('touchstart', this.handleTouchStart.bind(this), { passive: false });
        element.addEventListener('touchmove', this.handleTouchMove.bind(this), { passive: false });
        element.addEventListener('touchend', this.handleTouchEnd.bind(this));
        element.addEventListener('touchcancel', this.handleTouchEnd.bind(this));
    }
    
    handleTouchStart(e) {
        if (this.options.preventScrolling) {
            e.preventDefault();
        }
        
        this.touchState.active = true;
        this.touchState.touches = Array.from(e.touches);
        
        if (this.touchState.touches.length === 2 && this.options.enableGestures) {
            this.touchState.lastDistance = this.getTouchDistance(this.touchState.touches);
            this.handlePinchStart();
        }
    }
    
    handleTouchMove(e) {
        if (!this.touchState.active) return;
        
        if (this.options.preventScrolling) {
            e.preventDefault();
        }
        
        const currentTouches = Array.from(e.touches);
        
        if (currentTouches.length === 1) {
            // 单指拖拽
            const touch = currentTouches[0];
            const prevTouch = this.touchState.touches[0];
            const deltaX = touch.clientX - prevTouch.clientX;
            const deltaY = touch.clientY - prevTouch.clientY;
            
            this.moveCropBox(deltaX * this.options.touchSensitivity, deltaY * this.options.touchSensitivity);
        } else if (currentTouches.length === 2 && this.options.enableGestures) {
            // 双指缩放
            const currentDistance = this.getTouchDistance(currentTouches);
            const scale = currentDistance / this.touchState.lastDistance;
            
            this.handlePinchMove(scale);
            this.touchState.lastDistance = currentDistance;
        }
        
        this.touchState.touches = currentTouches;
    }
    
    handleTouchEnd(e) {
        if (e.touches.length === 0) {
            this.touchState.active = false;
            this.touchState.touches = [];
        }
    }
    
    getTouchDistance(touches) {
        const dx = touches[0].clientX - touches[1].clientX;
        const dy = touches[0].clientY - touches[1].clientY;
        return Math.sqrt(dx * dx + dy * dy);
    }
    
    handlePinchStart() {
        this.startCropBox = { ...this.cropBox };
    }
    
    handlePinchMove(scale) {
        if (!this.startCropBox) return;
        
        const newWidth = this.startCropBox.width * scale;
        const newHeight = this.startCropBox.height * scale;
        
        // 限制最小尺寸
        if (newWidth >= this.options.minWidth && newHeight >= this.options.minHeight) {
            this.cropBox.width = newWidth;
            this.cropBox.height = newHeight;
            
            // 保持中心位置
            const deltaX = (this.startCropBox.width - newWidth) / 2;
            const deltaY = (this.startCropBox.height - newHeight) / 2;
            
            this.cropBox.x = this.startCropBox.x + deltaX;
            this.cropBox.y = this.startCropBox.y + deltaY;
            
            this.updateCropBoxUI();
        }
    }
}
