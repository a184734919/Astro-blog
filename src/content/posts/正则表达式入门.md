## 概述

正则表达式（Regular Expression，简称 Regex）是一种强大的文本处理工具，用于匹配、查找、替换文本中的特定模式。它使用特殊的语法来描述文本模式，可以大大提高字符串处理的效率和准确性。本文将详细介绍正则表达式的基础语法、常用方法和实际应用。

## 核心概念

### 什么是正则表达式

正则表达式是一种描述文本模式的语法，它可以：
- 检查字符串是否符合特定格式
- 从文本中提取符合模式的内容
- 替换文本中的特定部分
- 验证用户输入的有效性

### 正则表达式的基本组成

1. **普通字符**：匹配自身，如 `abc` 匹配 "abc"
2. **元字符**：具有特殊含义的字符，如 `. * + ?` 等
3. **字符类**：匹配一组字符中的任意一个
4. **量词**：指定匹配次数
5. **定位符**：指定匹配位置

## 基本用法

### 字符匹配

```javascript
// 匹配普通字符
const pattern1 = /abc/;
console.log(pattern1.test('abc'));     // true
console.log(pattern1.test('abcd'));    // true
console.log(pattern1.test('ab'));      // false

// 匹配任意单个字符（除了换行符）
const pattern2 = /a.c/;
console.log(pattern2.test('abc'));     // true
console.log(pattern2.test('a*c'));     // true
console.log(pattern2.test('ac'));      // false

// 匹配数字
const pattern3 = /\d/;  // [0-9]
console.log(pattern3.test('123'));     // true
console.log(pattern3.test('abc'));     // false

// 匹配非数字
const pattern4 = /\D/;  // [^0-9]
console.log(pattern4.test('abc'));     // true
console.log(pattern4.test('123'));     // false

// 匹配字母、数字、下划线
const pattern5 = /\w/;  // [a-zA-Z0-9_]
console.log(pattern5.test('a1_'));     // true
console.log(pattern5.test('a-'));      // false

// 匹配非字母、数字、下划线
const pattern6 = /\W/;  // [^a-zA-Z0-9_]
console.log(pattern6.test('a-'));      // true
console.log(pattern6.test('a1_'));     // false

// 匹配空白字符
const pattern7 = /\s/;  // 空格、制表符、换行符等
console.log(pattern7.test('a b'));     // true
console.log(pattern7.test('ab'));      // false

// 匹配非空白字符
const pattern8 = /\S/;  // [^\s]
console.log(pattern8.test('a b'));     // true
console.log(pattern8.test('   '));     // false
```

### 字符类

```javascript
// 匹配字符类中的任意一个字符
const pattern1 = /[abc]/;
console.log(pattern1.test('a'));       // true
console.log(pattern1.test('b'));       // true
console.log(pattern1.test('c'));       // true
console.log(pattern1.test('d'));       // false

// 匹配字符范围
const pattern2 = /[a-z]/;  // 小写字母
console.log(pattern2.test('a'));       // true
console.log(pattern2.test('z'));       // true
console.log(pattern2.test('A'));       // false

const pattern3 = /[A-Z]/;  // 大写字母
console.log(pattern3.test('A'));       // true
console.log(pattern3.test('Z'));       // true
console.log(pattern3.test('a'));       // false

const pattern4 = /[0-9]/;  // 数字
console.log(pattern4.test('0'));       // true
console.log(pattern4.test('9'));       // true
console.log(pattern4.test('a'));       // false

// 组合字符类
const pattern5 = /[a-zA-Z0-9]/;
console.log(pattern5.test('a'));       // true
console.log(pattern5.test('Z'));       // true
console.log(pattern5.test('9'));       // true
console.log(pattern5.test('@'));       // false

// 反向字符类（匹配不在字符类中的字符）
const pattern6 = /[^abc]/;
console.log(pattern6.test('a'));       // false
console.log(pattern6.test('d'));       // true
console.log(pattern6.test('1'));       // true

// 预定义字符类
const patterns = {
  digit: /\d/,           // 数字 [0-9]
  nonDigit: /\D/,        // 非数字 [^0-9]
  word: /\w/,            // 单词字符 [a-zA-Z0-9_]
  nonWord: /\W/,         // 非单词字符 [^a-zA-Z0-9_]
  space: /\s/,           // 空白字符 [\t\n\f\r ]
  nonSpace: /\S/         // 非空白字符 [^\t\n\f\r ]
};

Object.entries(patterns).forEach(([name, pattern]) => {
  console.log(`${name}:`, pattern.test('a1_ @'));
});
```

### 量词

```javascript
// 匹配一次或多次
const pattern1 = /a+/;
console.log(pattern1.test('a'));       // true
console.log(pattern1.test('aa'));      // true
console.log(pattern1.test('aaa'));     // true
console.log(pattern1.test('b'));       // false

// 匹配零次或多次
const pattern2 = /a*/;
console.log(pattern2.test('a'));       // true
console.log(pattern2.test('aa'));      // true
console.log(pattern2.test('b'));       // true（匹配零个 a）
console.log(pattern2.test(''));        // true

// 匹配零次或一次
const pattern3 = /a?/;
console.log(pattern3.test('a'));       // true
console.log(pattern3.test('b'));       // true（匹配零个 a）
console.log(pattern3.test(''));        // true

// 精确匹配 n 次
const pattern4 = /a{3}/;
console.log(pattern4.test('aaa'));     // true
console.log(pattern4.test('aaaa'));    // true（包含 aaa）
console.log(pattern4.test('aa'));      // false

// 匹配至少 n 次
const pattern5 = /a{2,}/;
console.log(pattern5.test('aa'));      // true
console.log(pattern5.test('aaa'));     // true
console.log(pattern5.test('a'));       // false

// 匹配 n 到 m 次
const pattern6 = /a{2,4}/;
console.log(pattern6.test('aa'));      // true
console.log(pattern6.test('aaa'));     // true
console.log(pattern6.test('aaaa'));    // true
console.log(pattern6.test('a'));       // false
console.log(pattern6.test('aaaaa'));   // false（但会匹配前4个）

// 贪婪与非贪婪匹配
const greedy = /a.+/;      // 贪婪：尽可能多匹配
const lazy = /a.+?/;       // 非贪婪：尽可能少匹配

const text = 'abcabc';
console.log(text.match(greedy));   // ['abcabc']
console.log(text.match(lazy));     // ['abc']
```

### 定位符

```javascript
// 匹配字符串开头
const pattern1 = /^abc/;
console.log(pattern1.test('abc'));      // true
console.log(pattern1.test('abcd'));     // true
console.log(pattern1.test('xabc'));     // false

// 匹配字符串结尾
const pattern2 = /abc$/;
console.log(pattern2.test('abc'));      // true
console.log(pattern2.test('xabc'));     // true
console.log(pattern2.test('abcx'));     // false

// 匹配单词边界
const pattern3 = /\bword\b/;
console.log(pattern3.test('word'));     // true
console.log(pattern3.test('words'));    // false
console.log(pattern3.test('word!'));    // true

// 匹配非单词边界
const pattern4 = /\B/;
console.log(pattern4.test('word'));     // false
console.log(pattern4.test('wo'));       // true

// 组合使用定位符
const pattern5 = /^\d{3}-\d{2}-\d{4}$/;
console.log(pattern5.test('123-45-6789'));  // true（美国社保号格式）
console.log(pattern5.test('123-45-678'));   // false

// 多行模式
const text = `line1
line2
line3`;

const pattern6 = /^line/m;  // m 修饰符启用多行模式
const matches = text.match(pattern6);
console.log(matches);        // ['line', 'line', 'line']
```

### 选择和分组

```javascript
// 选择：匹配多个模式中的任意一个
const pattern1 = /apple|banana|orange/;
console.log(pattern1.test('apple'));    // true
console.log(pattern1.test('banana'));   // true
console.log(pattern1.test('orange'));   // true
console.log(pattern1.test('grape'));    // false

// 分组：将多个字符组合成一个整体
const pattern2 = /(ab)+/;
console.log(pattern2.test('ab'));       // true
console.log(pattern2.test('abab'));     // true
console.log(pattern2.test('a'));        // false

// 捕获组：记住匹配的内容
const pattern3 = /(\d{4})-(\d{2})-(\d{2})/;
const date = '2024-01-15';
const result = pattern3.exec(date);

console.log(result[0]);      // '2024-01-15' - 完整匹配
console.log(result[1]);      // '2024' - 第一个捕获组
console.log(result[2]);      // '01' - 第二个捕获组
console.log(result[3]);      // '15' - 第三个捕获组

// 替换时使用捕获组
const text = 'Hello World';
const replaced = text.replace(/(\w+) (\w+)/, '$2, $1');
console.log(replaced);       // 'World, Hello'

// 非捕获组：只分组不捕获
const pattern4 = /(?:abc|def)+/;
console.log(pattern4.test('abcabc'));  // true

// 分组的嵌套
const pattern5 = /((ab)+c)/;
const nested = 'ababc';
const nestedResult = pattern5.exec(nested);

console.log(nestedResult[1]);  // 'ababc' - 外层组
console.log(nestedResult[2]);  // 'ab' - 内层组
```

## 常用方法

### JavaScript 正则表达式方法

```javascript
// 1. test() - 测试字符串是否匹配模式
const pattern1 = /hello/;
console.log(pattern1.test('hello world'));  // true
console.log(pattern1.test('hi there'));     // false

// 2. exec() - 执行匹配搜索，返回匹配结果
const pattern2 = /hello/g;
const text1 = 'hello world hello';
let result;

while ((result = pattern2.exec(text1)) !== null) {
  console.log(result[0]);      // 匹配的文本
  console.log(result.index);   // 匹配的位置
}

// 3. match() - 返回所有匹配结果
const pattern3 = /hello/g;
const text2 = 'hello world hello';
const matches = text2.match(pattern3);
console.log(matches);          // ['hello', 'hello']

// 不使用 g 标志时返回详细信息
const detailed = text2.match(/hello/);
console.log(detailed);
// ['hello', index: 0, input: 'hello world hello', groups: undefined]

// 4. matchAll() - 返回所有匹配的迭代器（ES2020）
const pattern4 = /hello/g;
const allMatches = text2.matchAll(pattern4);
console.log([...allMatches]);

// 5. search() - 返回匹配位置的索引
const pattern5 = /world/;
const text3 = 'hello world';
console.log(text3.search(pattern5));     // 6

// 6. replace() - 替换匹配的内容
const pattern6 = /hello/g;
const text4 = 'hello world hello';
const replaced1 = text4.replace(pattern6, 'hi');
console.log(replaced1);                   // 'hi world hi'

// 使用回调函数进行替换
const pattern7 = /\d+/g;
const text5 = '价格：100元，数量：5件';
const replaced2 = text5.replace(pattern7, (match) => {
  return parseInt(match) * 2;
});
console.log(replaced2);                   // '价格：200元，数量：10件'

// 7. split() - 使用正则表达式分割字符串
const pattern8 = /[,，]/;  // 匹配中英文逗号
const text6 = 'apple,banana,橙子，grape';
const fruits = text6.split(pattern8);
console.log(fruits);                      // ['apple', 'banana', '橙子', 'grape']
```

### 实际应用示例

```javascript
// 1. 验证邮箱格式
function validateEmail(email) {
  const pattern = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  return pattern.test(email);
}

console.log(validateEmail('test@example.com'));    // true
console.log(validateEmail('invalid.email'));       // false
console.log(validateEmail('test@'));               // false

// 2. 验证手机号码
function validatePhone(phone) {
  const pattern = /^1[3-9]\d{9}$/;
  return pattern.test(phone);
}

console.log(validatePhone('13800138000'));  // true
console.log(validatePhone('12345678901'));  // false

// 3. 验证密码强度
function validatePassword(password) {
  // 至少8位，包含大小写字母和数字
  const pattern = /^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)[a-zA-Z\d]{8,}$/;
  return pattern.test(password);
}

console.log(validatePassword('Password123'));    // true
console.log(validatePassword('password123'));    // false
console.log(validatePassword('PASSWORD123'));    // false
console.log(validatePassword('Password'));       // false

// 4. 提取URL中的域名
function extractDomain(url) {
  const pattern = /https?:\/\/([^/]+)/;
  const match = url.match(pattern);
  return match ? match[1] : null;
}

console.log(extractDomain('https://www.example.com/path'));  // 'www.example.com'
console.log(extractDomain('http://example.com'));            // 'example.com'

// 5. 提取文本中的所有邮箱
function extractEmails(text) {
  const pattern = /[^\s@]+@[^\s@]+\.[^\s@]+/g;
  return text.match(pattern) || [];
}

const emailText = '联系我：test@example.com，或者 admin@company.com';
console.log(extractEmails(emailText));  // ['test@example.com', 'admin@company.com']

// 6. 格式化电话号码
function formatPhone(phone) {
  const pattern = /(\d{3})(\d{4})(\d{4})/;
  return phone.replace(pattern, '$1-$2-$3');
}

console.log(formatPhone('13800138000'));  // '138-0013-8000'

// 7. 移除字符串中的所有HTML标签
function removeHTML(html) {
  const pattern = /<[^>]*>/g;
  return html.replace(pattern, '');
}

const htmlText = '<div>Hello <span>World</span></div>';
console.log(removeHTML(htmlText));  // 'Hello World'

// 8. 检测字符串是否包含中文
function containsChinese(text) {
  const pattern = /[\u4e00-\u9fa5]/;
  return pattern.test(text);
}

console.log(containsChinese('Hello 世界'));  // true
console.log(containsChinese('Hello World')); // false

// 9. 去除字符串首尾空格
function trim(str) {
  const pattern = /^\s+|\s+$/g;
  return str.replace(pattern, '');
}

console.log(trim('  Hello World  '));  // 'Hello World'

// 10. 检测字符串是否为纯数字
function isNumber(str) {
  const pattern = /^\d+$/;
  return pattern.test(str);
}

console.log(isNumber('12345'));    // true
console.log(isNumber('123a5'));    // false
console.log(isNumber('12 345'));   // false
```

## 修饰符

### 常用修饰符

```javascript
// i - 忽略大小写
const pattern1 = /hello/i;
console.log(pattern1.test('Hello'));    // true
console.log(pattern1.test('HELLO'));    // true

// g - 全局匹配
const pattern2 = /hello/g;
const text1 = 'hello world hello';
console.log(text1.match(pattern2));     // ['hello', 'hello']

// m - 多行模式
const text2 = `line1 hello
line2 world
line3 hello`;

const pattern3 = /hello$/gm;
console.log(text2.match(pattern3));     // ['hello', 'hello']

// s - dotAll 模式（ES2018）- 让 . 匹配包括换行符的所有字符
const pattern4 = /hello.world/s;
const text3 = `hello
world`;
console.log(pattern4.test(text3));      // true

// u - Unicode 模式（ES2015）- 支持 Unicode 匹配
const pattern5 = /\u{1F600}/u;
console.log(pattern5.test('😀'));       // true

// y - sticky 模式（ES2015）- 从 lastIndex 开始匹配
const pattern6 = /hello/y;
pattern6.lastIndex = 6;
const text4 = 'world hello world';
console.log(pattern6.test(text4));      // true
```

### 修饰符的组合使用

```javascript
// 同时使用多个修饰符
const pattern1 = /hello/gi;
const text1 = 'Hello world HELLO';
console.log(text1.match(pattern1));     // ['Hello', 'HELLO']

// 全局多行匹配
const text2 = `Line1
Line2
Line3`;

const pattern2 = /^Line/gm;
console.log(text2.match(pattern2));     // ['Line', 'Line', 'Line']

// 实际应用：多行日志分析
const logText = `2024-01-15 ERROR: Database connection failed
2024-01-15 INFO: Server started
2024-01-15 ERROR: API request timeout
2024-01-15 INFO: Request processed`;

const errorPattern = /^2024-01-15 ERROR:.*$/gm;
const errors = logText.match(errorPattern);
console.log(errors);
// ['2024-01-15 ERROR: Database connection failed', '2024-01-15 ERROR: API request timeout']
```

### 1. 转义特殊字符

### 2. 性能考虑

```javascript
// 避免使用贪婪匹配
const badPattern = /<div>.*<\/div>/;  // 可能耗时很长
const goodPattern = /<div>.*?<\/div>/; // 使用非贪婪匹配

// 避免嵌套量词
const badPattern2 = /^(a+)+$/;  // 可能导致灾难性回溯
const goodPattern2 = /^a+$/;    // 简化的模式

// 使用锚点提高匹配效率
const pattern3 = /^hello$/;  // 明确指定开始和结束

const EMAIL_PATTERN = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;

function validateEmails(emails) {
  return emails.filter(email => EMAIL_PATTERN.test(email));
}
```

## 最佳实践

### 1. 命名和注释

```javascript
// 使用有意义的变量名
const EMAIL_PATTERN = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
const PHONE_PATTERN = /^1[3-9]\d{9}$/;
const DATE_PATTERN = /^\d{4}-\d{2}-\d{2}$/;

// 为复杂的正则表达式添加注释
const COMPLEX_PATTERN = new RegExp(
  '^' +                           // 字符串开始
  '(?=.*[a-z])' +                 // 至少一个小写字母
  '(?=.*[A-Z])' +                 // 至少一个大写字母
  '(?=.*\\d)' +                   // 至少一个数字
  '(?=.*[!@#$%^&*])' +           // 至少一个特殊字符
  '[a-zA-Z\\d!@#$%^&*]{8,}' +    // 8位以上，包含允许的字符
  '$',                            // 字符串结束
  'g'                             // 全局匹配
);

// 或者使用扩展语法（某些引擎支持）
const COMMENTED_PATTERN = `
  ^                    # 字符串开始
  (?=.*[a-z])          # 至少一个小写字母
  (?=.*[A-Z])          # 至少一个大写字母
  (?=.*\\d)            # 至少一个数字
  .{8,}                # 至少8个字符
  $                    # 字符串结束
`;
```

```javascript
console.log(DATETIME_PATTERN.test('2024-01-15 10:30:00'));  // true
```

```javascript
function testPattern(pattern, tests) {
```

### 4. 错误处理

```javascript
// 处理正则表达式语法错误
try {
  const badPattern = new RegExp('(');  // 不完整的正则表达式
} catch (error) {
  console.error('正则表达式语法错误:', error.message);
}

// 处理匹配失败的情况
function safeMatch(pattern, text) {
  try {
    const result = pattern.exec(text);
    return result || null;
  } catch (error) {
    console.error('正则匹配错误:', error.message);
    return null;
  }
}

const pattern = /(\d{4})-(\d{2})-(\d{2})/;
const result = safeMatch(pattern, '2024-01-15');
if (result) {
  console.log('匹配成功:', result[0]);
} else {
  console.log('匹配失败');
}
```
