## 概述

前端缓存是指将资源存储在客户端或中间代理服务器上，以减少网络请求、提高加载速度、降低服务器压力的技术手段。合理的缓存策略能够显著提升应用性能和用户体验。

## 核心概念

### 缓存类型分类

**浏览器缓存:**
- 强缓存
- 协商缓存
- 内存缓存
- 磁盘缓存

**HTTP 缓存头:**
- Cache-Control
- Expires
- ETag
- Last-Modified
- Vary

**本地存储缓存:**
- localStorage
- sessionStorage
- IndexedDB
- Web SQL (已废弃)

**应用级缓存:**
- Service Worker
- Cache API
- PWA 缓存

**CDN 缓存:**
- 边缘节点缓存
- 回源策略
- 缓存预热

### 缓存策略决策树

```javascript
// 缓存策略选择器
class CacheStrategySelector {
  constructor() {
    this.strategies = {
      // 静态资源：长期强缓存
      static: {
        cacheControl: 'public, max-age=31536000, immutable',
        expires: this.getOneYearLater()
      },
      
      // HTML 文件：不缓存或短期缓存
      html: {
        cacheControl: 'no-cache, no-store, must-revalidate',
        expires: new Date(0)
      },
      
      // API 响应：协商缓存
      api: {
        cacheControl: 'public, max-age=60, s-maxage=300',
        etag: true,
        lastModified: true
      },
      
      // 用户数据：个性化缓存
      user: {
        cacheControl: 'private, max-age=300',
        vary: ['User-Agent', 'Accept-Language']
      },
      
      // 动态内容：不缓存
      dynamic: {
        cacheControl: 'no-cache, no-store, must-revalidate'
      }
    };
  }
  
  selectStrategy(resourceType, context = {}) {
    // 根据资源类型选择缓存策略
    let strategy = this.strategies[resourceType];
    
    // 根据上下文调整策略
    if (context.isAuthRequired) {
      strategy = {
        cacheControl: 'private, max-age=0',
        vary: ['Authorization', 'Cookie']
      };
    }
    
    if (context.isRealtime) {
      strategy = {
        cacheControl: 'no-cache, no-store, must-revalidate'
      };
    }
    
    return strategy;
  }
  
  getOneYearLater() {
    const date = new Date();
    date.setFullYear(date.getFullYear() + 1);
    return date.toUTCString();
  }
}
```

## 基本用法

### 1. HTTP 缓存配置

```javascript
// HTTP 缓存配置工具
class HttpCacheManager {
  constructor() {
    this.cacheStrategySelector = new CacheStrategySelector();
  }
  
  // 设置响应缓存头
  setCacheHeaders(response, resourceType, context = {}) {
    const strategy = this.cacheStrategySelector.selectStrategy(resourceType, context);
    
    // 设置 Cache-Control
    if (strategy.cacheControl) {
      response.setHeader('Cache-Control', strategy.cacheControl);
    }
    
    // 设置 Expires
    if (strategy.expires) {
      response.setHeader('Expires', strategy.expires);
    }
    
    // 设置 ETag
    if (strategy.etag) {
      const etag = this.generateETag(response.body);
      response.setHeader('ETag', etag);
    }
    
    // 设置 Last-Modified
    if (strategy.lastModified) {
      response.setHeader('Last-Modified', new Date().toUTCString());
    }
    
    // 设置 Vary
    if (strategy.vary) {
      response.setHeader('Vary', strategy.vary.join(', '));
    }
  }
  
  // 生成 ETag
  generateETag(content) {
    const crypto = require('crypto');
    const hash = crypto.createHash('md5').update(content).digest('hex');
    return `"${hash}"`;
  }
  
  // 检查缓存有效性
  validateCache(request, cachedResponse) {
    const etag = request.headers['if-none-match'];
    const lastModified = request.headers['if-modified-since'];
    
    // 检查 ETag
    if (etag && cachedResponse.etag === etag) {
      return { valid: true, status: 304 };
    }
    
    // 检查 Last-Modified
    if (lastModified) {
      const cachedLastModified = new Date(cachedResponse.lastModified);
      const requestLastModified = new Date(lastModified);
      
      if (cachedLastModified <= requestLastModified) {
        return { valid: true, status: 304 };
      }
    }
    
    return { valid: false, status: 200 };
  }
}

// Node.js Express 中间件示例
const express = require('express');
const cacheManager = new HttpCacheManager();

const app = express();

// 静态资源缓存
app.use(express.static('public', {
  maxAge: '1y',
  immutable: true,
  setHeaders: (res, path) => {
    if (path.endsWith('.html')) {
      res.setHeader('Cache-Control', 'no-cache');
    }
  }
}));

// API 响应缓存
app.get('/api/data', (req, res) => {
  // 检查缓存
  const ifNoneMatch = req.get('If-None-Match');
  const cachedData = getCachedData('/api/data');
  
  if (cachedData && ifNoneMatch === cachedData.etag) {
    return res.status(304).end();
  }
  
  // 生成响应
  const data = generateData();
  const etag = cacheManager.generateETag(JSON.stringify(data));
  
  // 设置缓存头
  cacheManager.setCacheHeaders(res, 'api');
  res.setHeader('ETag', etag);
  
  res.json(data);
});

// HTML 文件不缓存
app.get('*', (req, res) => {
  res.setHeader('Cache-Control', 'no-cache, no-store, must-revalidate');
  res.sendFile('index.html');
});
```

### 2. 浏览器缓存管理

```javascript
// 浏览器缓存管理工具
class BrowserCacheManager {
  constructor() {
    this.memoryCache = new Map();
    this.diskCache = new IndexedDBCache('browser-cache');
    this.maxMemorySize = 50 * 1024 * 1024; // 50MB
    this.currentMemorySize = 0;
  }
  
  // 存储数据到缓存
  async set(key, value, options = {}) {
    const {
      ttl = 3600000, // 1小时
      strategy = 'auto', // auto, memory, disk
      priority = 'normal' // high, normal, low
    } = options;
    
    const cacheItem = {
      key,
      value,
      expires: Date.now() + ttl,
      size: this.calculateSize(value),
      priority,
      createdAt: Date.now(),
      accessedAt: Date.now(),
      accessCount: 0
    };
    
    // 选择缓存策略
    if (strategy === 'auto') {
      strategy = this.selectStrategy(cacheItem);
    }
    
    if (strategy === 'memory') {
      this.setMemoryCache(key, cacheItem);
    } else if (strategy === 'disk') {
      await this.setDiskCache(key, cacheItem);
    } else {
      // 混合策略：内存+磁盘
      if (this.canFitInMemory(cacheItem)) {
        this.setMemoryCache(key, cacheItem);
      } else {
        await this.setDiskCache(key, cacheItem);
      }
    }
  }
  
  // 从缓存获取数据
  async get(key) {
    // 先尝试内存缓存
    let item = this.getMemoryCache(key);
    if (item) {
      return item.value;
    }
    
    // 再尝试磁盘缓存
    item = await this.getDiskCache(key);
    if (item) {
      // 将热点数据提升到内存缓存
      if (item.accessCount > 5) {
        this.setMemoryCache(key, item);
      }
      return item.value;
    }
    
    return null;
  }
  
  // 内存缓存操作
  setMemoryCache(key, item) {
    // 检查是否有足够空间
    if (!this.canFitInMemory(item)) {
      this.evictFromMemory(item.size);
    }
    
    this.memoryCache.set(key, item);
    this.currentMemorySize += item.size;
  }
  
  getMemoryCache(key) {
    const item = this.memoryCache.get(key);
    
    if (!item) return null;
    
    // 检查是否过期
    if (item.expires < Date.now()) {
      this.memoryCache.delete(key);
      this.currentMemorySize -= item.size;
      return null;
    }
    
    // 更新访问信息
    item.accessedAt = Date.now();
    item.accessCount++;
    
    return item;
  }
  
  // 磁盘缓存操作
  async setDiskCache(key, item) {
    await this.diskCache.set(key, item);
  }
  
  async getDiskCache(key) {
    const item = await this.diskCache.get(key);
    
    if (!item) return null;
    
    // 检查是否过期
    if (item.expires < Date.now()) {
      await this.diskCache.delete(key);
      return null;
    }
    
    // 更新访问信息
    item.accessedAt = Date.now();
    item.accessCount++;
    await this.diskCache.set(key, item);
    
    return item;
  }
  
  // 缓存淘汰策略
  evictFromMemory(requiredSize) {
    const entries = Array.from(this.memoryCache.entries());
    
    // 按优先级和访问时间排序
    entries.sort((a, b) => {
      const priorityOrder = { low: 0, normal: 1, high: 2 };
      const priorityDiff = priorityOrder[b[1].priority] - priorityOrder[a[1].priority];
      
      if (priorityDiff !== 0) return priorityDiff;
      
      return a[1].accessedAt - b[1].accessedAt;
    });
    
    // 淘汰低优先级和长时间未访问的数据
    let freedSize = 0;
    for (const [key, item] of entries) {
      if (freedSize >= requiredSize) break;
      
      this.memoryCache.delete(key);
      this.currentMemorySize -= item.size;
      freedSize += item.size;
    }
  }
  
  // 缓存策略选择
  selectStrategy(item) {
    // 小文件、高优先级使用内存缓存
    if (item.size < 100 * 1024 && item.priority === 'high') {
      return 'memory';
    }
    
    // 大文件使用磁盘缓存
    if (item.size > 1024 * 1024) {
      return 'disk';
    }
    
    // 默认使用混合策略
    return 'hybrid';
  }
  
  canFitInMemory(item) {
    return (this.currentMemorySize + item.size) <= this.maxMemorySize;
  }
  
  calculateSize(value) {
    // 简单估算大小
    return JSON.stringify(value).length * 2; // 假设每个字符2字节
  }
  
  // 清理过期缓存
  async cleanExpired() {
    const now = Date.now();
    
    // 清理内存缓存
    for (const [key, item] of this.memoryCache.entries()) {
      if (item.expires < now) {
        this.memoryCache.delete(key);
        this.currentMemorySize -= item.size;
      }
    }
    
    // 清理磁盘缓存
    await this.diskCache.cleanExpired();
  }
  
  // 获取缓存统计信息
  getStats() {
    return {
      memory: {
        count: this.memoryCache.size,
        size: this.currentMemorySize,
        maxSize: this.maxMemorySize,
        utilization: (this.currentMemorySize / this.maxMemorySize * 100).toFixed(2) + '%'
      },
      disk: this.diskCache.getStats()
    };
  }
}

// IndexedDB 缓存实现
class IndexedDBCache {
  constructor(dbName, storeName = 'cache') {
    this.dbName = dbName;
    this.storeName = storeName;
    this.db = null;
  }
  
  async init() {
    return new Promise((resolve, reject) => {
      const request = indexedDB.open(this.dbName, 1);
      
      request.onerror = () => reject(request.error);
      request.onsuccess = () => {
        this.db = request.result;
        resolve();
      };
      
      request.onupgradeneeded = (event) => {
        const db = event.target.result;
        
        if (!db.objectStoreNames.contains(this.storeName)) {
          const store = db.createObjectStore(this.storeName, { keyPath: 'key' });
          store.createIndex('expires', 'expires', { unique: false });
        }
      };
    });
  }
  
  async set(key, value) {
    await this.init();
    
    return new Promise((resolve, reject) => {
      const transaction = this.db.transaction([this.storeName], 'readwrite');
      const store = transaction.objectStore(this.storeName);
      
      const request = store.put(value);
      
      request.onsuccess = () => resolve();
      request.onerror = () => reject(request.error);
    });
  }
  
  async get(key) {
    await this.init();
    
    return new Promise((resolve, reject) => {
      const transaction = this.db.transaction([this.storeName], 'readonly');
      const store = transaction.objectStore(this.storeName);
      
      const request = store.get(key);
      
      request.onsuccess = () => resolve(request.result);
      request.onerror = () => reject(request.error);
    });
  }
  
  async delete(key) {
    await this.init();
    
    return new Promise((resolve, reject) => {
      const transaction = this.db.transaction([this.storeName], 'readwrite');
      const store = transaction.objectStore(this.storeName);
      
      const request = store.delete(key);
      
      request.onsuccess = () => resolve();
      request.onerror = () => reject(request.error);
    });
  }
  
  async cleanExpired() {
    await this.init();
    
    return new Promise((resolve, reject) => {
      const transaction = this.db.transaction([this.storeName], 'readwrite');
      const store = transaction.objectStore(this.storeName);
      const index = store.index('expires');
      
      const request = index.openCursor(IDBKeyRange.upperBound(Date.now()));
      
      request.onsuccess = (event) => {
        const cursor = event.target.result;
        
        if (cursor) {
          cursor.delete();
          cursor.continue();
        } else {
          resolve();
        }
      };
      
      request.onerror = () => reject(request.error);
    });
  }
  
  getStats() {
    // 返回磁盘缓存统计信息
    return {
      storeName: this.storeName,
      // 这里可以添加更多统计信息
    };
  }
}
```

### 3. Service Worker 缓存

```javascript
// Service Worker 缓存策略
class ServiceWorkerCache {
  constructor(cacheName) {
    this.cacheName = cacheName;
    this.cache = null;
  }
  
  async init() {
    this.cache = await caches.open(this.cacheName);
  }
  
  // 缓存优先策略
  async cacheFirst(request) {
    const cachedResponse = await this.cache.match(request);
    
    if (cachedResponse) {
      return cachedResponse;
    }
    
    const networkResponse = await fetch(request);
    
    // 只缓存成功的 GET 请求
    if (networkResponse.ok && request.method === 'GET') {
      this.cache.put(request, networkResponse.clone());
    }
    
    return networkResponse;
  }
  
  // 网络优先策略
  async networkFirst(request) {
    try {
      const networkResponse = await fetch(request);
      
      if (networkResponse.ok && request.method === 'GET') {
        this.cache.put(request, networkResponse.clone());
      }
      
      return networkResponse;
    } catch (error) {
      const cachedResponse = await this.cache.match(request);
      
      if (cachedResponse) {
        return cachedResponse;
      }
      
      throw error;
    }
  }
  
  // 陈旧时重新验证策略
  async staleWhileRevalidate(request) {
    const cachedResponse = await this.cache.match(request);
    
    // 异步更新缓存
    const fetchPromise = fetch(request).then(networkResponse => {
      if (networkResponse.ok && request.method === 'GET') {
        this.cache.put(request, networkResponse.clone());
      }
      return networkResponse;
    });
    
    // 返回缓存响应，如果没有缓存则等待网络响应
    return cachedResponse || fetchPromise;
  }
  
  // 仅网络策略
  async networkOnly(request) {
    return fetch(request);
  }
  
  // 仅缓存策略
  async cacheOnly(request) {
    const cachedResponse = await this.cache.match(request);
    
    if (cachedResponse) {
      return cachedResponse;
    }
    
    throw new Error('No cached response available');
  }
  
  // 预缓存资源
  async precache(urls) {
    await this.init();
    
    const cachePromises = urls.map(url => {
      return fetch(url).then(response => {
        if (response.ok) {
          const request = new Request(url);
          return this.cache.put(request, response);
        }
      });
    });
    
    await Promise.all(cachePromises);
  }
  
  // 清理旧缓存
  async cleanup(expectedCacheNames) {
    const cacheNames = await caches.keys();
    
    const deletePromises = cacheNames.map(cacheName => {
      if (!expectedCacheNames.includes(cacheName)) {
        return caches.delete(cacheName);
      }
    });
    
    await Promise.all(deletePromises);
  }
}

// Service Worker 安装文件
const CACHE_VERSION = 'v1';
const CACHE_NAME = `my-app-cache-${CACHE_VERSION}`;
const swCache = new ServiceWorkerCache(CACHE_NAME);

// 需要预缓存的资源
const PRECACHE_URLS = [
  '/',
  '/index.html',
  '/styles/main.css',
  '/scripts/main.js',
  '/images/logo.png'
];

// 安装事件
self.addEventListener('install', (event) => {
  event.waitUntil(
    swCache.precache(PRECACHE_URLS).then(() => {
      return self.skipWaiting();
    })
  );
});

// 激活事件
self.addEventListener('activate', (event) => {
  event.waitUntil(
    swCache.cleanup([CACHE_NAME]).then(() => {
      return self.clients.claim();
    })
  );
});

// 拦截网络请求
self.addEventListener('fetch', (event) => {
  const request = event.request;
  const url = new URL(request.url);
  
  // 根据请求类型选择缓存策略
  if (request.method !== 'GET') {
    return; // 非 GET 请求不缓存
  }
  
  // 静态资源使用缓存优先
  if (url.pathname.match(/\.(js|css|png|jpg|jpeg|gif|webp|svg|woff|woff2)$/)) {
    event.respondWith(swCache.cacheFirst(request));
    return;
  }
  
  // API 请求使用网络优先
  if (url.pathname.startsWith('/api/')) {
    event.respondWith(swCache.networkFirst(request));
    return;
  }
  
  // HTML 文件使用陈旧时重新验证
  if (url.pathname.endsWith('.html') || url.pathname === '/') {
    event.respondWith(swCache.staleWhileRevalidate(request));
    return;
  }
  
  // 默认使用网络优先
  event.respondWith(swCache.networkFirst(request));
});
```

### 4. API 响应缓存

```javascript
// API 响应缓存装饰器
class ApiCacheDecorator {
  constructor(cacheManager, options = {}) {
    this.cacheManager = cacheManager;
    this.options = {
      ttl: 300000, // 5分钟
      keyGenerator: this.defaultKeyGenerator,
      shouldCache: this.defaultShouldCache,
      ...options
    };
  }
  
  // 装饰 API 方法
  decorate(apiMethod) {
    return async (...args) => {
      const cacheKey = this.options.keyGenerator(apiMethod.name, args);
      
      // 检查是否应该缓存
      if (!this.options.shouldCache(args)) {
        return await apiMethod(...args);
      }
      
      // 尝试从缓存获取
      const cachedResult = await this.cacheManager.get(cacheKey);
      
      if (cachedResult) {
        return cachedResult;
      }
      
      // 执行 API 调用
      const result = await apiMethod(...args);
      
      // 缓存结果
      await this.cacheManager.set(cacheKey, result, {
        ttl: this.options.ttl
      });
      
      return result;
    };
  }
  
  // 默认的缓存键生成器
  defaultKeyGenerator(methodName, args) {
    return `api:${methodName}:${JSON.stringify(args)}`;
  }
  
  // 默认的缓存条件检查
  defaultShouldCache(args) {
    // 只缓存 GET 请求
    return true;
  }
  
  // 创建缓存清理器
  createCacheCleaner(interval = 60000) {
    return setInterval(() => {
      this.cacheManager.cleanExpired();
    }, interval);
  }
}

// 使用示例
class UserService {
  constructor(cacheManager) {
    this.cacheDecorator = new ApiCacheDecorator(cacheManager, {
      ttl: 300000, // 5分钟
      keyGenerator: (methodName, args) => {
        if (methodName === 'getUser') {
          return `user:${args[0]}`; // userId
        }
        return `api:${methodName}:${JSON.stringify(args)}`;
      }
    });
    
    // 装饰方法
    this.getUser = this.cacheDecorator.decorate(this._getUser.bind(this));
    this.getUserList = this.cacheDecorator.decorate(this._getUserList.bind(this));
  }
  
  async _getUser(userId) {
    const response = await fetch(`/api/users/${userId}`);
    return response.json();
  }
  
  async _getUserList(filters = {}) {
    const response = await fetch(`/api/users?${new URLSearchParams(filters)}`);
    return response.json();
  }
}

// React Hook 版本的 API 缓存
function useApiCache(apiCall, cacheKey, options = {}) {
  const [data, setData] = React.useState(null);
  const [loading, setLoading] = React.useState(false);
  const [error, setError] = React.useState(null);
  const cacheManager = React.useMemo(() => new BrowserCacheManager(), []);
  
  const fetchData = React.useCallback(async () => {
    setLoading(true);
    setError(null);
    
    try {
      // 尝试从缓存获取
      const cachedData = await cacheManager.get(cacheKey);
      
      if (cachedData) {
        setData(cachedData);
        setLoading(false);
        return;
      }
      
      // 执行 API 调用
      const result = await apiCall();
      
      // 缓存结果
      await cacheManager.set(cacheKey, result, {
        ttl: options.ttl || 300000
      });
      
      setData(result);
    } catch (err) {
      setError(err);
    } finally {
      setLoading(false);
    }
  }, [apiCall, cacheKey, options.ttl, cacheManager]);
  
  React.useEffect(() => {
    fetchData();
  }, [fetchData]);
  
  // 手动刷新
  const refresh = React.useCallback(() => {
    cacheManager.cache?.delete(cacheKey);
    fetchData();
  }, [cacheKey, fetchData, cacheManager]);
  
  return { data, loading, error, refresh };
}
```

```javascript
// 综合缓存系统
class ComprehensiveCacheSystem {
  constructor(config = {}) {
    this.config = {
      httpCache: true,
      browserCache: true,
      serviceWorkerCache: true,
      apiCache: true,
      defaultTTL: 300000,
      ...config
    };
    
    // 初始化各个缓存模块
    this.httpCacheManager = new HttpCacheManager();
    this.browserCacheManager = new BrowserCacheManager();
    this.apiCacheDecorator = new ApiCacheDecorator(this.browserCacheManager);
    
    if (this.config.serviceWorkerCache && 'serviceWorker' in navigator) {
      this.initServiceWorker();
    }
    
    this.initCacheCleanup();
  }
  
  async initServiceWorker() {
    try {
      const registration = await navigator.serviceWorker.register('/sw.js');
      console.log('Service Worker 注册成功:', registration);
      
      // 监听 Service Worker 消息
      navigator.serviceWorker.addEventListener('message', (event) => {
        if (event.data.type === 'cache-update') {
          this.handleCacheUpdate(event.data);
        }
      });
    } catch (error) {
      console.error('Service Worker 注册失败:', error);
    }
  }
  
  initCacheCleanup() {
    // 定期清理过期缓存
    setInterval(() => {
      this.cleanupExpiredCache();
    }, 60000); // 每分钟清理一次
    
    // 页面卸载时清理
    window.addEventListener('beforeunload', () => {
      this.cleanupExpiredCache();
    });
  }
  
  async cleanupExpiredCache() {
    await this.browserCacheManager.cleanExpired();
  }
  
  // HTTP 响应缓存
  cacheHttpResponse(response, resourceType) {
    this.httpCacheManager.setCacheHeaders(response, resourceType);
  }
  
  // 浏览器缓存操作
  async setBrowserCache(key, value, options) {
    return await this.browserCacheManager.set(key, value, options);
  }
  
  async getBrowserCache(key) {
    return await this.browserCacheManager.get(key);
  }
  
  // API 调用缓存装饰
  cacheApiCall(apiMethod) {
    return this.apiCacheDecorator.decorate(apiMethod);
  }
  
  // 缓存预热
  async warmupCache(resources) {
    const promises = resources.map(async (resource) => {
      try {
        if (resource.type === 'api') {
          const result = await resource.fetch();
          await this.setBrowserCache(resource.key, result, {
            ttl: resource.ttl || this.config.defaultTTL
          });
        } else if (resource.type === 'http') {
          await fetch(resource.url);
        }
      } catch (error) {
        console.error('缓存预热失败:', resource, error);
      }
    });
    
    await Promise.all(promises);
  }
  
  // 缓存失效
  async invalidateCache(pattern) {
    // 使匹配模式的缓存失效
    if (pattern.startsWith('api:')) {
      // API 缓存失效
      const allKeys = await this.getAllCacheKeys();
      const matchingKeys = allKeys.filter(key => key.startsWith(pattern));
      
      for (const key of matchingKeys) {
        await this.browserCacheManager.memoryCache.delete(key);
        await this.browserCacheManager.diskCache.delete(key);
      }
    } else if (pattern.startsWith('http:')) {
      // HTTP 缓存失效（通过更新版本号）
      const url = pattern.replace('http:', '');
      this.updateResourceVersion(url);
    }
  }
  
  async getAllCacheKeys() {
    // 获取所有缓存键
    const memoryKeys = Array.from(this.browserCacheManager.memoryCache.keys());
    
    // 这里需要实现获取所有磁盘缓存键的方法
    const diskKeys = await this.getAllDiskCacheKeys();
    
    return [...memoryKeys, ...diskKeys];
  }
  
  async getAllDiskCacheKeys() {
    // 实现获取所有磁盘缓存键
    await this.browserCacheManager.diskCache.init();
    
    return new Promise((resolve, reject) => {
      const transaction = this.browserCacheManager.diskCache.db.transaction(
        [this.browserCacheManager.diskCache.storeName], 
        'readonly'
      );
      const store = transaction.objectStore(this.browserCacheManager.diskCache.storeName);
      
      const request = store.getAllKeys();
      
      request.onsuccess = () => {
        resolve(request.result);
      };
      
      request.onerror = () => reject(request.error);
    });
  }
  
  updateResourceVersion(url) {
    // 通过更新 URL 中的版本号来使缓存失效
    const version = Date.now();
    const newUrl = url.includes('?') 
      ? `${url}&v=${version}`
      : `${url}?v=${version}`;
    
    return newUrl;
  }
  
  // 获取缓存统计信息
  getCacheStats() {
    return {
      browser: this.browserCacheManager.getStats(),
      serviceWorker: this.getServiceWorkerStats(),
      timestamp: Date.now()
    };
  }
  
  getServiceWorkerStats() {
    if (!('serviceWorker' in navigator) || !navigator.serviceWorker.controller) {
      return { enabled: false };
    }
    
    return {
      enabled: true,
      ready: true
      // 可以添加更多统计信息
    };
  }
  
  // 处理缓存更新
  handleCacheUpdate(data) {
    console.log('缓存更新:', data);

  }
```

```javascript
// 缓存一致性管理
class CacheConsistencyManager {
  constructor(cacheSystem) {
    this.cacheSystem = cacheSystem;
    this.subscribers = new Map();
  }
  
  // 订阅缓存变化
  subscribe(pattern, callback) {
    if (!this.subscribers.has(pattern)) {
      this.subscribers.set(pattern, new Set());
    }
    
    this.subscribers.get(pattern).add(callback);
    
    // 返回取消订阅函数
    return () => {
      this.subscribers.get(pattern).delete(callback);
      if (this.subscribers.get(pattern).size === 0) {
        this.subscribers.delete(pattern);
      }
    };
  }
  
  // 通知缓存变化
  notify(pattern, data) {
    for (const [keyPattern, callbacks] of this.subscribers.entries()) {
      if (this.matchesPattern(pattern, keyPattern)) {
        callbacks.forEach(callback => callback(data));
      }
    }
  }
  
  matchesPattern(pattern, keyPattern) {
    // 简单的模式匹配
    if (keyPattern.endsWith('*')) {
      const prefix = keyPattern.slice(0, -1);
      return pattern.startsWith(prefix);
    }
    
    return pattern === keyPattern;
  }
  
  // 主动使缓存失效
  async invalidate(pattern) {
    await this.cacheSystem.invalidateCache(pattern);
    this.notify(pattern, { type: 'invalidate', pattern });
  }
  
  // 缓存更新时保持一致性
  async updateWithConsistency(key, updater) {
    // 获取当前值
    const currentValue = await this.cacheSystem.getBrowserCache(key);
    
    // 执行更新
    const newValue = await updater(currentValue);
    
    // 更新缓存
    await this.cacheSystem.setBrowserCache(key, newValue);
    
    // 通知订阅者
    this.notify(key, { type: 'update', key, value: newValue });
    
    return newValue;
  }
}
```

```javascript
    this.metrics.totalAccessTime += duration;
    this.metrics.averageAccessTime = this.metrics.totalAccessTime / this.metrics.accessCount;
    
    this.operationTimings.push({
      duration,
      timestamp: Date.now()
    });

      this.operationTimings.shift();
    }

      ...this.metrics,
```

```javascript
// 安全的缓存管理
class SecureCacheManager {
  constructor(baseCacheManager) {
    this.baseCacheManager = baseCacheManager;
    this.encryptionKey = null;
    this.init();
  }
  
  async init() {
    // 生成或获取加密密钥
    this.encryptionKey = await this.getEncryptionKey();
  }
  
  async getEncryptionKey() {
    // 从安全存储获取或生成密钥
    let key = localStorage.getItem('cache_encryption_key');
    
    if (!key) {
      key = this.generateKey();
      localStorage.setItem('cache_encryption_key', key);
    }
    
    return key;
  }
  
  generateKey() {
    // 生成随机密钥
    const array = new Uint8Array(32);
    crypto.getRandomValues(array);
    return Array.from(array, byte => byte.toString(16).padStart(2, '0')).join('');
  }
  
  async encrypt(data) {
    // 加密数据
    const jsonString = JSON.stringify(data);
    const encoder = new TextEncoder();
    const dataBytes = encoder.encode(jsonString);
    
    // 使用 Web Crypto API 进行加密
    const key = await crypto.subtle.importKey(
      'raw',
      new TextEncoder().encode(this.encryptionKey),
      { name: 'AES-GCM' },
      false,
      ['encrypt']
    );
    
    const iv = crypto.getRandomValues(new Uint8Array(12));
    const encryptedData = await crypto.subtle.encrypt(
      { name: 'AES-GCM', iv },
      key,
      dataBytes
    );
    
    return {
      iv: Array.from(iv),
      data: Array.from(new Uint8Array(encryptedData))
    };
  }
  
  async decrypt(encryptedData) {
    // 解密数据
    const key = await crypto.subtle.importKey(
      'raw',
      new TextEncoder().encode(this.encryptionKey),
      { name: 'AES-GCM' },
      false,
      ['decrypt']
    );
    
    const decryptedData = await crypto.subtle.decrypt(
      { name: 'AES-GCM', iv: new Uint8Array(encryptedData.iv) },
      key,
      new Uint8Array(encryptedData.data)
    );
    
    const decoder = new TextDecoder();
    const jsonString = decoder.decode(decryptedData);
    
    return JSON.parse(jsonString);
  }
  
  async set(key, value, options) {
    // 加密敏感数据
    if (this.isSensitiveData(key, value)) {
      value = await this.encrypt(value);
    }
    
    return await this.baseCacheManager.set(key, value, options);
  }
  
  async get(key) {
    const value = await this.baseCacheManager.get(key);
    
    if (!value) return null;
    
    // 解密数据
    if (this.isEncrypted(value)) {
      return await this.decrypt(value);
    }
    
    return value;
  }
  
  isSensitiveData(key, value) {
    // 判断是否为敏感数据
    const sensitiveKeys = ['user', 'auth', 'token', 'password'];
    return sensitiveKeys.some(sensitiveKey => 
      key.toLowerCase().includes(sensitiveKey)
    );
  }
  
  isEncrypted(value) {
    // 判断数据是否已加密
    return value && typeof value === 'object' && value.iv && value.data;
  }
}
```

## 总结

前端缓存是提升应用性能的重要手段，关键点包括：

1. **多层级缓存**: HTTP 缓存、浏览器缓存、Service Worker 缓存等多层缓存策略
2. **合理策略**: 根据资源类型选择合适的缓存策略
3. **性能监控**: 监控缓存命中率、访问时间等指标
4. **一致性保证**: 确保缓存数据的一致性和及时更新
5. **安全考虑**: 对敏感数据进行加密存储
6. **容量管理**: 合理的缓存淘汰和容量控制

实施建议：

- 静态资源使用长期强缓存
- HTML 文件使用短期缓存或协商缓存
- API 响应根据业务需求选择缓存策略
- 使用 Service Worker 实现离线支持
- 定期清理过期缓存，控制存储空间
- 监控缓存效果，持续优化策略

通过合理的缓存策略，我们可以显著提升应用性能，减少网络请求，改善用户体验。
