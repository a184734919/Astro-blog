## 概述

前端异常捕获是指在应用程序中及时发现、记录和处理各种运行时错误的技术手段。良好的异常捕获机制能够提高应用的稳定性，帮助开发者快速定位和解决问题。

## 核心概念

### 异常类型分类

**JavaScript 错误:**
- SyntaxError: 语法错误
- ReferenceError: 引用错误
- TypeError: 类型错误
- RangeError: 范围错误
- EvalError: eval 错误
- URIError: URI 错误

**网络异常:**
- 请求失败
- 超时错误
- 跨域问题
- 网络断开

**资源加载异常:**
- 图片加载失败
- 脚本加载失败
- 样式加载失败

**Promise 异常:**
- Promise rejection
- async/await 错误

### 1. 全局错误捕获

```javascript
// 全局错误处理
class GlobalErrorHandler {
  constructor(config = {}) {
    this.config = {
      reportUrl: '/api/errors',
      enableConsole: true,
      enableReport: true,
      maxQueueSize: 50,
      ...config
    };
    
    this.errorQueue = [];
    this.setupHandlers();
  }
  
  setupHandlers() {
    // 捕获同步错误
    window.onerror = this.handleError.bind(this);
    
    // 捕获资源加载错误
    window.addEventListener('error', this.handleResourceError.bind(this), true);
    
    // 捕获 Promise 错误
    window.addEventListener('unhandledrejection', this.handlePromiseError.bind(this));
    
    // 捕获框架错误（如 React）
    this.setupFrameworkHandlers();
  }
  
  handleError(message, source, lineno, colno, error) {
    const errorInfo = {
      type: 'javascript',
      message: message,
      source: source,
      line: lineno,
      column: colno,
      error: error ? this.parseError(error) : null,
      userAgent: navigator.userAgent,
      url: window.location.href,
      timestamp: Date.now(),
      stack: error ? error.stack : null
    };
    
    this.processError(errorInfo);
    return true; // 阻止默认错误处理
  }
  
  handleResourceError(event) {
    if (event.target !== window) {
      const target = event.target;
      
      const errorInfo = {
        type: 'resource',
        resourceType: this.getResourceType(target),
        source: target.src || target.href,
        tagName: target.tagName,
        error: event.message,
        url: window.location.href,
        timestamp: Date.now()
      };
      
      this.processError(errorInfo);
    }
  }
  
  handlePromiseError(event) {
    const error = event.reason;
    
    const errorInfo = {
      type: 'promise',
      message: error.message || 'Unhandled Promise Rejection',
      error: this.parseError(error),
      stack: error.stack,
      url: window.location.href,
      timestamp: Date.now()
    };
    
    this.processError(errorInfo);
    event.preventDefault(); // 阻止默认错误处理
  }
  
  setupFrameworkHandlers() {
    // React 错误边界
    if (typeof window !== 'undefined' && window.__REACT_ERROR_BOUNDARY__) {
      window.__REACT_ERROR_BOUNDARY__.subscribe(this.handleError.bind(this));
    }
    
    // Vue 错误处理器
    if (typeof window !== 'undefined' && window.__VUE__) {
      window.__VUE__.config.errorHandler = this.handleVueError.bind(this);
    }
  }
  
  handleVueError(error, vm, info) {
    const errorInfo = {
      type: 'vue',
      message: error.message,
      error: this.parseError(error),
      stack: error.stack,
      component: vm?.$options?.name || vm?.$options?.__name || 'Unknown',
      lifecycle: info,
      url: window.location.href,
      timestamp: Date.now()
    };
    
    this.processError(errorInfo);
  }
  
  getResourceType(target) {
    if (target.tagName === 'IMG') return 'image';
    if (target.tagName === 'SCRIPT') return 'script';
    if (target.tagName === 'LINK' && target.rel === 'stylesheet') return 'stylesheet';
    if (target.tagName === 'VIDEO') return 'video';
    if (target.tagName === 'AUDIO') return 'audio';
    return 'unknown';
  }
  
  parseError(error) {
    if (!error) return null;
    
    return {
      name: error.name,
      message: error.message,
      stack: error.stack
    };
  }
  
  processError(errorInfo) {
    // 添加到队列
    this.errorQueue.push(errorInfo);
    
    // 控制台输出
    if (this.config.enableConsole) {
      console.error('捕获到错误:', errorInfo);
    }
    
    // 上报错误
    if (this.config.enableReport) {
      this.reportError(errorInfo);
    }
    
    // 清理队列
    if (this.errorQueue.length > this.config.maxQueueSize) {
      this.errorQueue.shift();
    }
  }
  
  reportError(errorInfo) {
    // 使用 sendBeacon 确保错误能够上报
    if ('sendBeacon' in navigator) {
      const blob = new Blob([JSON.stringify(errorInfo)], {
        type: 'application/json'
      });
      navigator.sendBeacon(this.config.reportUrl, blob);
    } else {
      // 回退到 fetch
      fetch(this.config.reportUrl, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(errorInfo),
        keepalive: true
      }).catch(err => console.error('错误上报失败:', err));
    }
  }
  
  getErrorStats() {
    return {
      total: this.errorQueue.length,
      byType: this.groupErrorsByType(),
      recent: this.errorQueue.slice(-10)
    };
  }
  
  groupErrorsByType() {
    return this.errorQueue.reduce((acc, error) => {
      acc[error.type] = (acc[error.type] || 0) + 1;
      return acc;
    }, {});
  }
}

// 使用示例
const globalErrorHandler = new GlobalErrorHandler({
  reportUrl: '/api/errors',
  enableConsole: true
});
```

### 2. Promise 错误处理

```javascript
// Promise 错误处理工具
class PromiseErrorHandler {
  constructor() {
    this.setupUnhandledRejectionHandler();
    this.setupAsyncErrorCapture();
  }
  
  setupUnhandledRejectionHandler() {
    // 全局未处理的 Promise rejection
    window.addEventListener('unhandledrejection', (event) => {
      const error = event.reason;
      
      this.handleError({
        type: 'unhandled_promise_rejection',
        error: error,
        promise: event.promise,
        message: error?.message || 'Unhandled Promise Rejection',
        stack: error?.stack,
        timestamp: Date.now()
      });
      
      // 阻止默认的控制台错误输出
      event.preventDefault();
    });
  }
  
  setupAsyncErrorCapture() {
    // 捕获 async/await 错误
    const originalSetTimeout = window.setTimeout;
    const originalSetInterval = window.setInterval;
    
    window.setTimeout = (callback, delay, ...args) => {
      return originalSetTimeout(() => {
        try {
          callback(...args);
        } catch (error) {
          this.handleError({
            type: 'async_timeout_error',
            error: error,
            message: error.message,
            stack: error.stack,
            timestamp: Date.now()
          });
        }
      }, delay, ...args);
    };
    
    window.setInterval = (callback, delay, ...args) => {
      return originalSetInterval(() => {
        try {
          callback(...args);
        } catch (error) {
          this.handleError({
            type: 'async_interval_error',
            error: error,
            message: error.message,
            stack: error.stack,
            timestamp: Date.now()
          });
        }
      }, delay, ...args);
    };
  }
  
  handleError(errorInfo) {
    console.error('Promise 错误:', errorInfo);
    
    // 上报错误
    this.reportError(errorInfo);
  }
  
  reportError(errorInfo) {
    fetch('/api/promise-errors', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(errorInfo),
      keepalive: true
    }).catch(err => console.error('上报失败:', err));
  }
  
  // 创建安全的 Promise 包装器
  static safePromise(promise) {
    return promise
      .catch(error => {
        console.error('Promise 错误:', error);
        
        // 上报错误
        new PromiseErrorHandler().handleError({
          type: 'caught_promise_error',
          error: error,
          message: error.message,
          stack: error.stack,
          timestamp: Date.now()
        });
        
        // 返回默认值
        return null;
      });
  }
  
  // Promise 重试机制
  static async retry(fn, options = {}) {
    const {
      retries = 3,
      delay = 1000,
      backoff = 2,
      onRetry = null
    } = options;
    
    let lastError;
    
    for (let i = 0; i < retries; i++) {
      try {
        return await fn();
      } catch (error) {
        lastError = error;
        
        if (onRetry) {
          onRetry(error, i + 1);
        }
        
        if (i < retries - 1) {
          const waitTime = delay * Math.pow(backoff, i);
          await new Promise(resolve => setTimeout(resolve, waitTime));
        }
      }
    }
    
    throw lastError;
  }
}

// 使用示例
const promiseErrorHandler = new PromiseErrorHandler();

// 安全的 Promise
const safeData = PromiseErrorHandler.safePromise(
  fetch('/api/data').then(res => res.json())
);

// 带重试的 Promise
const dataWithRetry = PromiseErrorHandler.retry(
  () => fetch('/api/data').then(res => res.json()),
  { retries: 3, delay: 1000 }
);
```

### 3. 网络请求错误处理

```javascript
// 网络请求错误处理
class NetworkErrorHandler {
  constructor() {
    this.setupInterceptors();
  }
  
  setupInterceptors() {
    // 拦截 fetch 请求
    this.interceptFetch();
    
    // 拦截 XMLHttpRequest
    this.interceptXHR();
  }
  
  interceptFetch() {
    const originalFetch = window.fetch;
    
    window.fetch = async (...args) => {
      const [url, options = {}] = args;
      const startTime = performance.now();
      
      try {
        const response = await originalFetch(...args);
        const endTime = performance.now();
        
        // 检查响应状态
        if (!response.ok) {
          this.handleResponseError(response, {
            url,
            method: options.method || 'GET',
            duration: endTime - startTime,
            status: response.status,
            statusText: response.statusText
          });
        }
        
        return response;
      } catch (error) {
        const endTime = performance.now();
        
        this.handleNetworkError(error, {
          url,
          method: options.method || 'GET',
          duration: endTime - startTime,
          message: error.message
        });
        
        throw error;
      }
    };
  }
  
  interceptXHR() {
    const originalOpen = XMLHttpRequest.prototype.open;
    const originalSend = XMLHttpRequest.prototype.send;
    
    XMLHttpRequest.prototype.open = function(method, url, ...rest) {
      this._method = method;
      this._url = url;
      this._startTime = performance.now();
      return originalOpen.call(this, method, url, ...rest);
    };
    
    XMLHttpRequest.prototype.send = function(...args) {
      this.addEventListener('error', () => {
        const endTime = performance.now();
        new NetworkErrorHandler().handleNetworkError(new Error('XHR Error'), {
          url: this._url,
          method: this._method,
          duration: endTime - this._startTime
        });
      });
      
      this.addEventListener('load', () => {
        if (this.status >= 400) {
          const endTime = performance.now();
          new NetworkErrorHandler().handleResponseError({
            status: this.status,
            statusText: this.statusText
          }, {
            url: this._url,
            method: this._method,
            duration: endTime - this._startTime
          });
        }
      });
      
      return originalSend.apply(this, args);
    };
  }
  
  handleResponseError(response, context) {
    const errorInfo = {
      type: 'http_error',
      status: response.status,
      statusText: response.statusText,
      url: context.url,
      method: context.method,
      duration: context.duration,
      timestamp: Date.now()
    };
    
    console.error('HTTP 错误:', errorInfo);
    this.reportError(errorInfo);
  }
  
  handleNetworkError(error, context) {
    const errorInfo = {
      type: 'network_error',
      message: error.message,
      url: context.url,
      method: context.method,
      duration: context.duration,
      isOnline: navigator.onLine,
      timestamp: Date.now()
    };
    
    console.error('网络错误:', errorInfo);
    this.reportError(errorInfo);
  }
  
  reportError(errorInfo) {
    fetch('/api/network-errors', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(errorInfo),
      keepalive: true
    }).catch(err => console.error('上报失败:', err));
  }
  
  // 创建带错误处理的请求函数
  static async safeRequest(url, options = {}) {
    const startTime = performance.now();
    
    try {
      const response = await fetch(url, options);
      const endTime = performance.now();
      
      if (!response.ok) {
        throw new Error(`HTTP ${response.status}: ${response.statusText}`);
      }
      
      return await response.json();
    } catch (error) {
      const endTime = performance.now();
      
      new NetworkErrorHandler().handleNetworkError(error, {
        url,
        method: options.method || 'GET',
        duration: endTime - startTime
      });
      
      throw error;
    }
  }
}

// 使用示例
const networkErrorHandler = new NetworkErrorHandler();

// 安全的请求
try {
  const data = NetworkErrorHandler.safeRequest('/api/data', {
    method: 'GET',
    headers: { 'Content-Type': 'application/json' }
  });
} catch (error) {
  console.error('请求失败:', error);
}
```

```javascript
// React 错误边界组件
import React from 'react';

class ErrorBoundary extends React.Component {
  constructor(props) {
    super(props);
    this.state = { 
      hasError: false,
      error: null,
      errorInfo: null
    };
  }
  
  static getDerivedStateFromError(error) {
    return { hasError: true };
  }
  
  componentDidCatch(error, errorInfo) {
    // 记录错误
    this.setState({
      error: error,
      errorInfo: errorInfo
    });
    
    // 上报错误
    this.reportError(error, errorInfo);
    
    // 调用自定义错误处理器
    if (this.props.onError) {
      this.props.onError(error, errorInfo);
    }
  }
  
  reportError(error, errorInfo) {
    const errorData = {
      type: 'react_error',
      message: error.message,
      stack: error.stack,
      componentStack: errorInfo.componentStack,
      timestamp: Date.now(),
      url: window.location.href
    };
    
    console.error('React 错误:', errorData);
    
    // 上报错误
    fetch('/api/react-errors', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(errorData),
      keepalive: true
    }).catch(err => console.error('上报失败:', err));
  }
  
  render() {
    if (this.state.hasError) {
      // 自定义错误 UI
      if (this.props.fallback) {
        return this.props.fallback(this.state.error, this.state.errorInfo);
      }
      
      return (
        <div className="error-boundary">
          <h1>出错了</h1>
          <p>应用程序遇到了一个错误</p>
          {this.props.showDetails && (
            <details className="error-details">
              <summary>错误详情</summary>
              <pre>
                {this.state.error && this.state.error.toString()}
                <br />
                {this.state.errorInfo && this.state.errorInfo.componentStack}
              </pre>
            </details>
          )}
          <button onClick={() => window.location.reload()}>
            刷新页面
          </button>
        </div>
      );
    }
    
    return this.props.children;
  }
}

// Hook 版本的错误边界
function useErrorHandler() {
  const [error, setError] = React.useState(null);
  
  const resetError = React.useCallback(() => {
    setError(null);
  }, []);
  
  const handleError = React.useCallback((error) => {
    setError(error);
  }, []);
  
  // 错误边界捕获后调用
  ErrorBoundary.prototype componentDidCatch = function(error, errorInfo) {
    setError(error);
    
    if (this.props.onError) {
      this.props.onError(error, errorInfo);
    }
  };
  
  return { error, resetError, handleError };
}

// 使用示例
function App() {
  return (
    <ErrorBoundary
      fallback={(error, errorInfo) => (
        <div className="error-fallback">
          <h2>应用出错</h2>
          <button onClick={() => window.location.reload()}>
            重新加载
          </button>
        </div>
      )}
      onError={(error, errorInfo) => {
        console.error('应用错误:', error, errorInfo);
      }}
    >
      <MyComponent />
    </ErrorBoundary>
  );
}
```

### 综合异常处理系统

```javascript
// 综合异常处理系统
class ExceptionHandlingSystem {
  constructor(config = {}) {
    this.config = {
      reportUrl: '/api/exceptions',
      enableConsole: true,
      enableReport: true,
      maxQueueSize: 100,
      retryAttempts: 3,
      ...config
    };
    
    this.exceptionQueue = [];
    this.init();
  }
  
  init() {
    // 初始化各个错误处理器
    this.globalHandler = new GlobalErrorHandler(this.config);
    this.promiseHandler = new PromiseErrorHandler();
    this.networkHandler = new NetworkErrorHandler();
    this.setupCustomHandlers();
    
    // 定期清理队列
    setInterval(() => this.cleanupQueue(), 60000); // 每分钟清理一次
  }
  
  setupCustomHandlers() {
    // 监控 iframe 错误
    window.addEventListener('message', (event) => {
      if (event.data && event.data.type === 'iframe-error') {
        this.handleIframeError(event.data);
      }
    });
    
    // 监控 Web Worker 错误
    if ('Worker' in window) {
      const originalWorker = window.Worker;
      
      window.Worker = class extends originalWorker {
        constructor(url, options) {
          super(url, options);
          this.addEventListener('error', (event) => {
            new ExceptionHandlingSystem().handleWorkerError(event, url);
          });
        }
      };
    }
    
    // 监控 Service Worker 错误
    if ('serviceWorker' in navigator) {
      navigator.serviceWorker.addEventListener('error', (event) => {
        this.handleServiceWorkerError(event);
      });
    }
  }
  
  handleIframeError(errorData) {
    this.addException({
      type: 'iframe_error',
      source: errorData.source,
      message: errorData.message,
      stack: errorData.stack,
      timestamp: Date.now()
    });
  }
  
  handleWorkerError(event, workerUrl) {
    this.addException({
      type: 'worker_error',
      workerUrl: workerUrl,
      message: event.message,
      filename: event.filename,
      lineno: event.lineno,
      timestamp: Date.now()
    });
  }
  
  handleServiceWorkerError(event) {
    this.addException({
      type: 'service_worker_error',
      message: event.message,
      filename: event.filename,
      lineno: event.lineno,
      timestamp: Date.now()
    });
  }
  
  addException(exception) {
    // 添加上下文信息
    const enrichedException = {
      ...exception,
      context: this.getContext(),
      userAgent: navigator.userAgent,
      url: window.location.href,
      timestamp: exception.timestamp || Date.now()
    };
    
    this.exceptionQueue.push(enrichedException);
    
    // 控制台输出
    if (this.config.enableConsole) {
      console.error('捕获到异常:', enrichedException);
    }
    
    // 上报异常
    if (this.config.enableReport) {
      this.reportException(enrichedException);
    }
    
    // 清理队列
    if (this.exceptionQueue.length > this.config.maxQueueSize) {
      this.exceptionQueue.shift();
    }
  }
  
  getContext() {
    return {
      screen: {
        width: window.screen.width,
        height: window.screen.height
      },
      viewport: {
        width: window.innerWidth,
        height: window.innerHeight
      },
      language: navigator.language,
      platform: navigator.platform,
      connection: this.getConnectionInfo(),
      memory: this.getMemoryInfo()
    };
  }
  
  getConnectionInfo() {
    if ('connection' in navigator) {
      const connection = navigator.connection;
      return {
        effectiveType: connection.effectiveType,
        downlink: connection.downlink,
        rtt: connection.rtt,
        saveData: connection.saveData
      };
    }
    return null;
  }
  
  getMemoryInfo() {
    if ('memory' in performance) {
      const memory = performance.memory;
      return {
        usedJSHeapSize: memory.usedJSHeapSize,
        totalJSHeapSize: memory.totalJSHeapSize,
        jsHeapSizeLimit: memory.jsHeapSizeLimit
      };
    }
    return null;
  }
  
  async reportException(exception) {
    for (let attempt = 1; attempt <= this.config.retryAttempts; attempt++) {
      try {
        const response = await fetch(this.config.reportUrl, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify(exception),
          keepalive: true
        });
        
        if (response.ok) {
          return;
        }
      } catch (error) {
        if (attempt === this.config.retryAttempts) {
          console.error('异常上报失败:', error);
          
          // 最后一次尝试使用 sendBeacon
          if ('sendBeacon' in navigator) {
            const blob = new Blob([JSON.stringify(exception)], {
              type: 'application/json'
            });
            navigator.sendBeacon(this.config.reportUrl, blob);
          }
        } else {
          // 等待后重试
          await new Promise(resolve => 
            setTimeout(resolve, Math.pow(2, attempt) * 1000)
          );
        }
      }
    }
  }
  
  cleanupQueue() {
    // 移除过期的异常记录（超过24小时）
    const oneDayAgo = Date.now() - 24 * 60 * 60 * 1000;
    this.exceptionQueue = this.exceptionQueue.filter(
      exception => exception.timestamp > oneDayAgo
    );
  }
  
  getExceptionStats() {
    return {
      total: this.exceptionQueue.length,
      byType: this.groupExceptionsByType(),
      recent: this.exceptionQueue.slice(-20),
      summary: this.generateSummary()
    };
  }
  
  groupExceptionsByType() {
    return this.exceptionQueue.reduce((acc, exception) => {
      acc[exception.type] = (acc[exception.type] || 0) + 1;
      return acc;
    }, {});
  }
  
  generateSummary() {
    const byType = this.groupExceptionsByType();
    const total = this.exceptionQueue.length;
    
    return {
      mostCommonType: Object.entries(byType)
        .sort(([, a], [, b]) => b - a)[0]?.[0] || 'none',
      totalExceptions: total,
      exceptionsInLastHour: this.exceptionQueue.filter(
        e => e.timestamp > Date.now() - 3600000
      ).length,
      exceptionsInLastDay: this.exceptionQueue.filter(
        e => e.timestamp > Date.now() - 86400000
      ).length
    };
  }
  
  // 创建安全函数包装器
  createSafeFunction(fn, context = null) {
    return (...args) => {
      try {
        return fn.apply(context, args);
      } catch (error) {
        this.addException({
          type: 'function_error',
          message: error.message,
          stack: error.stack,
          functionName: fn.name || 'anonymous',
          timestamp: Date.now()
        });
        
        return null;
      }
    };
  }
  
  // 创建安全异步函数包装器
  createSafeAsyncFunction(fn, context = null) {
    return async (...args) => {
      try {
        return await fn.apply(context, args);
      } catch (error) {
        this.addException({
          type: 'async_function_error',
          message: error.message,
          stack: error.stack,
          functionName: fn.name || 'anonymous',
          timestamp: Date.now()
        });
        
        return null;
      }
    };
  }
}
```

```javascript
// 控制错误处理对性能的影响
class PerformanceAwareErrorHandler {
  constructor() {
    this.errorCount = 0;
    this.throttleThreshold = 100; // 同一时间最多处理100个错误
    this.throttleTimeout = null;
  }
  
  handleError(error) {
    // 防止错误处理过于频繁
    if (this.errorCount >= this.throttleThreshold) {
      if (!this.throttleTimeout) {
        this.throttleTimeout = setTimeout(() => {
          this.errorCount = 0;
          this.throttleTimeout = null;
        }, 60000); // 1分钟后重置
      }
      return;
    }
    
    this.errorCount++;
    
    // 使用 requestIdleCallback 在空闲时处理
    if ('requestIdleCallback' in window) {
      requestIdleCallback(() => this.processError(error), { timeout: 2000 });
    } else {
      setTimeout(() => this.processError(error), 0);
    }
  }
  
  processError(error) {
    // 实际的错误处理逻辑
    console.error('处理错误:', error);
    this.reportError(error);
  }
}
```

### 2. 敏感信息过滤

```javascript
// 过滤敏感信息
class SecureErrorHandler {
  constructor() {
    this.sensitivePatterns = [
      /password/i,
      /token/i,
      /secret/i,
      /key/i,
      /authorization/i,
      /\b\d{16}\b/, // 信用卡号
      /\b\d{3}-\d{2}-\d{4}\b/ // SSN
    ];
  }
  
  sanitizeError(error) {
    const sanitized = { ...error };
    
    // 过滤错误消息中的敏感信息
    if (sanitized.message) {
      sanitized.message = this.removeSensitiveInfo(sanitized.message);
    }
    
    // 过滤堆栈信息中的敏感信息
    if (sanitized.stack) {
      sanitized.stack = this.removeSensitiveInfo(sanitized.stack);
    }
    
    // 过滤自定义字段
    Object.keys(sanitized).forEach(key => {
      if (typeof sanitized[key] === 'string') {
        sanitized[key] = this.removeSensitiveInfo(sanitized[key]);
      }
    });
    
    return sanitized;
  }
  
  removeSensitiveInfo(text) {
    return this.sensitivePatterns.reduce((cleaned, pattern) => {
      return cleaned.replace(pattern, '[REDACTED]');
    }, text);
  }
}
```

### 3. 用户体验优化

```javascript
// 用户友好的错误处理
class UserFriendlyErrorHandler {
  constructor() {
    this.errorMessages = {
      network: '网络连接异常，请检查网络设置',
      server: '服务器繁忙，请稍后再试',
      timeout: '请求超时，请重试',
      permission: '您没有权限执行此操作',
      notFound: '请求的资源不存在',
      unknown: '发生未知错误，请联系客服'
    };
  }
  
  getUserFriendlyMessage(error) {
    const errorType = this.classifyError(error);
    return this.errorMessages[errorType] || this.errorMessages.unknown;
  }
  
  classifyError(error) {
    if (!navigator.onLine) return 'network';
    if (error.message.includes('timeout')) return 'timeout';
    if (error.message.includes('403')) return 'permission';
    if (error.message.includes('404')) return 'notFound';
    if (error.message.includes('500')) return 'server';
    return 'unknown';
  }
  
  showErrorToUser(error) {
    const message = this.getUserFriendlyMessage(error);
    
    // 创建用户友好的错误提示
    const errorToast = document.createElement('div');
    errorToast.className = 'error-toast';
    errorToast.textContent = message;
    
    document.body.appendChild(errorToast);
    
    // 3秒后自动消失
    setTimeout(() => {
      errorToast.classList.add('fade-out');
      setTimeout(() => errorToast.remove(), 300);
    }, 3000);
  }
}
```

## 总结

前端异常捕获是保证应用稳定性的重要手段，关键点包括：

1. **全面覆盖**: 覆盖 JavaScript 错误、Promise 错误、网络错误、资源错误等
2. **分类处理**: 不同类型的错误采用不同的处理策略
3. **上下文信息**: 收集足够的上下文信息便于问题定位
4. **性能考虑**: 控制错误处理对性能的影响
5. **安全保护**: 过滤敏感信息，保护用户隐私
6. **用户体验**: 提供友好的错误提示，减少用户困扰

实施建议：

- 建立多层次错误捕获机制
- 实现错误上报和监控
- 设置合理的错误阈值和报警
- 定期分析错误数据，指导优化
- 提供友好的用户界面
- 建立错误处理的最佳实践

通过完善的异常捕获机制，我们可以及时发现和解决问题，提高应用的稳定性和用户体验。
