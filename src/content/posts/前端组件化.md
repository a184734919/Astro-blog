## 概述

前端组件化是指将复杂的用户界面拆分成多个独立、可复用的组件，每个组件负责特定的功能和展示。组件化开发提高了代码的可维护性、可测试性和开发效率。

## 核心概念

### 组件化原则

**单一职责原则 (SRP):**
- 每个组件只负责一个功能
- 组件应该专注于一个特定的业务需求
- 避免组件承担过多责任

**开闭原则 (OCP):**
- 组件应该对扩展开放，对修改关闭
- 通过组合和继承实现功能扩展
- 避免直接修改现有组件

**依赖倒置原则 (DIP):**
- 组件应该依赖于抽象而非具体实现
- 使用接口和类型定义来定义组件契约
- 降低组件间的耦合度

### 组件分类

**基础组件:**
- Button、Input、Select 等UI基础元素
- 提供基本的交互功能
- 高度可复用，样式可定制

**业务组件:**
- UserCard、ProductItem、OrderForm 等
- 封装特定的业务逻辑
- 可组合使用构建复杂界面

**布局组件:**
- Container、Grid、Layout 等
- 负责页面结构和布局
- 提供一致的布局模式

**功能组件:**
- DataProvider、ErrorHandler、AuthProvider 等
- 提供跨组件的功能
- 使用高阶组件或 Hook 实现

// 基础按钮组件
import React, { forwardRef } from 'react';
import PropTypes from 'prop-types';

const Button = forwardRef(({
  children,
  variant = 'primary',
  size = 'medium',
  disabled = false,
  loading = false,
  icon,
  iconPosition = 'left',
  fullWidth = false,
  onClick,
  className = '',
  ...props
}, ref) => {
  const baseClasses = 'btn';
  
  const variantClasses = {
    primary: 'btn-primary',
    secondary: 'btn-secondary',
    outline: 'btn-outline',
    ghost: 'btn-ghost',
    danger: 'btn-danger'
  };
  
  const sizeClasses = {
    small: 'btn-small',
    medium: 'btn-medium',
    large: 'btn-large'
  };
  
  const classes = [
    baseClasses,
    variantClasses[variant],
    sizeClasses[size],
    fullWidth ? 'btn-full-width' : '',
    disabled ? 'btn-disabled' : '',
    loading ? 'btn-loading' : '',
    className
  ].filter(Boolean).join(' ');
  
  const handleClick = (event) => {
    if (!disabled && !loading && onClick) {
      onClick(event);
    }
  };
  
  return (
    <button
      ref={ref}
      className={classes}
      disabled={disabled || loading}
      onClick={handleClick}
      {...props}
    >
      {loading && <span className="btn-spinner" />}
      
      {icon && iconPosition === 'left' && !loading && (
        <span className="btn-icon btn-icon-left">{icon}</span>
      )}
      
      <span className="btn-content">{children}</span>
      
      {icon && iconPosition === 'right' && !loading && (
        <span className="btn-icon btn-icon-right">{icon}</span>
      )}
    </button>
  );
});

Button.propTypes = {
  children: PropTypes.node.isRequired,
  variant: PropTypes.oneOf(['primary', 'secondary', 'outline', 'ghost', 'danger']),
  size: PropTypes.oneOf(['small', 'medium', 'large']),
  disabled: PropTypes.bool,
  loading: PropTypes.bool,
  icon: PropTypes.node,
  iconPosition: PropTypes.oneOf(['left', 'right']),
  fullWidth: PropTypes.bool,
  onClick: PropTypes.func,
  className: PropTypes.string
};

Button.displayName = 'Button';

export default Button;

// 基础输入框组件
import React, { forwardRef, useState, useCallback } from 'react';
import PropTypes from 'prop-types';

const Input = forwardRef(({
  type = 'text',
  value,
  defaultValue,
  placeholder,
  disabled = false,
  error,
  helperText,
  label,
  required = false,
  maxLength,
  minLength,
  pattern,
  onChange,
  onBlur,
  onFocus,
  className = '',
  ...props
}, ref) => {
  const [focused, setFocused] = useState(false);
  const [internalValue, setInternalValue] = useState(defaultValue || '');
  
  const currentValue = value !== undefined ? value : internalValue;
  
  const handleChange = useCallback((event) => {
    const newValue = event.target.value;
    
    if (value === undefined) {
      setInternalValue(newValue);
    }
    
    if (onChange) {
      onChange(newValue, event);
    }
  }, [value, onChange]);
  
  const handleFocus = useCallback((event) => {
    setFocused(true);
    if (onFocus) {
      onFocus(event);
    }
  }, [onFocus]);
  
  const handleBlur = useCallback((event) => {
    setFocused(false);
    if (onBlur) {
      onBlur(event);
    }
  }, [onBlur]);
  
  const inputClasses = [
    'input',
    focused && 'input-focused',
    error && 'input-error',
    disabled && 'input-disabled',
    className
  ].filter(Boolean).join(' ');
  
  return (
    <div className="input-wrapper">
      {label && (
        <label className="input-label">
          {label}
          {required && <span className="input-required">*</span>}
        </label>
      )}
      
      <div className="input-container">
        <input
          ref={ref}
          type={type}
          value={currentValue}
          placeholder={placeholder}
          disabled={disabled}
          maxLength={maxLength}
          minLength={minLength}
          pattern={pattern}
          onChange={handleChange}
          onFocus={handleFocus}
          onBlur={handleBlur}
          className={inputClasses}
          {...props}
        />
        
        {error && (
          <span className="input-error-icon" aria-hidden="true">
            ⚠️
          </span>
        )}
      </div>
      
      {error && (
        <span className="input-error-text">{error}</span>
      )}
      
      {helperText && !error && (
        <span className="input-helper-text">{helperText}</span>
      )}
    </div>
  );
});

Input.propTypes = {
  type: PropTypes.oneOf(['text', 'password', 'email', 'number', 'tel', 'url']),
  value: PropTypes.string,
  defaultValue: PropTypes.string,
  placeholder: PropTypes.string,
  disabled: PropTypes.bool,
  error: PropTypes.string,
  helperText: PropTypes.string,
  label: PropTypes.string,
  required: PropTypes.bool,
  maxLength: PropTypes.number,
  minLength: PropTypes.number,
  pattern: PropTypes.string,
  onChange: PropTypes.func,
  onBlur: PropTypes.func,
  onFocus: PropTypes.func,
  className: PropTypes.string
};

Input.displayName = 'Input';

export default Input;

const ProductItem = ({
  product,
  onAddToCart,
  onWishlist,
  isInWishlist,
  onQuickView,
  className = '',
  ...props
}) => {
  const {
    id,
    name,
    price,
    originalPrice,
    discount,
    image,
    rating,
    reviewCount,
    inStock,
    isNew,
    isBestseller,
    category
  } = product;
  
  const hasDiscount = discount && originalPrice && originalPrice > price;
  const discountPercent = hasDiscount 
    ? Math.round((1 - price / originalPrice) * 100) 
    : 0;
  
  const renderBadges = () => {
    const badges = [];
    
    if (isNew) {
      badges.push(<Badge key="new" variant="success">新品</Badge>);

      badges.push(<Badge key="bestseller" variant="warning">热销</Badge>);
    }
    
    if (hasDiscount) {
      badges.push(<Badge key="discount" variant="danger">-{discountPercent}%</Badge>);
    }
    
    if (!inStock) {
      badges.push(<Badge key="outofstock" variant="secondary">缺货</Badge>);
    }
    
    return badges;
  };
  
  return (
    <div className={`product-item ${className}`} {...props}>
      <div className="product-item-image">
        <img src={image} alt={name} loading="lazy" />
        
        <div className="product-item-badges">
          {renderBadges()}
        </div>
        
        <div className="product-item-actions">
          {onQuickView && (
            <button
              className="product-item-action-btn"
              onClick={() => onQuickView(id)}
              aria-label="快速查看"
            >
              👁️
            </button>
          )}
          
          {onWishlist && (
            <button
              className={`product-item-action-btn ${isInWishlist ? 'active' : ''}`}
              onClick={() => onWishlist(id)}
              aria-label={isInWishlist ? '从愿望清单移除' : '加入愿望清单'}
            >
              {isInWishlist ? '❤️' : '🤍'}
            </button>
          )}
        </div>
      </div>
      
      <div className="product-item-content">
        <p className="product-item-category">{category}</p>
        <h3 className="product-item-name">{name}</h3>
        
        <div className="product-item-rating">
          <span className="rating-stars">{'★'.repeat(Math.floor(rating))}</span>
          <span className="rating-score">{rating}</span>
          <span className="rating-count">({reviewCount} 条评论)</span>
        </div>

            <span className="price-original">¥{originalPrice.toFixed(2)}</span>
          )}

          <Button
            variant={inStock ? 'primary' : 'secondary'}

            fullWidth
            disabled={!inStock}

          >
            {inStock ? '加入购物车' : '缺货'}

        )}
      </div>

  );

// 认证高阶组件
import React from 'react';
import PropTypes from 'prop-types';

const withAuth = (WrappedComponent, options = {}) => {
  const {
    requireAuth = true,
    redirectTo = '/login',
    fallbackComponent = null
  } = options;
  
  return function AuthenticatedComponent(props) {
    const isAuthenticated = useAuth();
    const router = useRouter();
    
    React.useEffect(() => {
      if (requireAuth && !isAuthenticated) {
        router.push(redirectTo);
      }
    }, [isAuthenticated, requireAuth, redirectTo, router]);
    
    if (requireAuth && !isAuthenticated) {
      return fallbackComponent || <div>加载中...</div>;
    }
    
    return <WrappedComponent {...props} />;
  };
  
  AuthenticatedComponent.displayName = `withAuth(${WrappedComponent.displayName || WrappedComponent.name || 'Component'})`;
  
  return AuthenticatedComponent;
};

// 使用示例
const UserProfile = ({ user }) => {
  return (
    <div>
      <h1>用户资料</h1>
      <p>欢迎, {user.name}!</p>
    </div>
  );
};

const ProtectedUserProfile = withAuth(UserProfile, {
  requireAuth: true,
  redirectTo: '/login'
});

// 数据加载高阶组件
const withData = (WrappedComponent, dataLoader) => {
  return function DataLoaderComponent(props) {
    const [data, setData] = React.useState(null);
    const [loading, setLoading] = React.useState(true);
    const [error, setError] = React.useState(null);
    
    React.useEffect(() => {
      const loadData = async () => {
        try {
          setLoading(true);
          const result = await dataLoader(props);
          setData(result);
        } catch (err) {
          setError(err);
        } finally {
          setLoading(false);
        }
      };
      
      loadData();
    }, [props]);
    
    if (loading) {
      return <div>加载中...</div>;
    }
    
    if (error) {
      return <div>加载失败: {error.message}</div>;
    }
    
    return <WrappedComponent data={data} {...props} />;
  };
  
  DataLoaderComponent.displayName = `withData(${WrappedComponent.displayName || WrappedComponent.name || 'Component'})`;
  
  return DataLoaderComponent;
};

// 使用示例
const UserList = ({ data }) => {
  return (
    <div>
      <h1>用户列表</h1>
      <ul>
        {data.map(user => (
          <li key={user.id}>{user.name}</li>
        ))}
      </ul>
    </div>
  );
};

const loadUsers = async () => {
  const response = await fetch('/api/users');
  return response.json();
};

const UserListWithData = withData(UserList, loadUsers);

// 错误边界高阶组件
const withErrorBoundary = (WrappedComponent, fallbackComponent) => {
  return class ErrorBoundaryWrapper extends React.Component {
    constructor(props) {
      super(props);
      this.state = { hasError: false, error: null };
    }
    
    static getDerivedStateFromError(error) {
      return { hasError: true, error };
    }
    
    componentDidCatch(error, errorInfo) {
      console.error('Error caught by boundary:', error, errorInfo);
    }
    
    render() {
      if (this.state.hasError) {
        if (fallbackComponent) {
          return React.cloneElement(fallbackComponent, { 
            error: this.state.error 
          });
        }
        
        return (
          <div className="error-boundary">
            <h1>出错了</h1>
            <p>应用程序遇到了一个错误</p>
            <button onClick={() => this.setState({ hasError: false, error: null })}>
              重试
            </button>
          </div>
        );
      }
      
      return <WrappedComponent {...this.props} />;
    }
  };
  
  ErrorBoundaryWrapper.displayName = `withErrorBoundary(${WrappedComponent.displayName || WrappedComponent.name || 'Component'})`;
  
  return ErrorBoundaryWrapper;
};

  const value = {
    theme,
    setTheme,
    toggleTheme
  };
  
  return (
    <ThemeContext.Provider value={value}>
      <div className={`theme-${theme}`}>
        {children}
      </div>
    </ThemeContext.Provider>
  );
};

export const useTheme = () => {
  const context = useContext(ThemeContext);
  if (!context) {
    throw new Error('useTheme must be used within a ThemeProvider');
  }
  return context;
};

class EventBus {
  constructor() {
    this.events = {};
  }
  
  on(event, callback) {
    if (!this.events[event]) {
      this.events[event] = [];
    }
    this.events[event].push(callback);
  }
  
  off(event, callback) {
    if (!this.events[event]) return;
    
    this.events[event] = this.events[event].filter(cb => cb !== callback);
  }
  
  emit(event, data) {
    if (!this.events[event]) return;
    
    this.events[event].forEach(callback => callback(data));
  }
}

const eventBus = new EventBus();

// 组件 A 发布事件
const ComponentA = () => {
  const handleClick = () => {
    eventBus.emit('data-update', { timestamp: Date.now() });
  };
  
  return <button onClick={handleClick}>Update Data</button>;
};

const ComponentB = () => {

    const handleUpdate = (data) => {
      setLastUpdate(data);
    };

    return () => {
      eventBus.off('data-update', handleUpdate);
