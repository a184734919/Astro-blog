## 概述

前端模块化是指将代码拆分成独立、可复用的模块，每个模块拥有自己的作用域和接口。模块化开发提高了代码的可维护性、可测试性和开发效率，是现代前端开发的基础。

## 核心概念

### 模块化发展历程

**全局函数时代:**
- 所有函数都在全局作用域
- 容易产生命名冲突
- 无法管理依赖关系

**命名空间时代:**
- 使用对象作为命名空间
- 减少全局污染
- 仍然存在依赖管理问题

**IIFE 时代:**
- 使用立即执行函数创建作用域
- 实现私有变量和方法
- 依赖管理仍然困难

**CommonJS 时代:**
- Node.js 默认模块系统
- 同步加载，适合服务端
- 不直接支持浏览器

**AMD/UMD 时代:**
- 异步模块定义
- 支持浏览器环境
- 语法相对复杂

**ES6 模块时代:**
- 原生模块支持
- 静态分析，tree-shaking
- 现代前端开发标准

### 模块化特性

**作用域隔离:**
- 每个模块拥有独立作用域
- 避免全局变量污染
- 提供清晰的边界

**依赖管理:**
- 显式声明依赖关系
- 自动处理模块加载顺序
- 支持循环依赖处理

**接口导出:**
- 明确的导出接口
- 支持默认导出和命名导出
- 便于模块复用

**静态分析:**
- 编译时确定依赖关系
- 支持 tree-shaking 优化
- 提供更好的 IDE 支持

```javascript
export const formatDate = (date, format = 'YYYY-MM-DD') => {
  const d = new Date(date);
  const year = d.getFullYear();
  const month = String(d.getMonth() + 1).padStart(2, '0');
  const day = String(d.getDate()).padStart(2, '0');

    .replace('MM', month)
    .replace('DD', day);

export const debounce = (fn, delay = 300) => {
  let timer = null;
  
  return function(...args) {
    clearTimeout(timer);
    timer = setTimeout(() => {
      fn.apply(this, args);
    }, delay);
  };
};

export const throttle = (fn, delay = 300) => {
  let lastTime = 0;
  
  return function(...args) {
    const now = Date.now();
    if (now - lastTime >= delay) {
      lastTime = now;
      fn.apply(this, args);
    }
  };
};

// api.js - API 模块
const BASE_URL = 'https://api.example.com';

export class ApiClient {
  constructor(baseUrl = BASE_URL) {
    this.baseUrl = baseUrl;
    this.headers = {
      'Content-Type': 'application/json'
    };
  }
  
  setAuthToken(token) {
    this.headers['Authorization'] = `Bearer ${token}`;
  }
  
  async request(endpoint, options = {}) {
    const url = `${this.baseUrl}${endpoint}`;
    const config = {
      ...options,
      headers: {
        ...this.headers,
        ...options.headers
      }
    };
    
    try {
      const response = await fetch(url, config);
      
      if (!response.ok) {
        throw new Error(`HTTP ${response.status}: ${response.statusText}`);
      }
      
      return await response.json();
    } catch (error) {
      console.error('API request failed:', error);
      throw error;
    }
  }
  
  get(endpoint, params = {}) {
    const queryString = new URLSearchParams(params).toString();
    const url = queryString ? `${endpoint}?${queryString}` : endpoint;
    return this.request(url, { method: 'GET' });
  }
  
  post(endpoint, data) {
    return this.request(endpoint, {
      method: 'POST',
      body: JSON.stringify(data)
    });
  }
  
  put(endpoint, data) {
    return this.request(endpoint, {
      method: 'PUT',
      body: JSON.stringify(data)
    });
  }
  
  delete(endpoint) {
    return this.request(endpoint, { method: 'DELETE' });
  }
}

export default new ApiClient();
```

### 3. AMD 模块

```javascript
// math.js - AMD 模块
define([], function() {
  function add(a, b) {
    return a + b;
  }
  
  function subtract(a, b) {
    return a - b;
  }
  
  function multiply(a, b) {
    return a * b;
  }
  
  function divide(a, b) {
    return a / b;
  }
  
  // 返回模块接口
  return {
    add: add,
    subtract: subtract,
    multiply: multiply,
    divide: divide
  };
});

// utils.js - AMD 模块，依赖其他模块
define(['./math'], function(math) {
  function calculateAverage(numbers) {
    const sum = numbers.reduce((acc, num) => math.add(acc, num), 0);
    return math.divide(sum, numbers.length);
  }
  
  function debounce(fn, delay = 300) {
    let timer = null;
    
    return function(...args) {
      clearTimeout(timer);
      timer = setTimeout(() => {
        fn.apply(this, args);
      }, delay);
    };
  }
  
  return {
    calculateAverage: calculateAverage,
    debounce: debounce
  };
});

// main.js - 使用 AMD 模块
require(['./math', './utils'], function(math, utils) {
  // 使用数学模块
  console.log('加法:', math.add(5, 3));
  console.log('减法:', math.subtract(10, 4));
  
  // 使用工具模块
  const numbers = [1, 2, 3, 4, 5];
  const average = utils.calculateAverage(numbers);
  console.log('平均值:', average);
  
  // 使用防抖函数
  const debouncedLog = utils.debounce(function() {
    console.log('防抖函数执行');
  }, 500);
  
  // 多次调用只会执行最后一次
  debouncedLog();
  debouncedLog();
  debouncedLog();
});
```

### 4. UMD 模块

```javascript
// math.js - UMD 模块
(function (root, factory) {
  if (typeof define === 'function' && define.amd) {
    // AMD 环境
    define([], factory);
  } else if (typeof module === 'object' && module.exports) {
    // CommonJS 环境
    module.exports = factory();
  } else {
    // 浏览器全局变量
    root.MathLibrary = factory();
  }
}(typeof self !== 'undefined' ? self : this, function () {
  'use strict';
  
  function add(a, b) {
    return a + b;
  }
  
  function subtract(a, b) {
    return a - b;
  }
  
  function multiply(a, b) {
    return a * b;
  }
  
  function divide(a, b) {
    return a / b;
  }
  
  return {
    add: add,
    subtract: subtract,
    multiply: multiply,
    divide: divide
  };
}));

// 使用 UMD 模块
// 在浏览器中：
console.log(MathLibrary.add(5, 3)); // 8

// 在 CommonJS 中：
const math = require('./math.js');
console.log(math.add(5, 3)); // 8

// 在 AMD 中：
require(['./math'], function(math) {
  console.log(math.add(5, 3)); // 8
});
```

```javascript
export class ModuleVersionManager {
  constructor() {
    this.versions = new Map();
  }
  
  registerModule(name, version, module) {
    if (!this.versions.has(name)) {
      this.versions.set(name, new Map());
    }
    
    this.versions.get(name).set(version, module);
  }
  
  getModule(name, version = 'latest') {
    const moduleVersions = this.versions.get(name);
    
    if (!moduleVersions) {
      throw new Error(`Module ${name} not found`);
    }
    
    if (version === 'latest') {
      const versions = Array.from(moduleVersions.keys()).sort();
      return moduleVersions.get(versions[versions.length - 1]);
    }
    
    return moduleVersions.get(version);
  }
  
  listVersions(name) {
    const moduleVersions = this.versions.get(name);
    return moduleVersions ? Array.from(moduleVersions.keys()) : [];
  }
}

export class ModuleDependencyAnalyzer {
  constructor() {
    this.dependencies = new Map();
  }
  
  analyzeModule(modulePath, imports) {
    this.dependencies.set(modulePath, {
      imports: imports,
      dependents: []
    });
    
    // 更新依赖关系
    imports.forEach(importPath => {
      if (this.dependencies.has(importPath)) {
        this.dependencies.get(importPath).dependents.push(modulePath);
      } else {
        this.dependencies.set(importPath, {
          imports: [],
          dependents: [modulePath]
        });
      }
    });
  }
  
  getDependencies(modulePath) {
    const module = this.dependencies.get(modulePath);
    return module ? module.imports : [];
  }
  
  getDependents(modulePath) {
    const module = this.dependencies.get(modulePath);
    return module ? module.dependents : [];
  }
  
  getDependencyTree(modulePath, visited = new Set()) {
    if (visited.has(modulePath)) {
      return { circular: true, path: modulePath };
    }
    
    visited.add(modulePath);
    
    const dependencies = this.getDependencies(modulePath);
    const tree = {
      path: modulePath,
      dependencies: []
    };
    
    dependencies.forEach(dep => {
      const depTree = this.getDependencyTree(dep, new Set(visited));
      tree.dependencies.push(depTree);
    });
    
    return tree;
  }
  
  detectCircularDependencies() {
    const circularDeps = [];
    
    for (const [modulePath] of this.dependencies) {
      const visited = new Set();
      const tree = this.getDependencyTree(modulePath, visited);
      
      if (this.hasCircularDependency(tree)) {
        circularDeps.push(modulePath);
      }
    }
    
    return circularDeps;
  }
  
  hasCircularDependency(tree, visited = new Set()) {
    if (visited.has(tree.path)) {
      return true;
    }
    
    visited.add(tree.path);
    
    return tree.dependencies.some(dep => 
      this.hasCircularDependency(dep, visited)
    );
  }
}
```
