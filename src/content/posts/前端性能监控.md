## 概述

前端性能监控是指通过采集和分析用户在浏览器中的各种性能数据，来评估和优化应用性能的技术手段。它是提升用户体验、发现性能瓶颈的重要工具。

## 核心概念

### 监控指标分类

**加载性能指标:**
- 页面加载时间
- 资源加载时间
- DOM 构建时间
- 首次渲染时间

**运行时性能指标:**
- 内存使用情况
- CPU 使用率
- 帧率 (FPS)
- 交互响应时间

**用户体验指标:**
- 首次内容绘制 (FCP)
- 最大内容绘制 (LCP)
- 首次输入延迟 (FID)
- 累积布局偏移 (CLS)

### 1. 页面加载性能监控

```javascript
// 页面性能数据采集
class PageLoadMonitor {
  constructor() {
    this.metrics = {};
    this.collectMetrics();
  }
  
  collectMetrics() {
    // 等待页面加载完成
    if (document.readyState === 'complete') {
      this.collect();
    } else {
      window.addEventListener('load', () => this.collect());
    }
  }
  
  collect() {
    const perfData = performance.getEntriesByType('navigation')[0];
    
    this.metrics = {
      // 页面加载时间
      pageLoadTime: perfData.loadEventEnd - perfData.navigationStart,
      
      // DNS 查询时间
      dnsLookupTime: perfData.domainLookupEnd - perfData.domainLookupStart,
      
      // TCP 连接时间
      tcpConnectionTime: perfData.connectEnd - perfData.connectStart,
      
      // 请求响应时间
      requestTime: perfData.responseEnd - perfData.requestStart,
      
      // DOM 解析时间
      domParsingTime: perfData.domComplete - perfData.domInteractive,
      
      // 资源加载时间
      resourceLoadTime: perfData.loadEventEnd - perfData.domContentLoadedEventEnd,
      
      // 白屏时间
      whiteScreenTime: perfData.responseStart - perfData.navigationStart,
      
      // 首屏时间
      firstScreenTime: this.getFirstScreenTime()
    };
  }
  
  getFirstScreenTime() {
    // 使用 Performance Observer 获取首屏时间
    return new Promise((resolve) => {
      if (!('PerformanceObserver' in window)) {
        resolve(0);
        return;
      }
      
      const observer = new PerformanceObserver((list) => {
        const entries = list.getEntries();
        const firstContentfulPaint = entries.find(
          entry => entry.name === 'first-contentful-paint'
        );
        resolve(firstContentfulPaint ? firstContentfulPaint.startTime : 0);
      });
      
      observer.observe({ entryTypes: ['paint'] });
    });
  }
  
  getMetrics() {
    return this.metrics;
  }
}

// 使用示例
const pageMonitor = new PageLoadMonitor();
pageMonitor.getFirstScreenTime().then(firstScreenTime => {
  console.log('首屏时间:', firstScreenTime);
});
```

### 2. 资源加载监控

```javascript
// 资源加载性能监控
class ResourceMonitor {
  constructor() {
    this.resources = [];
    this.collectResources();
  }
  
  collectResources() {
    // 收集已加载的资源
    this.resources = performance.getEntriesByType('resource');
    
    // 监控新加载的资源
    if ('PerformanceObserver' in window) {
      const observer = new PerformanceObserver((list) => {
        const newResources = list.getEntries();
        this.resources.push(...newResources);
        this.analyzeResources(newResources);
      });
      
      observer.observe({ entryTypes: ['resource'] });
    }
  }
  
  analyzeResources(resources) {
    const analysis = {
      total: resources.length,
      byType: {},
      slowResources: [],
      largeResources: [],
      errors: []
    };
    
    resources.forEach(resource => {
      // 按类型统计
      const type = this.getResourceType(resource);
      analysis.byType[type] = (analysis.byType[type] || 0) + 1;
      
      // 识别慢资源
      const duration = resource.duration;
      if (duration > 1000) { // 超过1秒
        analysis.slowResources.push({
          name: resource.name,
          type: type,
          duration: duration
        });
      }
      
      // 识别大资源
      const size = resource.transferSize;
      if (size > 500 * 1024) { // 超过500KB
        analysis.largeResources.push({
          name: resource.name,
          type: type,
          size: size
        });
      }
    });
    
    return analysis;
  }
  
  getResourceType(resource) {
    const url = resource.name;
    if (url.match(/\.(js)$/)) return 'javascript';
    if (url.match(/\.(css)$/)) return 'css';
    if (url.match(/\.(png|jpg|jpeg|gif|webp|svg)$/)) return 'image';
    if (url.match(/\.(woff|woff2|ttf|otf)$/)) return 'font';
    return 'other';
  }
  
  getResourceStats() {
    return {
      count: this.resources.length,
      totalSize: this.resources.reduce((sum, r) => sum + r.transferSize, 0),
      totalTime: this.resources.reduce((sum, r) => sum + r.duration, 0),
      analysis: this.analyzeResources(this.resources)
    };
  }
}

// 使用示例
const resourceMonitor = new ResourceMonitor();
console.log('资源统计:', resourceMonitor.getResourceStats());
```

### 3. 用户交互监控

```javascript
// 用户交互性能监控
class InteractionMonitor {
  constructor() {
    this.interactions = [];
    this.init();
  }
  
  init() {
    // 监控点击事件
    document.addEventListener('click', this.handleClick.bind(this), true);
    
    // 监控滚动事件
    document.addEventListener('scroll', this.handleScroll.bind(this), { passive: true });
    
    // 监控输入事件
    document.addEventListener('input', this.handleInput.bind(this), true);
  }
  
  handleClick(event) {
    const target = event.target;
    const interaction = {
      type: 'click',
      element: this.getElementSelector(target),
      timestamp: Date.now(),
      page: window.location.href,
      coordinates: {
        x: event.clientX,
        y: event.clientY
      }
    };
    
    this.interactions.push(interaction);
    this.recordInteractionDelay(interaction);
  }
  
  handleScroll(event) {
    const scrollDepth = this.getScrollDepth();
    
    if (scrollDepth > 0) {
      const interaction = {
        type: 'scroll',
        timestamp: Date.now(),
        page: window.location.href,
        scrollDepth: scrollDepth
      };
      
      this.interactions.push(interaction);
    }
  }
  
  handleInput(event) {
    const target = event.target;
    if (target.tagName === 'INPUT' || target.tagName === 'TEXTAREA') {
      const interaction = {
        type: 'input',
        element: this.getElementSelector(target),
        inputType: event.inputType || 'unknown',
        timestamp: Date.now(),
        page: window.location.href
      };
      
      this.interactions.push(interaction);
    }
  }
  
  recordInteractionDelay(interaction) {
    // 记录交互延迟
    const start = performance.now();
    
    requestAnimationFrame(() => {
      const delay = performance.now() - start;
      interaction.delay = delay;
      
      // 如果延迟过长，上报警告
      if (delay > 100) {
        this.reportSlowInteraction(interaction);
      }
    });
  }
  
  getScrollDepth() {
    const scrollTop = window.pageYOffset || document.documentElement.scrollTop;
    const docHeight = document.documentElement.scrollHeight - window.innerHeight;
    return Math.round((scrollTop / docHeight) * 100);
  }
  
  getElementSelector(element) {
    // 生成元素选择器用于标识
    if (element.id) {
      return `#${element.id}`;
    }
    
    if (element.className) {
      return `${element.tagName.toLowerCase()}.${element.className.split(' ')[0]}`;
    }
    
    return element.tagName.toLowerCase();
  }
  
  reportSlowInteraction(interaction) {
    console.warn('慢交互检测:', interaction);
    
    // 上报到服务器
    this.reportToServer({
      type: 'slow_interaction',
      data: interaction
    });
  }
  
  reportToServer(data) {
    fetch('/api/performance', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(data)
    }).catch(err => console.error('上报失败:', err));
  }
  
  getInteractionStats() {
    return {
      total: this.interactions.length,
      byType: this.groupByType(),
      averageDelay: this.calculateAverageDelay()
    };
  }
  
  groupByType() {
    return this.interactions.reduce((acc, interaction) => {
      acc[interaction.type] = (acc[interaction.type] || 0) + 1;
      return acc;
    }, {});
  }
  
  calculateAverageDelay() {
    const delays = this.interactions
      .filter(i => i.delay)
      .map(i => i.delay);
    
    if (delays.length === 0) return 0;
    
    return delays.reduce((sum, delay) => sum + delay, 0) / delays.length;
  }
}
```

### 4. Web Vitals 监控

```javascript
// Web Vitals 核心指标监控
class WebVitalsMonitor {
  constructor() {
    this.metrics = {};
    this.init();
  }
  
  init() {
    // 监控 LCP
    this.observeLCP();
    // 监控 FID
    this.observeFID();
    // 监控 CLS
    this.observeCLS();
    // 监控 FCP
    this.observeFCP();
    // 监控 TTFB
    this.observeTTFB();
  }
  
  observeLCP() {
    if (!('PerformanceObserver' in window)) return;
    
    const observer = new PerformanceObserver((list) => {
      const entries = list.getEntries();
      const lastEntry = entries[entries.length - 1];
      
      this.metrics.lcp = {
        value: lastEntry.startTime,
        rating: this.rateLCP(lastEntry.startTime),
        element: lastEntry.element?.outerHTML?.substring(0, 100)
      };
      
      this.reportMetric('LCP', this.metrics.lcp);
    });
    
    observer.observe({ entryTypes: ['largest-contentful-paint'] });
  }
  
  observeFID() {
    if (!('PerformanceObserver' in window)) return;
    
    const observer = new PerformanceObserver((list) => {
      const entries = list.getEntries();
      const firstEntry = entries[0];
      
      this.metrics.fid = {
        value: firstEntry.processingStart - firstEntry.startTime,
        rating: this.rateFID(firstEntry.processingStart - firstEntry.startTime),
        eventName: firstEntry.name
      };
      
      this.reportMetric('FID', this.metrics.fid);
    });
    
    observer.observe({ entryTypes: ['first-input'] });
  }
  
  observeCLS() {
    if (!('PerformanceObserver' in window)) return;
    
    let clsValue = 0;
    let sessionValue = 0;
    let sessionEntries = [];
    
    const observer = new PerformanceObserver((list) => {
      for (const entry of list.getEntries()) {
        // 只计算没有最近用户输入的条目
        if (!entry.hadRecentInput) {
          const firstSessionEntry = sessionEntries[0];
          const lastSessionEntry = sessionEntries[sessionEntries.length - 1];
          
          // 如果条目与会话的第一个条目相隔超过 1 秒，或者
          // 与会话的最后一个条目相隔超过 5 秒，开始新会话
          if (sessionValue &&
              (entry.startTime - lastSessionEntry.startTime >= 5000 ||
               entry.startTime - firstSessionEntry.startTime >= 1000)) {
            this.reportMetric('CLS', {
              value: sessionValue,
              rating: this.rateCLS(sessionValue)
            });
            
            sessionValue = 0;
            sessionEntries = [];
          }
          
          sessionValue += entry.value;
          sessionEntries.push(entry);
          clsValue += entry.value;
        }
      }
    });
    
    observer.observe({ entryTypes: ['layout-shift'] });
    
    // 页面隐藏时上报最终的 CLS 值
    document.addEventListener('visibilitychange', () => {
      if (document.visibilityState === 'hidden') {
        this.metrics.cls = {
          value: clsValue,
          rating: this.rateCLS(clsValue)
        };
        this.reportMetric('CLS', this.metrics.cls);
      }
    });
  }
  
  observeFCP() {
    if (!('PerformanceObserver' in window)) return;
    
    const observer = new PerformanceObserver((list) => {
      const entries = list.getEntries();
      const fcpEntry = entries.find(entry => entry.name === 'first-contentful-paint');
      
      if (fcpEntry) {
        this.metrics.fcp = {
          value: fcpEntry.startTime,
          rating: this.rateFCP(fcpEntry.startTime)
        };
        
        this.reportMetric('FCP', this.metrics.fcp);
        observer.disconnect();
      }
    });
    
    observer.observe({ entryTypes: ['paint'] });
  }
  
  observeTTFB() {
    const navigationEntry = performance.getEntriesByType('navigation')[0];
    
    if (navigationEntry) {
      const ttfb = navigationEntry.responseStart - navigationEntry.requestStart;
      
      this.metrics.ttfb = {
        value: ttfb,
        rating: this.rateTTFB(ttfb)
      };
      
      this.reportMetric('TTFB', this.metrics.ttfb);
    }
  }
  
  // 评分函数
  rateLCP(value) {
    if (value <= 2500) return 'good';
    if (value <= 4000) return 'needs-improvement';
    return 'poor';
  }
  
  rateFID(value) {
    if (value <= 100) return 'good';
    if (value <= 300) return 'needs-improvement';
    return 'poor';
  }
  
  rateCLS(value) {
    if (value <= 0.1) return 'good';
    if (value <= 0.25) return 'needs-improvement';
    return 'poor';
  }
  
  rateFCP(value) {
    if (value <= 1800) return 'good';
    if (value <= 3000) return 'needs-improvement';
    return 'poor';
  }
  
  rateTTFB(value) {
    if (value <= 600) return 'good';
    if (value <= 1500) return 'needs-improvement';
    return 'poor';
  }
  
  reportMetric(name, metric) {
    console.log(`${name}:`, metric);
    
    // 上报到服务器
    fetch('/api/web-vitals', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        metric: name,
        value: metric.value,
        rating: metric.rating,
        url: window.location.href,
        timestamp: Date.now()
      })
    }).catch(err => console.error('上报失败:', err));
  }
  
  getMetrics() {
    return this.metrics;
  }
}
```

### 综合监控系统

```javascript
// 综合性能监控系统
class PerformanceMonitoringSystem {
  constructor(config) {
    this.config = {
      sampleRate: 0.1, // 10% 采样率
      reportUrl: '/api/performance',
      batchSize: 10,
      reportInterval: 30000, // 30秒
      ...config
    };
    
    this.metricsQueue = [];
    this.isEnabled = Math.random() < this.config.sampleRate;
    
    if (this.isEnabled) {
      this.init();
    }
  }
  
  init() {
    // 初始化各个监控模块
    this.pageMonitor = new PageLoadMonitor();
    this.resourceMonitor = new ResourceMonitor();
    this.interactionMonitor = new InteractionMonitor();
    this.webVitalsMonitor = new WebVitalsMonitor();
    
    // 定期上报
    setInterval(() => this.flushMetrics(), this.config.reportInterval);
    
    // 页面卸载时上报剩余数据
    window.addEventListener('beforeunload', () => this.flushMetrics());
    
    // 监控内存使用
    this.monitorMemory();
    
    // 监控长任务
    this.monitorLongTasks();
  }
  
  monitorMemory() {
    if ('memory' in performance) {
      setInterval(() => {
        const memory = performance.memory;
        
        this.addMetric({
          type: 'memory',
          value: {
            usedJSHeapSize: memory.usedJSHeapSize,
            totalJSHeapSize: memory.totalJSHeapSize,
            jsHeapSizeLimit: memory.jsHeapSizeLimit,
            usageRatio: memory.usedJSHeapSize / memory.jsHeapSizeLimit
          },
          timestamp: Date.now()
        });
        
        // 内存使用率过高时报警
        if (memory.usedJSHeapSize / memory.jsHeapSizeLimit > 0.9) {
          this.reportMemoryWarning(memory);
        }
      }, 10000); // 每10秒检查一次
    }
  }
  
  monitorLongTasks() {
    if (!('PerformanceObserver' in window)) return;
    
    const observer = new PerformanceObserver((list) => {
      for (const entry of list.getEntries()) {
        if (entry.duration > 50) { // 超过50ms的任务
          this.addMetric({
            type: 'long_task',
            value: {
              duration: entry.duration,
              startTime: entry.startTime,
              name: entry.name
            },
            timestamp: Date.now()
          });
          
          this.reportLongTask(entry);
        }
      }
    });
    
    observer.observe({ entryTypes: ['longtask'] });
  }
  
  addMetric(metric) {
    this.metricsQueue.push(metric);
    
    // 达到批量大小立即上报
    if (this.metricsQueue.length >= this.config.batchSize) {
      this.flushMetrics();
    }
  }
  
  flushMetrics() {
    if (this.metricsQueue.length === 0) return;
    
    const metrics = [...this.metricsQueue];
    this.metricsQueue = [];
    
    // 收集所有监控数据
    const reportData = {
      pageMetrics: this.pageMonitor.getMetrics(),
      resourceMetrics: this.resourceMonitor.getResourceStats(),
      interactionMetrics: this.interactionMonitor.getInteractionStats(),
      webVitals: this.webVitalsMonitor.getMetrics(),
      additionalMetrics: metrics,
      context: {
        url: window.location.href,
        userAgent: navigator.userAgent,
        screen: {
          width: window.screen.width,
          height: window.screen.height
        },
        viewport: {
          width: window.innerWidth,
          height: window.innerHeight
        },
        connection: this.getConnectionInfo()
      },
      timestamp: Date.now()
    };
    
    // 上报到服务器
    this.sendReport(reportData);
  }
  
  getConnectionInfo() {
    if ('connection' in navigator) {
      const connection = navigator.connection;
      return {
        effectiveType: connection.effectiveType,
        downlink: connection.downlink,
        rtt: connection.rtt,
        saveData: connection.saveData
      };
    }
    return null;
  }
  
  sendReport(data) {
    // 使用 sendBeacon 确保数据能够发送
    if ('sendBeacon' in navigator) {
      const blob = new Blob([JSON.stringify(data)], { 
        type: 'application/json' 
      });
      navigator.sendBeacon(this.config.reportUrl, blob);
    } else {
      // 回退到 fetch
      fetch(this.config.reportUrl, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(data),
        keepalive: true
      }).catch(err => console.error('上报失败:', err));
    }
  }
  
  reportMemoryWarning(memory) {
    console.warn('内存使用率过高:', memory);
    
    this.addMetric({
      type: 'memory_warning',
      value: memory,
      timestamp: Date.now()
    });
  }
  
  reportLongTask(entry) {
    console.warn('检测到长任务:', entry);
    
    this.addMetric({
      type: 'long_task_warning',
      value: entry,
      timestamp: Date.now()
    });
  }
  
  getPerformanceReport() {
    return {
      pageMetrics: this.pageMonitor.getMetrics(),
      resourceMetrics: this.resourceMonitor.getResourceStats(),
      interactionMetrics: this.interactionMonitor.getInteractionStats(),
      webVitals: this.webVitalsMonitor.getMetrics(),
      queuedMetrics: this.metricsQueue
    };
  }
}

  batchSize: 10
});
```

```javascript
// 隐私保护的数据处理
class PrivacyAwareMonitor {
  constructor() {
    this.sensitiveFields = ['password', 'creditCard', 'ssn'];
    this.sensitivePatterns = [
      /\b\d{16}\b/, // 信用卡号
      /\b\d{3}-\d{2}-\d{4}\b/ // SSN
    ];
  }
  
  sanitizeData(data) {
    const sanitized = JSON.parse(JSON.stringify(data));
    
    return this.removeSensitiveInfo(sanitized);
  }
  
  removeSensitiveInfo(obj) {
    for (const key in obj) {
      if (this.isSensitiveKey(key)) {
        obj[key] = '[REDACTED]';
      } else if (typeof obj[key] === 'string' && this.containsSensitiveInfo(obj[key])) {
        obj[key] = '[REDACTED]';
      } else if (typeof obj[key] === 'object' && obj[key] !== null) {
        this.removeSensitiveInfo(obj[key]);
      }
    }
    return obj;
  }
  
  isSensitiveKey(key) {
    return this.sensitiveFields.some(field => 
      key.toLowerCase().includes(field.toLowerCase())
    );
  }
  
  containsSensitiveInfo(value) {
    return this.sensitivePatterns.some(pattern => 
      pattern.test(value)
    );
  }
}
```

### 2. 性能开销控制

```javascript
// 控制监控自身的性能开销
class LightweightMonitor {
  constructor() {
    this.metrics = [];
    this.maxMetrics = 100; // 限制存储的指标数量
    this.throttleTime = 1000; // 节流时间
    this.lastReportTime = 0;
  }
  
  addMetric(metric) {
    // 限制存储数量
    if (this.metrics.length >= this.maxMetrics) {
      this.metrics.shift(); // 移除最旧的数据
    }
    
    this.metrics.push(metric);
    
    // 节流上报
    const now = Date.now();
    if (now - this.lastReportTime >= this.throttleTime) {
      this.flushMetrics();
      this.lastReportTime = now;
    }
  }
  
  flushMetrics() {
    if (this.metrics.length === 0) return;
    
    const metrics = [...this.metrics];
    this.metrics = [];
    
    // 使用 requestIdleCallback 在空闲时上报
    if ('requestIdleCallback' in window) {
      requestIdleCallback(() => this.sendMetrics(metrics), { timeout: 2000 });
    } else {
      setTimeout(() => this.sendMetrics(metrics), 0);
    }
  }
  
  sendMetrics(metrics) {
    // 批量上报，减少网络请求
    fetch('/api/performance', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ metrics }),
      keepalive: true
    }).catch(err => console.error('上报失败:', err));
  }
}
```

```javascript
// 完善的错误处理
class SafeMonitor {
  constructor() {
    this.errorBoundary = this.createErrorBoundary();
  }
  
  createErrorBoundary() {
    return {
      try: (fn, errorHandler) => {
        try {
          return fn();
        } catch (error) {
          if (errorHandler) {
            errorHandler(error);
          }
          this.handleMonitorError(error);
          return null;
        }
      },
      
      async tryAsync: async (fn, errorHandler) => {
        try {
          return await fn();
        } catch (error) {
          if (errorHandler) {
            await errorHandler(error);
          }
          this.handleMonitorError(error);
          return null;
        }
      }
    };
  }
  
  handleMonitorError(error) {
    // 监控代码本身的错误
    console.error('监控错误:', error);
    
    // 上报监控错误
    this.reportError({
      type: 'monitor_error',
      error: error.message,
      stack: error.stack,
      timestamp: Date.now()
    });
  }
  
  reportError(errorData) {
    // 简单的上报，避免递归错误
    if (navigator.sendBeacon) {
      const blob = new Blob([JSON.stringify(errorData)], {
        type: 'application/json'
      });
      navigator.sendBeacon('/api/monitor-error', blob);
    }
  }
}
```

## 总结

前端性能监控是保证应用质量的重要手段，关键点包括：

1. **核心指标监控**: 重点关注 Web Vitals 指标（LCP、FID、CLS）
2. **多维度数据**: 采集加载性能、运行时性能、用户体验等多维度数据
3. **智能分析**: 自动识别性能瓶颈和异常情况
4. **合理采样**: 通过采样减少性能开销和存储成本
5. **隐私保护**: 妥善处理用户数据，避免敏感信息泄露
6. **错误处理**: 完善的错误处理机制，避免监控代码影响业务

实施建议：

- 从关键指标开始，逐步完善监控体系
- 设置合理的阈值和报警机制
- 定期分析监控数据，指导性能优化
- 关注监控工具自身的性能开销
- 建立完善的错误处理和降级机制
- 结合业务指标，评估性能优化的实际效果

通过持续的性能监控，我们可以及时发现问题、优化用户体验，最终提升产品的整体质量。
