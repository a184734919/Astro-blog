## 概述

前端性能优化是提升用户体验的核心手段，直接影响页面加载速度、交互响应时间和整体性能。性能优化不是一次性工作，而是需要持续关注和改进的系统工程。

## 核心概念

### 性能指标体系

**核心性能指标:**

- **FP (First Paint)**: 首次绘制，用户第一次看到内容
- **FCP (First Contentful Paint)**: 首次内容绘制，首次有意义的绘制
- **LCP (Largest Contentful Paint)**: 最大内容绘制，主要内容完成渲染
- **FID (First Input Delay)**: 首次输入延迟，用户首次交互响应时间
- **CLS (Cumulative Layout Shift)**: 累积布局偏移，视觉稳定性指标
- **TTI (Time to Interactive)**: 可交互时间，页面完全可用的时刻

**性能预算:**

#### 虚拟列表优化

```javascript
// 虚拟列表实现
class VirtualList {
  constructor(container, itemHeight, totalItems) {
    this.container = container;
    this.itemHeight = itemHeight;
    this.totalItems = totalItems;
    this.visibleItems = Math.ceil(container.clientHeight / itemHeight) + 2;
    this.startIndex = 0;
    
    this.init();
  }
  
  init() {
    this.container.addEventListener('scroll', this.onScroll.bind(this), {
      passive: true
    });
    this.render();
  }
  
  onScroll() {
    const scrollTop = this.container.scrollTop;
    const startIndex = Math.floor(scrollTop / this.itemHeight);
    
    if (startIndex !== this.startIndex) {
      this.startIndex = startIndex;
      requestAnimationFrame(() => this.render());
    }
  }
  
  render() {
    const fragment = document.createDocumentFragment();
    const offsetY = this.startIndex * this.itemHeight;
    
    for (let i = this.startIndex; i < this.startIndex + this.visibleItems; i++) {
      if (i >= this.totalItems) break;
      
      const item = document.createElement('div');
      item.className = 'list-item';
      item.style.height = `${this.itemHeight}px`;
      item.style.transform = `translateY(${i * this.itemHeight}px)`;
      item.textContent = `Item ${i}`;
      fragment.appendChild(item);
    }
    
    this.container.innerHTML = '';
    this.container.appendChild(fragment);
  }
}
```

### 3. JavaScript 优化

#### 防抖和节流

```javascript
// 防抖函数
function debounce(fn, delay = 300) {
  let timer = null;
  
  return function(...args) {
    clearTimeout(timer);
    timer = setTimeout(() => {
      fn.apply(this, args);
    }, delay);
  };
}

// 节流函数
function throttle(fn, delay = 300) {
  let lastTime = 0;
  let timer = null;
  
  return function(...args) {
    const now = Date.now();
    const remaining = delay - (now - lastTime);
    
    if (remaining <= 0) {
      clearTimeout(timer);
      lastTime = now;
      fn.apply(this, args);
    } else {
      clearTimeout(timer);
      timer = setTimeout(() => {
        lastTime = Date.now();
        fn.apply(this, args);
      }, remaining);
    }
  };
}

// 使用示例
const handleResize = debounce(() => {
  console.log('Resize handler');
}, 200);

const handleScroll = throttle(() => {
  console.log('Scroll handler');
}, 100);

window.addEventListener('resize', handleResize);
window.addEventListener('scroll', handleScroll, { passive: true });
```

## 实际应用

### 综合性能优化方案

```javascript
// 性能监控和优化工具类
class PerformanceOptimizer {
  constructor() {
    this.metrics = {};
    this.init();
  }
  
  init() {
    // 监控页面加载性能
    if ('performance' in window) {
      window.addEventListener('load', () => {
        this.collectMetrics();
        this.reportMetrics();
      });
    }
    
    // 注册 Service Worker
    this.registerServiceWorker();
    
    // 优化资源加载
    this.optimizeResourceLoading();
  }
  
  collectMetrics() {
    const perfData = performance.getEntriesByType('navigation')[0];
    
    this.metrics = {
      // 页面加载时间
      pageLoadTime: perfData.loadEventEnd - perfData.navigationStart,
      // DOM 构建时间
      domReadyTime: perfData.domContentLoadedEventEnd - perfData.navigationStart,
      // 首次绘制
      firstPaint: this.getFirstMetric('paint'),
      // 首次内容绘制
      firstContentfulPaint: this.getFirstMetric('first-contentful-paint'),
      // 最大内容绘制
      largestContentfulPaint: this.getLCP(),
      // 首次输入延迟
      firstInputDelay: this.getFID(),
      // 累积布局偏移
      cumulativeLayoutShift: this.getCLS()
    };
  }
  
  getFirstMetric(name) {
    const paintEntries = performance.getEntriesByType('paint');
    const entry = paintEntries.find(entry => entry.name === name);
    return entry ? entry.startTime : 0;
  }
  
  async getLCP() {
    return new Promise((resolve) => {
      if (!('PerformanceObserver' in window)) {
        resolve(0);
        return;
      }
      
      const observer = new PerformanceObserver((list) => {
        const entries = list.getEntries();
        const lastEntry = entries[entries.length - 1];
        resolve(lastEntry.startTime);
      });
      
      observer.observe({ entryTypes: ['largest-contentful-paint'] });
    });
  }
  
  getFID() {
    return new Promise((resolve) => {
      if (!('PerformanceObserver' in window)) {
        resolve(0);
        return;
      }
      
      const observer = new PerformanceObserver((list) => {
        const entries = list.getEntries();
        if (entries.length > 0) {
          resolve(entries[0].processingStart - entries[0].startTime);
        }
      });
      
      observer.observe({ entryTypes: ['first-input'] });
    });
  }
  
  getCLS() {
    return new Promise((resolve) => {
      if (!('PerformanceObserver' in window)) {
        resolve(0);
        return;
      }
      
      let clsValue = 0;
      const observer = new PerformanceObserver((list) => {
        for (const entry of list.getEntries()) {
          if (!entry.hadRecentInput) {
            clsValue += entry.value;
          }
        }
      });
      
      observer.observe({ entryTypes: ['layout-shift'] });
      
      // 3秒后返回结果
      setTimeout(() => {
        observer.disconnect();
        resolve(clsValue);
      }, 3000);
    });
  }
  
  reportMetrics() {
    // 上报性能指标到服务器
    fetch('/api/performance', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        metrics: this.metrics,
        userAgent: navigator.userAgent,
        url: window.location.href,
        timestamp: Date.now()
      })
    }).catch(err => console.error('Failed to report metrics:', err));
  }
  
  async registerServiceWorker() {
    if ('serviceWorker' in navigator) {
      try {
        await navigator.serviceWorker.register('/sw.js');
        console.log('Service Worker registered successfully');
      } catch (error) {
        console.error('Service Worker registration failed:', error);
      }
    }
  }
  
  optimizeResourceLoading() {
    // 延迟加载非关键图片
    const lazyImages = document.querySelectorAll('img[data-src]');
    
    const imageObserver = new IntersectionObserver((entries, observer) => {
      entries.forEach(entry => {
        if (entry.isIntersecting) {
          const img = entry.target;
          img.src = img.dataset.src;
          img.classList.remove('lazy');
          observer.unobserve(img);
        }
      });
    }, { rootMargin: '200px 0px' });
    
    lazyImages.forEach(img => imageObserver.observe(img));
    
    // 预加载关键资源
    this.preloadCriticalResources();
  }
  
  preloadCriticalResources() {
    const criticalResources = [
      { href: '/styles/critical.css', as: 'style' },
      { href: '/scripts/critical.js', as: 'script' }
    ];
    
    criticalResources.forEach(resource => {
      const link = document.createElement('link');
      link.rel = 'preload';
      link.href = resource.href;
      link.as = resource.as;
      document.head.appendChild(link);
    });
  }
}

// 初始化性能优化器
new PerformanceOptimizer();
```

## 注意事项

### 1. 过度优化

```javascript
// ❌ 过度优化的例子：过早微优化
function calculateSum(arr) {
  // 过于复杂的优化，可能适得其反
  let sum = 0;
  for (let i = 0, len = arr.length; i < len; i++) {
    sum += arr[i];
  }
  return sum;
}

// ✅ 简洁清晰的代码：现代JS引擎已经优化
function calculateSum(arr) {
  return arr.reduce((acc, val) => acc + val, 0);
}
```

## 总结

前端性能优化是一个系统性的工程，需要从多个维度考虑：

1. **网络层面**: 减少请求数量、压缩资源、利用缓存、预加载关键资源
2. **渲染层面**: 优化 DOM 操作、避免布局抖动、使用虚拟列表
3. **JavaScript 层面**: 代码分割、懒加载、防抖节流、优化算法
4. **资源层面**: 图片优化、使用现代格式、响应式加载
5. **监控层面**: 建立性能指标体系、持续监控、及时发现问题

记住这些关键原则：

- 先测量后优化，避免过早优化
- 关注用户体验，而不仅仅是技术指标
- 保持代码可维护性，不要为了优化而牺牲可读性
- 建立性能预算，持续监控和改进
- 考虑渐进增强和优雅降级

性能优化是一个持续的过程，需要不断学习新的技术和最佳实践，并根据实际情况调整优化策略。
