## 概述

前端埋点是指在网页或应用中通过代码手段收集用户行为数据的技术，用于分析用户行为、优化产品体验、支持业务决策。合理的埋点方案能够提供准确、全面的数据支持。

## 核心概念

### 埋点类型分类

**页面浏览埋点:**
- PV (Page View): 页面浏览量
- UV (Unique Visitor): 独立访客数
- 页面停留时长
- 页面跳出率
- 页面转化率

**用户行为埋点:**
- 点击事件
- 表单提交
- 滚动行为
- 悬停时长
- 输入行为

**业务事件埋点:**
- 订单创建
- 支付完成
- 用户注册
- 内容分享
- 搜索行为

**性能埋点:**
- 页面加载时间
- 接口响应时间
- 资源加载时间
- 错误监控数据

## 基本用法

### 1. 页面浏览埋点

```javascript
// 页面浏览追踪
class PageTracker {
  constructor(trackingSystem) {
    this.trackingSystem = trackingSystem;
    this.currentPage = null;
    this.pageStartTime = null;
    this.referrer = document.referrer;
    this.init();
  }
  
  init() {
    // 页面加载时追踪
    if (document.readyState === 'complete') {
      this.trackPageView();
    } else {
      window.addEventListener('load', () => this.trackPageView());
    }
    
    // 监听页面可见性变化
    document.addEventListener('visibilitychange', () => {
      if (document.visibilityState === 'visible') {
        this.trackPageView();
      } else if (document.visibilityState === 'hidden') {
        this.trackPageLeave();
      }
    });
    
    // 监听路由变化（适用于 SPA）
    this.setupRouteTracking();
  }
  
  trackPageView() {
    const pageData = {
      eventType: 'pageview',
      pageUrl: window.location.href,
      pageTitle: document.title,
      pagePath: window.location.pathname,
      referrer: this.referrer,
      timestamp: Date.now(),
      sessionId: this.trackingSystem.sessionId,
      userId: this.trackingSystem.userId,
      userAgent: navigator.userAgent,
      screen: {
        width: window.screen.width,
        height: window.screen.height
      },
      viewport: {
        width: window.innerWidth,
        height: window.innerHeight
      },
      performance: this.getPerformanceData(),
      utm: this.getUTMParameters()
    };
    
    this.currentPage = pageData;
    this.pageStartTime = Date.now();
    this.trackingSystem.track(pageData);
  }
  
  trackPageLeave() {
    if (!this.currentPage || !this.pageStartTime) return;
    
    const pageData = {
      eventType: 'pageleave',
      pageUrl: this.currentPage.pageUrl,
      pageTitle: this.currentPage.pageTitle,
      duration: Date.now() - this.pageStartTime,
      scrollDepth: this.getScrollDepth(),
      timestamp: Date.now(),
      sessionId: this.trackingSystem.sessionId,
      userId: this.trackingSystem.userId
    };
    
    this.trackingSystem.track(pageData);
    this.currentPage = null;
    this.pageStartTime = null;
  }
  
  setupRouteTracking() {
    // 拦截 pushState 和 replaceState
    const originalPushState = history.pushState;
    const originalReplaceState = history.replaceState;
    
    history.pushState = function(...args) {
      originalPushState.apply(this, args);
      window.dispatchEvent(new Event('routechange'));
    };
    
    history.replaceState = function(...args) {
      originalReplaceState.apply(this, args);
      window.dispatchEvent(new Event('routechange'));
    };
    
    // 监听路由变化
    window.addEventListener('routechange', () => {
      setTimeout(() => this.trackPageView(), 100);
    });
    
    // 监听 popstate 事件
    window.addEventListener('popstate', () => {
      setTimeout(() => this.trackPageView(), 100);
    });
  }
  
  getPerformanceData() {
    if (!('performance' in window)) return null;
    
    const perfData = performance.getEntriesByType('navigation')[0];
    if (!perfData) return null;
    
    return {
      pageLoadTime: perfData.loadEventEnd - perfData.navigationStart,
      domReadyTime: perfData.domContentLoadedEventEnd - perfData.navigationStart,
      firstPaint: this.getFirstMetric('paint'),
      firstContentfulPaint: this.getFirstMetric('first-contentful-paint')
    };
  }
  
  getFirstMetric(name) {
    const paintEntries = performance.getEntriesByType('paint');
    const entry = paintEntries.find(entry => entry.name === name);
    return entry ? entry.startTime : 0;
  }
  
  getScrollDepth() {
    const scrollTop = window.pageYOffset || document.documentElement.scrollTop;
    const docHeight = document.documentElement.scrollHeight - window.innerHeight;
    return docHeight > 0 ? Math.round((scrollTop / docHeight) * 100) : 0;
  }
  
  getUTMParameters() {
    const urlParams = new URLSearchParams(window.location.search);
    return {
      source: urlParams.get('utm_source'),
      medium: urlParams.get('utm_medium'),
      campaign: urlParams.get('utm_campaign'),
      term: urlParams.get('utm_term'),
      content: urlParams.get('utm_content')
    };
  }
}
```

### 2. 用户行为埋点

```javascript
// 用户行为追踪
class EventTracker {
  constructor(trackingSystem) {
    this.trackingSystem = trackingSystem;
    this.init();
  }
  
  init() {
    // 自动追踪点击事件
    this.setupClickTracking();
    // 自动追踪表单提交
    this.setupFormTracking();
    // 自动追踪滚动行为
    this.setupScrollTracking();
    // 自动追踪输入行为
    this.setupInputTracking();
  }
  
  setupClickTracking() {
    document.addEventListener('click', (event) => {
      const target = event.target;
      const trackingData = this.getElementTrackingData(target);
      
      if (trackingData) {
        const eventData = {
          eventType: 'click',
          element: trackingData.selector,
          elementId: trackingData.id,
          elementClass: trackingData.className,
          elementText: this.getElementText(target),
          coordinates: {
            x: event.clientX,
            y: event.clientY
          },
          pageUrl: window.location.href,
          timestamp: Date.now(),
          sessionId: this.trackingSystem.sessionId,
          userId: this.trackingSystem.userId
        };
        
        this.trackingSystem.track(eventData);
      }
    }, true);
  }
  
  setupFormTracking() {
    document.addEventListener('submit', (event) => {
      const form = event.target;
      const trackingData = this.getElementTrackingData(form);
      
      const formData = {
        eventType: 'form_submit',
        formId: trackingData.id,
        formName: form.name || trackingData.selector,
        formAction: form.action,
        formMethod: form.method,
        fields: this.getFormData(form),
        pageUrl: window.location.href,
        timestamp: Date.now(),
        sessionId: this.trackingSystem.sessionId,
        userId: this.trackingSystem.userId
      };
      
      this.trackingSystem.track(formData);
    }, true);
  }
  
  setupScrollTracking() {
    let lastScrollDepth = 0;
    const scrollDepths = [25, 50, 75, 90, 100];
    
    const handleScroll = () => {
      const scrollDepth = this.getScrollDepth();
      
      // 检查是否到达新的滚动深度
      scrollDepths.forEach(depth => {
        if (scrollDepth >= depth && lastScrollDepth < depth) {
          const eventData = {
            eventType: 'scroll',
            scrollDepth: depth,
            pageUrl: window.location.href,
            timestamp: Date.now(),
            sessionId: this.trackingSystem.sessionId,
            userId: this.trackingSystem.userId
          };
          
          this.trackingSystem.track(eventData);
        }
      });
      
      lastScrollDepth = scrollDepth;
    };
    
    // 使用节流优化滚动事件
    let scrollTimeout;
    document.addEventListener('scroll', () => {
      if (scrollTimeout) {
        return;
      }
      
      scrollTimeout = setTimeout(() => {
        handleScroll();
        scrollTimeout = null;
      }, 100);
    }, { passive: true });
  }
  
  setupInputTracking() {
    const trackedInputs = new Set();
    
    document.addEventListener('input', (event) => {
      const target = event.target;
      if (target.tagName === 'INPUT' || target.tagName === 'TEXTAREA') {
        const inputId = target.id || target.name || this.getElementSelector(target);
        
        if (!trackedInputs.has(inputId)) {
          trackedInputs.add(inputId);
          
          const eventData = {
            eventType: 'input',
            inputId: inputId,
            inputType: target.type || 'text',
            inputName: target.name,
            inputValue: this.sanitizeInputValue(target.value),
            pageUrl: window.location.href,
            timestamp: Date.now(),
            sessionId: this.trackingSystem.sessionId,
            userId: this.trackingSystem.userId
          };
          
          this.trackingSystem.track(eventData);
        }
      }
    }, true);
  }
  
  getElementTrackingData(element) {
    // 检查是否应该追踪此元素
    if (!this.shouldTrackElement(element)) {
      return null;
    }
    
    return {
      selector: this.getElementSelector(element),
      id: element.id,
      className: element.className,
      tagName: element.tagName.toLowerCase()
    };
  }
  
  shouldTrackElement(element) {
    // 检查元素是否设置了 data-track 属性
    if (element.hasAttribute('data-track')) {
      return element.getAttribute('data-track') !== 'false';
    }
    
    // 检查父元素是否设置了追踪属性
    const parent = element.closest('[data-track]');
    if (parent) {
      return parent.getAttribute('data-track') !== 'false';
    }
    
    // 默认追踪某些元素类型
    const trackableTags = ['A', 'BUTTON', 'INPUT', 'SELECT', 'TEXTAREA'];
    return trackableTags.includes(element.tagName);
  }
  
  getElementSelector(element) {
    if (element.id) {
      return `#${element.id}`;
    }
    
    if (element.className) {
      return `${element.tagName.toLowerCase()}.${element.className.split(' ')[0]}`;
    }
    
    return element.tagName.toLowerCase();
  }
  
  getElementText(element) {
    return element.textContent?.trim().substring(0, 100) || '';
  }
  
  getFormData(form) {
    const formData = new FormData(form);
    const fields = {};
    
    for (let [key, value] of formData.entries()) {
      // 过滤敏感字段
      if (!this.isSensitiveField(key)) {
        fields[key] = this.sanitizeInputValue(value);
      }
    }
    
    return fields;
  }
  
  isSensitiveField(fieldName) {
    const sensitivePatterns = ['password', 'credit', 'ssn', 'secret'];
    return sensitivePatterns.some(pattern => 
      fieldName.toLowerCase().includes(pattern)
    );
  }
  
  sanitizeInputValue(value) {
    // 简单的值清理，可以根据需要扩展
    const stringValue = String(value).trim();
    return stringValue.substring(0, 255); // 限制长度
  }
  
  getScrollDepth() {
    const scrollTop = window.pageYOffset || document.documentElement.scrollTop;
    const docHeight = document.documentElement.scrollHeight - window.innerHeight;
    return docHeight > 0 ? Math.round((scrollTop / docHeight) * 100) : 0;
  }
}
```

```javascript
  // 邮箱哈希（保护隐私）
  hashEmail(email) {
    if (!email) return null;
    
    // 简单的哈希函数，实际应用中应该使用更安全的哈希算法
    let hash = 0;
    const lowerEmail = email.toLowerCase().trim();
    
    for (let i = 0; i < lowerEmail.length; i++) {
      const char = lowerEmail.charCodeAt(i);
      hash = ((hash << 5) - hash) + char;
      hash = hash & hash; // 转换为32位整数
    }
    
    return hash.toString(16);
  }
}
```

### 4. 数据上报机制

```javascript
// 数据上报系统
class DataReporter {
  constructor(config) {
    this.config = {
      reportUrl: '/api/tracking',
      batchSize: 10,
      batchInterval: 5000,
      retryAttempts: 3,
      retryDelay: 1000,
      ...config
    };
    
    this.eventQueue = [];
    this.isReporting = false;
    this.init();
  }
  
  init() {
    // 定期批量上报
    setInterval(() => this.flushQueue(), this.config.batchInterval);
    
    // 页面卸载时上报剩余数据
    window.addEventListener('beforeunload', () => this.flushQueue());
    
    // 页面隐藏时上报
    document.addEventListener('visibilitychange', () => {
      if (document.visibilityState === 'hidden') {
        this.flushQueue();
      }
    });
  }
  
  addEvent(event) {
    this.eventQueue.push(event);
    
    // 达到批量大小立即上报
    if (this.eventQueue.length >= this.config.batchSize) {
      this.flushQueue();
    }
  }
  
  async flushQueue() {
    if (this.eventQueue.length === 0 || this.isReporting) {
      return;
    }
    
    this.isReporting = true;
    const events = [...this.eventQueue];
    this.eventQueue = [];
    
    try {
      await this.reportEvents(events);
    } catch (error) {
      // 上报失败，重新加入队列
      this.eventQueue.unshift(...events);
      console.error('上报失败:', error);
      
      // 尝试使用 sendBeacon
      this.trySendBeacon(events);
    } finally {
      this.isReporting = false;
    }
  }
  
  async reportEvents(events) {
    for (let attempt = 1; attempt <= this.config.retryAttempts; attempt++) {
      try {
        const response = await fetch(this.config.reportUrl, {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json',
            'X-Tracking-ID': this.getTrackingId()
          },
          body: JSON.stringify({
            events: events,
            timestamp: Date.now(),
            userAgent: navigator.userAgent
          }),
          keepalive: true
        });
        
        if (response.ok) {
          return;
        }
        
        throw new Error(`HTTP ${response.status}: ${response.statusText}`);
      } catch (error) {
        if (attempt === this.config.retryAttempts) {
          throw error;
        }
        
        // 等待后重试
        await new Promise(resolve => 
          setTimeout(resolve, this.config.retryDelay * Math.pow(2, attempt - 1))
        );
      }
    }
  }
  
  trySendBeacon(events) {
    if ('sendBeacon' in navigator) {
      const data = JSON.stringify({
        events: events,
        timestamp: Date.now(),
        userAgent: navigator.userAgent
      });
      
      const blob = new Blob([data], { type: 'application/json' });
      const success = navigator.sendBeacon(this.config.reportUrl, blob);
      
      if (!success) {
        console.error('sendBeacon 失败');
      }
    }
  }
  
  getTrackingId() {
    // 生成或获取追踪 ID
    let trackingId = localStorage.getItem('tracking_id');
    
    if (!trackingId) {
      trackingId = this.generateTrackingId();
      localStorage.setItem('tracking_id', trackingId);
    }
    
    return trackingId;
  }
  
  generateTrackingId() {
    return 'xxxxxxxx-xxxx-4xxx-yxxx-xxxxxxxxxxxx'.replace(/[xy]/g, function(c) {
      const r = Math.random() * 16 | 0;
      const v = c === 'x' ? r : (r & 0x3 | 0x8);
      return v.toString(16);
    });
  }
  
  // 获取上报统计信息
  getStats() {
    return {
      queueSize: this.eventQueue.length,
      isReporting: this.isReporting,
      totalEvents: this.getTotalEventsCount()
    };
  }
  
  getTotalEventsCount() {
    const count = parseInt(localStorage.getItem('total_events_count') || '0');
    return count;
  }
}
```

```javascript
// 综合埋点系统
class ComprehensiveTrackingSystem {
  constructor(config = {}) {
    this.config = {
      trackingId: config.trackingId || '',
      reportUrl: config.reportUrl || '/api/tracking',
      enableAutoTracking: config.enableAutoTracking !== false,
      enablePageTracking: config.enablePageTracking !== false,
      enableEventTracking: config.enableEventTracking !== false,
      batchSize: config.batchSize || 10,
      batchInterval: config.batchInterval || 5000,
      ...config
    };
    
    // 生成会话 ID
    this.sessionId = this.generateSessionId();
    this.userId = this.getUserId();
    
    // 初始化各个模块
    this.dataReporter = new DataReporter(this.config);
    this.businessTracker = new BusinessEventTracker(this);
    
    if (this.config.enableAutoTracking) {
      if (this.config.enablePageTracking) {
        this.pageTracker = new PageTracker(this);
      }
      
      if (this.config.enableEventTracking) {
        this.eventTracker = new EventTracker(this);
      }
    }
  }
  
  // 生成会话 ID
  generateSessionId() {
    // 尝试从 sessionStorage 获取现有会话
    let sessionId = sessionStorage.getItem('session_id');
    
    if (!sessionId) {
      sessionId = 'session_' + Date.now() + '_' + Math.random().toString(36).substr(2, 9);
      sessionStorage.setItem('session_id', sessionId);
    }
    
    return sessionId;
  }
  
  // 获取用户 ID
  getUserId() {
    // 从 localStorage 获取或设置用户 ID
    let userId = localStorage.getItem('user_id');
    
    if (!userId) {
      userId = 'user_' + Date.now() + '_' + Math.random().toString(36).substr(2, 9);
      localStorage.setItem('user_id', userId);
    }
    
    return userId;
  }
  
  // 设置用户 ID
  setUserId(userId) {
    this.userId = userId;
    localStorage.setItem('user_id', userId);
  }
  
  // 追踪事件
  track(event) {
    // 添加通用字段
    const enrichedEvent = {
      ...event,
      trackingId: this.config.trackingId,
      sessionId: this.sessionId,
      userId: this.userId
    };
    
    // 添加到上报队列
    this.dataReporter.addEvent(enrichedEvent);
  }
  
  // 页面浏览追踪
  trackPageView(pageData = {}) {
    const eventData = {
      eventType: 'pageview',
      pageUrl: pageData.url || window.location.href,
      pageTitle: pageData.title || document.title,
      pagePath: pageData.path || window.location.pathname,
      referrer: document.referrer,
      timestamp: Date.now(),
      customData: pageData.customData
    };
    
    this.track(eventData);
  }
  
  // 自定义事件追踪
```

### 1. 隐私保护

```javascript
// 隐私保护的埋点实现
class PrivacyAwareTracking {
  constructor() {
    this.consentGiven = this.checkConsent();
    this.sensitiveFields = ['password', 'creditCard', 'ssn', 'token'];
  }
  
  checkConsent() {
    // 检查用户是否同意追踪
    const consent = localStorage.getItem('tracking_consent');
    return consent === 'true';
  }
  
  grantConsent() {
    localStorage.setItem('tracking_consent', 'true');
    this.consentGiven = true;
  }
  
  revokeConsent() {
    localStorage.setItem('tracking_consent', 'false');
    this.consentGiven = false;
    this.clearTrackingData();
  }
  
  clearTrackingData() {
    // 清除已收集的追踪数据
    localStorage.removeItem('tracking_id');
    localStorage.removeItem('user_id');
    sessionStorage.removeItem('session_id');
  }
  
  sanitizeEventData(eventData) {
    if (!this.consentGiven) {
      return null;
    }
    
    const sanitized = JSON.parse(JSON.stringify(eventData));
    
    // 移除敏感字段
    Object.keys(sanitized).forEach(key => {
      if (this.isSensitiveField(key)) {
        delete sanitized[key];
      } else if (typeof sanitized[key] === 'string' && this.containsSensitiveInfo(sanitized[key])) {
        sanitized[key] = '[REDACTED]';
      } else if (typeof sanitized[key] === 'object' && sanitized[key] !== null) {
        sanitized[key] = this.sanitizeEventData(sanitized[key]);
      }
    });
    
    return sanitized;
  }
  
  isSensitiveField(fieldName) {
    return this.sensitiveFields.some(field => 
      fieldName.toLowerCase().includes(field.toLowerCase())
    );
  }
  
  containsSensitiveInfo(value) {
    const sensitivePatterns = [
      /\b\d{16}\b/, // 信用卡号
      /\b\d{3}-\d{2}-\d{4}\b/, // SSN
      /\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}\b/ // 邮箱
    ];
    
    return sensitivePatterns.some(pattern => pattern.test(value));
  }
}
```

### 2. 性能优化

```javascript
// 高性能埋点实现
class HighPerformanceTracking {
  constructor() {
    this.eventBuffer = [];
    this.maxBufferSize = 100;
    this.flushThreshold = 50;
    this.flushInterval = 5000;
    this.isProcessing = false;
    
    this.init();
  }
  
  init() {
    // 使用 requestIdleCallback 在空闲时处理
    setInterval(() => {
      if ('requestIdleCallback' in window) {
        requestIdleCallback(() => this.flushBuffer(), { timeout: 2000 });
      } else {
        setTimeout(() => this.flushBuffer(), 0);
      }
    }, this.flushInterval);
  }
  
  trackEvent(event) {
    // 使用 buffer 限制内存使用
    if (this.eventBuffer.length >= this.maxBufferSize) {
      this.eventBuffer.shift(); // 移除最旧的事件
    }
    
    this.eventBuffer.push(event);
    
    // 达到阈值立即处理
    if (this.eventBuffer.length >= this.flushThreshold) {
      this.flushBuffer();
    }
  }
  
  async flushBuffer() {
    if (this.isProcessing || this.eventBuffer.length === 0) {
      return;
    }
    
    this.isProcessing = true;
    const events = [...this.eventBuffer];
    this.eventBuffer = [];
    
    try {
      await this.sendEvents(events);
    } catch (error) {
      console.error('发送事件失败:', error);
      // 失败的事件重新加入 buffer
      this.eventBuffer.unshift(...events.slice(0, this.flushThreshold));
    } finally {
      this.isProcessing = false;
    }
  }
  
  async sendEvents(events) {
    // 使用 sendBeacon 进行异步发送
    if ('sendBeacon' in navigator) {
      const data = JSON.stringify({ events });
      const blob = new Blob([data], { type: 'application/json' });
      navigator.sendBeacon('/api/tracking', blob);
    } else {
      // 回退到 fetch
      await fetch('/api/tracking', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ events }),
        keepalive: true
      });
    }
  }
}
```

```javascript
class TrackingDebugger {
  constructor(trackingSystem) {
```

## 总结

前端埋点是数据驱动决策的重要工具，关键点包括：

1. **全面覆盖**: 覆盖页面浏览、用户行为、业务事件等多个维度
2. **准确可靠**: 确保数据的准确性和上报的可靠性
3. **性能优化**: 减少对应用性能的影响
4. **隐私保护**: 遵循隐私法规，保护用户数据
5. **灵活扩展**: 支持自定义事件和配置
6. **便于调试**: 提供调试工具，便于开发和测试

实施建议：

- 根据业务需求设计埋点方案
- 使用声明式埋点，减少代码侵入
- 建立数据质量监控机制
- 定期分析埋点数据，指导产品优化
- 遵循隐私法规，获得用户同意
- 建立埋点文档和规范

通过合理的埋点方案，我们可以深入了解用户行为，优化产品体验，支持数据驱动的决策。
