## 概述

前端文件预览是现代Web应用中的重要功能，允许用户在不下载文件的情况下直接在浏览器中查看文件内容。这大大提升了用户体验，特别是在办公自动化、文档管理系统、在线教育等场景中。本文将详细介绍各种文件类型的预览方案、技术实现和最佳实践。

## 核心概念

### 文件预览分类

**1. 原生支持的文件类型**
- 图片：JPG, PNG, GIF, SVG, WebP
- 音频：MP3, WAV, OGG
- 视频：MP4, WebM, OGG
- PDF：大部分浏览器原生支持
- 文本：TXT, JSON, XML, CSV

**2. 需要转换或特殊处理的文件类型**
- Word 文档：.doc, .docx
- Excel 表格：.xls, .xlsx
- PowerPoint 演示：.ppt, .pptx
- 其他 Office 文件

**3. 预览方案对比**

| 方案 | 优点 | 缺点 | 适用场景 |
|------|------|------|----------|
| 原生浏览器支持 | 性能好，无需额外依赖 | 文件类型限制 | 图片、音视频、PDF |
| 第三方服务 | 功能强大，兼容性好 | 依赖外部服务，成本高 | 复杂Office文档 |
| Canvas 渲染 | 纯前端实现，可控性强 | 开发复杂度高 | 图片处理、图表 |
| WebAssembly | 性能接近原生 | 文件体积大 | 复杂文件格式解析 |

```javascript
// 基于扩展名检测
function getFileType(filename) {
    const ext = filename.split('.').pop().toLowerCase();
    
    const typeMap = {
        // 图片
        'jpg': 'image', 'jpeg': 'image', 'png': 'image', 'gif': 'image',
        'svg': 'image', 'webp': 'image', 'bmp': 'image',
        
        // 文档
        'pdf': 'pdf', 'doc': 'office', 'docx': 'office',
        'xls': 'office', 'xlsx': 'office', 'ppt': 'office', 'pptx': 'office',
        
        // 音视频
        'mp3': 'audio', 'wav': 'audio', 'ogg': 'audio',
        'mp4': 'video', 'webm': 'video', 'mov': 'video',
        
        // 文本
        'txt': 'text', 'json': 'text', 'xml': 'text', 'csv': 'text',
        'md': 'text', 'html': 'text'
    };
    
    return typeMap[ext] || 'unknown';
}

// 基于MIME类型检测
function getFileTypeByMime(mimeType) {
    if (mimeType.startsWith('image/')) return 'image';
    if (mimeType.startsWith('audio/')) return 'audio';
    if (mimeType.startsWith('video/')) return 'video';
    if (mimeType === 'application/pdf') return 'pdf';
    if (mimeType.includes('word') || mimeType.includes('excel') || mimeType.includes('powerpoint')) return 'office';
    if (mimeType.startsWith('text/')) return 'text';
    return 'unknown';
}

// 综合检测
function detectFileType(file) {
    // 优先使用MIME类型
    if (file.type) {
        const mimeType = getFileTypeByMime(file.type);
        if (mimeType !== 'unknown') return mimeType;
    }
    
    // 回退到扩展名检测
    return getFileType(file.name);
}
```

### 1. 图片预览

```javascript
// 基本图片预览
function previewImage(file) {
    return new Promise((resolve, reject) => {
        if (!file.type.startsWith('image/')) {
            reject(new Error('不是图片文件'));
            return;
        }
        
        const reader = new FileReader();
        reader.onload = (e) => {
            const img = new Image();
            img.src = e.target.result;
            img.onload = () => resolve(img);
            img.onerror = () => reject(new Error('图片加载失败'));
        };
        reader.onerror = () => reject(new Error('文件读取失败'));
        reader.readAsDataURL(file);
    });
}

// 图片预览组件
class ImagePreviewer {
    constructor(containerId) {
        this.container = document.getElementById(containerId);
        this.createUI();
    }
    
    createUI() {
        this.container.innerHTML = `
            <div class="image-previewer">
                <input type="file" accept="image/*" multiple class="file-input">
                <div class="preview-container"></div>
            </div>
        `;
        
        this.fileInput = this.container.querySelector('.file-input');
        this.previewContainer = this.container.querySelector('.preview-container');
        
        this.fileInput.addEventListener('change', this.handleFileSelect.bind(this));
    }
    
    async handleFileSelect(e) {
        const files = Array.from(e.target.files);
        this.previewContainer.innerHTML = '';
        
        for (const file of files) {
            try {
                await this.previewSingleImage(file);
            } catch (error) {
                console.error(`预览图片失败: ${file.name}`, error);
                this.showError(file.name, error.message);
            }
        }
    }
    
    async previewSingleImage(file) {
        const previewData = await previewImage(file);
        
        const previewItem = document.createElement('div');
        previewItem.className = 'preview-item';
        
        const img = document.createElement('img');
        img.src = previewData.src;
        img.alt = file.name;
        
        const info = document.createElement('div');
        info.className = 'preview-info';
        info.innerHTML = `
            <div class="filename">${file.name}</div>
            <div class="file-info">
                <span>${previewData.naturalWidth} × ${previewData.naturalHeight}</span>
                <span>${this.formatFileSize(file.size)}</span>
            </div>
        `;
        
        previewItem.appendChild(img);
        previewItem.appendChild(info);
        this.previewContainer.appendChild(previewItem);
    }
    
    showError(filename, message) {
        const errorItem = document.createElement('div');
        errorItem.className = 'preview-item error';
        errorItem.innerHTML = `
            <div class="error-icon">❌</div>
            <div class="error-info">
                <div class="filename">${filename}</div>
                <div class="error-message">${message}</div>
            </div>
        `;
        this.previewContainer.appendChild(errorItem);
    }
    
    formatFileSize(bytes) {
        if (bytes === 0) return '0 B';
        const k = 1024;
        const sizes = ['B', 'KB', 'MB', 'GB'];
        const i = Math.floor(Math.log(bytes) / Math.log(k));
        return Math.round((bytes / Math.pow(k, i)) * 100) / 100 + ' ' + sizes[i];
    }
}
```

```javascript
// 基本PDF预览
function previewPDF(file) {
    return new Promise((resolve, reject) => {
        if (file.type !== 'application/pdf') {
            reject(new Error('不是PDF文件'));
            return;
        }
        
        const reader = new FileReader();
        reader.onload = (e) => {
            resolve(e.target.result);
        };
        reader.onerror = () => reject(new Error('文件读取失败'));
        reader.readAsDataURL(file);
    });
}

// PDF预览组件
class PDFPreviewer {
    constructor(containerId) {
        this.container = document.getElementById(containerId);
        this.pdfLib = null;
        this.init();
    }
    
    async init() {
        // 动态加载PDF.js库
        if (typeof pdfjsLib === 'undefined') {
            await this.loadPDFLib();
        }
        
        this.createUI();
    }
    
    async loadPDFLib() {
        return new Promise((resolve, reject) => {
            const script = document.createElement('script');
            script.src = 'https://cdnjs.cloudflare.com/ajax/libs/pdf.js/3.11.174/pdf.min.js';
            script.onload = () => {
                pdfjsLib.GlobalWorkerOptions.workerSrc = 
                    'https://cdnjs.cloudflare.com/ajax/libs/pdf.js/3.11.174/pdf.worker.min.js';
                this.pdfLib = pdfjsLib;
                resolve();
            };
            script.onerror = () => reject(new Error('PDF.js加载失败'));
            document.head.appendChild(script);
        });
    }
    
    createUI() {
        this.container.innerHTML = `
            <div class="pdf-previewer">
                <input type="file" accept=".pdf" class="file-input">
                <div class="pdf-controls">
                    <button class="prev-page" disabled>上一页</button>
                    <span class="page-info">0 / 0</span>
                    <button class="next-page" disabled>下一页</button>
                    <button class="zoom-out" disabled>缩小</button>
                    <span class="zoom-info">100%</span>
                    <button class="zoom-in" disabled>放大</button>
                </div>
                <div class="pdf-container"></div>
            </div>
        `;
        
        this.fileInput = this.container.querySelector('.file-input');
        this.pdfContainer = this.container.querySelector('.pdf-container');
        this.prevPageBtn = this.container.querySelector('.prev-page');
        this.nextPageBtn = this.container.querySelector('.next-page');
        this.zoomInBtn = this.container.querySelector('.zoom-in');
        this.zoomOutBtn = this.container.querySelector('.zoom-out');
        this.pageInfo = this.container.querySelector('.page-info');
        this.zoomInfo = this.container.querySelector('.zoom-info');
        
        this.currentPage = 1;
        this.totalPages = 0;
        this.scale = 1.0;
        this.pdfDocument = null;
        
        this.bindEvents();
    }
    
    bindEvents() {
        this.fileInput.addEventListener('change', this.handleFileSelect.bind(this));
        this.prevPageBtn.addEventListener('click', () => this.changePage(-1));
        this.nextPageBtn.addEventListener('click', () => this.changePage(1));
        this.zoomInBtn.addEventListener('click', () => this.changeZoom(0.1));
        this.zoomOutBtn.addEventListener('click', () => this.changeZoom(-0.1));
    }
    
    async handleFileSelect(e) {
        const file = e.target.files[0];
        if (!file) return;
        
        try {
            const arrayBuffer = await file.arrayBuffer();
            this.pdfDocument = await this.pdfLib.getDocument(arrayBuffer).promise;
            
            this.totalPages = this.pdfDocument.numPages;
            this.currentPage = 1;
            
            this.updateControls();
            await this.renderPage(this.currentPage);
            
        } catch (error) {
            console.error('PDF加载失败:', error);
            alert('PDF加载失败，请检查文件是否损坏');
        }
    }
    
    async renderPage(pageNum) {
        const page = await this.pdfDocument.getPage(pageNum);
        const viewport = page.getViewport({ scale: this.scale });
        
        const canvas = document.createElement('canvas');
        const context = canvas.getContext('2d');
        canvas.height = viewport.height;
        canvas.width = viewport.width;
        
        const renderContext = {
            canvasContext: context,
            viewport: viewport
        };
        
        await page.render(renderContext).promise;
        
        this.pdfContainer.innerHTML = '';
        this.pdfContainer.appendChild(canvas);
        this.pageInfo.textContent = `${this.currentPage} / ${this.totalPages}`;
    }
    
    async changePage(delta) {
        const newPage = this.currentPage + delta;
        if (newPage >= 1 && newPage <= this.totalPages) {
            this.currentPage = newPage;
            await this.renderPage(this.currentPage);
            this.updateControls();
        }
    }
    
    async changeZoom(delta) {
        const newScale = this.scale + delta;
        if (newScale >= 0.5 && newScale <= 3.0) {
            this.scale = newScale;
            this.zoomInfo.textContent = `${Math.round(this.scale * 100)}%`;
            await this.renderPage(this.currentPage);
            this.updateControls();
        }
    }
    
    updateControls() {
        this.prevPageBtn.disabled = this.currentPage <= 1;
        this.nextPageBtn.disabled = this.currentPage >= this.totalPages;
        this.zoomInBtn.disabled = this.scale >= 3.0;
        this.zoomOutBtn.disabled = this.scale <= 0.5;
    }
}
```

### 3. 文本文件预览

```javascript
// 基本文本文件预览
function previewTextFile(file) {
    return new Promise((resolve, reject) => {
        if (!file.type.startsWith('text/') && !file.name.match(/\.(txt|json|xml|csv|md|html)$/i)) {
            reject(new Error('不是文本文件'));
            return;
        }
        
        const reader = new FileReader();
        reader.onload = (e) => {
            resolve({
                content: e.target.result,
                encoding: 'utf-8'
            });
        };
        reader.onerror = () => reject(new Error('文件读取失败'));
        reader.readAsText(file, 'utf-8');
    });
}

// 文本预览组件
class TextPreviewer {
    constructor(containerId) {
        this.container = document.getElementById(containerId);
        this.createUI();
    }
    
    createUI() {
        this.container.innerHTML = `
            <div class="text-previewer">
                <input type="file" accept=".txt,.json,.xml,.csv,.md,.html" class="file-input">
                <div class="text-controls">
                    <select class="line-endings">
                        <option value="lf">LF (Unix)</option>
                        <option value="crlf">CRLF (Windows)</option>
                    </select>
                    <button class="copy-btn">复制</button>
                    <button class="download-btn">下载</button>
                </div>
                <div class="text-container">
                    <pre class="text-content"></pre>
                </div>
            </div>
        `;
        
        this.fileInput = this.container.querySelector('.file-input');
        this.textContent = this.container.querySelector('.text-content');
        this.copyBtn = this.container.querySelector('.copy-btn');
        this.downloadBtn = this.container.querySelector('.download-btn');
        
        this.currentFile = null;
        this.currentContent = '';
        
        this.bindEvents();
    }
    
    bindEvents() {
        this.fileInput.addEventListener('change', this.handleFileSelect.bind(this));
        this.copyBtn.addEventListener('click', this.copyContent.bind(this));
        this.downloadBtn.addEventListener('click', this.downloadContent.bind(this));
    }
    
    async handleFileSelect(e) {
        const file = e.target.files[0];
        if (!file) return;
        
        try {
            const { content } = await previewTextFile(file);
            this.currentFile = file;
            this.currentContent = content;
            
            // 根据文件类型进行特殊处理
            const displayContent = this.formatContent(content, file.name);
            this.textContent.textContent = displayContent;
            
            // 高亮代码（如果需要）
            this.highlightSyntax(displayContent, file.name);
            
        } catch (error) {
            console.error('文本文件预览失败:', error);
            this.textContent.textContent = `预览失败: ${error.message}`;
        }
    }
    
    formatContent(content, filename) {
        // 根据文件扩展名进行格式化
        const ext = filename.split('.').pop().toLowerCase();
        
        switch (ext) {
            case 'json':
                try {
                    return JSON.stringify(JSON.parse(content), null, 2);
                } catch {
                    return content;
                }
            case 'xml':
                return this.formatXML(content);
            case 'csv':
                return this.formatCSV(content);
            default:
                return content;
        }
    }
    
    formatXML(xmlString) {
        // 简单的XML格式化
        let formatted = '';
        let indent = 0;
        const tab = '  ';
        
        xmlString.split(/>\s*</).forEach(function(node) {
            if (node.match(/^\/\w/)) indent -= 1;
            formatted += tab.repeat(indent) + '<' + node + '>\r\n';
            if (node.match(/^<?\w[^>]*[^\/]$/)) indent += 1;
        });
        
        return formatted.substring(1, formatted.length - 3);
    }
    
    formatCSV(csvString) {
        // 简单的CSV格式化
        const lines = csvString.split('\n');
        const headers = lines[0].split(',');
        const rows = lines.slice(1).map(line => line.split(','));
        
        let table = '| ' + headers.join(' | ') + ' |\n';
        table += '| ' + headers.map(() => '---').join(' | ') + ' |\n';
        
        rows.forEach(row => {
            table += '| ' + row.join(' | ') + ' |\n';
        });
        
        return table;
    }
    
    async highlightSyntax(content, filename) {
        // 这里可以集成代码高亮库如 Prism.js 或 Highlight.js
        // 示例使用 Prism.js
        if (typeof Prism !== 'undefined') {
            const ext = filename.split('.').pop().toLowerCase();
            const langMap = {
                'js': 'javascript',
                'json': 'javascript',
                'html': 'html',
                'css': 'css',
                'md': 'markdown',
                'py': 'python',
                'java': 'java'
            };
            
            const lang = langMap[ext] || 'plaintext';
            this.textContent.innerHTML = Prism.highlight(content, Prism.languages[lang], lang);
        }
    }
    
    async copyContent() {
        if (!this.currentContent) return;
        
        try {
            await navigator.clipboard.writeText(this.currentContent);
            this.showCopySuccess();
        } catch (error) {
            console.error('复制失败:', error);
            this.showCopyError();
        }
    }
    
    showCopySuccess() {
        const originalText = this.copyBtn.textContent;
        this.copyBtn.textContent = '已复制';
        this.copyBtn.classList.add('success');
        
        setTimeout(() => {
            this.copyBtn.textContent = originalText;
            this.copyBtn.classList.remove('success');
        }, 2000);
    }
    
    showCopyError() {
        const originalText = this.copyBtn.textContent;
        this.copyBtn.textContent = '复制失败';
        this.copyBtn.classList.add('error');
        
        setTimeout(() => {
            this.copyBtn.textContent = originalText;
            this.copyBtn.classList.remove('error');
        }, 2000);
    }
    
    downloadContent() {
        if (!this.currentContent || !this.currentFile) return;
        
        const blob = new Blob([this.currentContent], { type: 'text/plain' });
        const url = URL.createObjectURL(blob);
        const link = document.createElement('a');
        link.href = url;
        link.download = this.currentFile.name;
        document.body.appendChild(link);
        link.click();
        document.body.removeChild(link);
        URL.revokeObjectURL(url);
    }
}
```

```javascript
// Office 文档预览管理器
class OfficePreviewer {
    constructor(containerId) {
        this.container = document.getElementById(containerId);
        this.previewOptions = {

            custom: ''
        };

        this.createUI();
    }
    
    createUI() {
        this.container.innerHTML = `
            <div class="office-previewer">
                <input type="file" accept=".doc,.docx,.xls,.xlsx,.ppt,.pptx" class="file-input">
                <div class="preview-options">
                    <label>预览服务:</label>
                    <select class="service-select">
                        <option value="microsoft">Microsoft Office Online</option>
                        <option value="google">Google Docs Viewer</option>
                    </select>
                    <button class="preview-btn">预览</button>
                </div>
                <div class="preview-container">
                    <div class="placeholder">
                        <div class="placeholder-icon">📄</div>
                        <p>选择文件开始预览</p>
                    </div>
                </div>
            </div>
        `;
        
        this.fileInput = this.container.querySelector('.file-input');
        this.serviceSelect = this.container.querySelector('.service-select');
        this.previewBtn = this.container.querySelector('.preview-btn');
        this.previewContainer = this.container.querySelector('.preview-container');
        this.placeholder = this.container.querySelector('.placeholder');
        
        this.bindEvents();
    }
    
    bindEvents() {
        this.serviceSelect.addEventListener('change', (e) => {
            this.currentService = e.target.value;
        });
        
        this.previewBtn.addEventListener('click', this.previewFile.bind(this));
    }
    
    async previewFile() {
        const file = this.fileInput.files[0];
        if (!file) {
            alert('请选择要预览的文件');
            return;
        }
        
        if (!this.isOfficeFile(file)) {
            alert('请选择Office文档（.doc, .docx, .xls, .xlsx, .ppt, .pptx）');
            return;
        }
        
        try {
            // 上传文件到服务器获取可访问的URL
            const fileUrl = await this.uploadFile(file);
            
            // 生成预览URL
            const previewUrl = this.generatePreviewUrl(fileUrl);
            
            // 显示预览
            this.showPreview(previewUrl);
            
        } catch (error) {
            console.error('预览失败:', error);
            alert('预览失败: ' + error.message);
        }
    }
    
    isOfficeFile(file) {
        const ext = file.name.split('.').pop().toLowerCase();
        return ['doc', 'docx', 'xls', 'xlsx', 'ppt', 'pptx'].includes(ext);
    }
    
    async uploadFile(file) {
        // 实际项目中应该上传到服务器
        // 这里使用临时URL作为示例
        return new Promise((resolve) => {
            setTimeout(() => {
                // 模拟服务器返回的文件URL
                resolve('https://example.com/files/' + encodeURIComponent(file.name));
            }, 1000);
        });
    }
    
    generatePreviewUrl(fileUrl) {
        const serviceUrl = this.previewOptions[this.currentService];
        return serviceUrl + encodeURIComponent(fileUrl);
    }
    
    showPreview(previewUrl) {
        this.placeholder.style.display = 'none';
        
        const iframe = document.createElement('iframe');
        iframe.src = previewUrl;
        iframe.className = 'preview-frame';
        iframe.style.width = '100%';
        iframe.style.height = '600px';
        iframe.style.border = 'none';
        
        // 清除之前的预览内容
        const existingIframe = this.previewContainer.querySelector('iframe');
        if (existingIframe) {
            existingIframe.remove();
        }
        
        this.previewContainer.appendChild(iframe);
    }
}
```

```javascript
        this.fileInput.addEventListener('change', this.handleFileSelect.bind(this));

            btn.addEventListener('click', (e) => {
                this.typeButtons.forEach(b => b.classList.remove('active'));
                e.target.classList.add('active');
                this.currentType = e.target.dataset.type;
                
                if (this.currentFile) {
                    this.previewFile(this.currentFile);
                }
            });
        });
    }
    
    async handleFileSelect(e) {
        const file = e.target.files[0];
        if (!file) return;
        
        this.currentFile = file;
        await this.previewFile(file);
    }
    
    async previewFile(file) {
        // 隐藏所有预览器
        Object.values(this.previewers).forEach(previewer => {
            previewer.container.style.display = 'none';
        });
        
        // 确定预览类型
        let previewType = this.currentType;
        if (previewType === 'auto') {
            previewType = this.detectFileType(file);
        }
        
        // 显示对应的预览器
        const previewer = this.previewers[previewType];
        if (previewer) {
            previewer.container.style.display = 'block';
            
            // 将文件传递给对应的预览器
            if (previewer.fileInput) {
                const dataTransfer = new DataTransfer();
                dataTransfer.items.add(file);
                previewer.fileInput.files = dataTransfer.files;
                
                // 触发change事件
                const event = new Event('change');
                previewer.fileInput.dispatchEvent(event);
            }
        } else {
            this.showUnsupportedMessage(file);
        }
    }
    
    detectFileType(file) {
        const fileType = detectFileType(file);
        const supportedTypes = ['image', 'pdf', 'text', 'office'];
        
        return supportedTypes.includes(fileType) ? fileType : 'unsupported';
    }
    
    showUnsupportedMessage(file) {
        this.previewContent.innerHTML = `
            <div class="unsupported-message">
                <div class="icon">❌</div>
                <h3>不支持的文件类型</h3>
                <p>文件名: ${file.name}</p>
                <p>文件类型: ${file.type || '未知'}</p>
                <p>大小: ${this.formatFileSize(file.size)}</p>
            </div>
        `;
    }
    
    formatFileSize(bytes) {
        if (bytes === 0) return '0 B';
        const k = 1024;
        const sizes = ['B', 'KB', 'MB', 'GB'];
        const i = Math.floor(Math.log(bytes) / Math.log(k));
        return Math.round((bytes / Math.pow(k, i)) * 100) / 100 + ' ' + sizes[i];
    }
}
```

```javascript
class SecureFilePreviewer {
    constructor() {
        this.maxFileSize = 10 * 1024 * 1024; // 10MB
        this.allowedTypes = [
            'image/jpeg', 'image/png', 'image/gif', 'image/webp',
            'application/pdf',
            'text/plain', 'text/json', 'text/xml', 'text/csv'
        ];
    }
    
    validateFile(file) {
        // 文件大小检查
        if (file.size > this.maxFileSize) {
            throw new Error(`文件大小超过限制 (${this.formatFileSize(this.maxFileSize)})`);
        }
        
        // 文件类型检查
        if (!this.isAllowedType(file)) {
            throw new Error('不支持的文件类型');
        }
        
        // 文件名安全检查
        if (!this.isSafeFilename(file.name)) {
            throw new Error('文件名包含非法字符');
        }
        
        return true;
    }
    
    isAllowedType(file) {
        return this.allowedTypes.includes(file.type) || 
               this.hasSafeExtension(file.name);
    }
    
    hasSafeFilename(filename) {
        // 检查文件名是否包含危险字符
        const dangerousPatterns = [
            /\.\./,  // 路径遍历
            /[<>:"|?*]/,  // 文件系统非法字符
            /^\./,  // 隐藏文件
            /^(CON|PRN|AUX|NUL|COM[1-9]|LPT[1-9])$/i  // Windows保留文件名
        ];
        
        return !dangerousPatterns.some(pattern => pattern.test(filename));
    }
    
    hasSafeExtension(filename) {
        const safeExtensions = [
            '.jpg', '.jpeg', '.png', '.gif', '.webp',
            '.pdf',
            '.txt', '.json', '.xml', '.csv', '.md'
        ];
        
        const ext = filename.substring(filename.lastIndexOf('.')).toLowerCase();
        return safeExtensions.includes(ext);
    }
    
    sanitizeContent(content, type) {
        switch (type) {
            case 'html':
            case 'xml':
                // 防止XSS攻击
                return this.escapeHtml(content);
            case 'json':
                try {
                    // 验证JSON格式
                    JSON.parse(content);
                    return content;
                } catch {
                    throw new Error('无效的JSON格式');
                }
            default:
                return content;
        }
    }
    
    escapeHtml(text) {
        const div = document.createElement('div');
        div.textContent = text;
        return div.innerHTML;
    }
    
    formatFileSize(bytes) {
        if (bytes === 0) return '0 B';
        const k = 1024;
        const sizes = ['B', 'KB', 'MB', 'GB'];
        const i = Math.floor(Math.log(bytes) / Math.log(k));
        return Math.round((bytes / Math.pow(k, i)) * 100) / 100 + ' ' + sizes[i];
    }
}
```

```javascript
class CompatiblePreviewer {

            clipboard: typeof navigator !== 'undefined' && typeof navigator.clipboard !== 'undefined',
            intersectionObserver: typeof IntersectionObserver !== 'undefined',
            webAssembly: typeof WebAssembly !== 'undefined',
            serviceWorker: 'serviceWorker' in navigator

        const fileType = this.detectFileType(file);
        
        try {
            switch (fileType) {
                case 'image':
                    return await this.previewImageCompat(file);
                case 'pdf':
                    return await this.previewPDFCompat(file);
                case 'text':
                    return await this.previewTextCompat(file);
                default:
                    return this.showUnsupportedMessage(file);
            }
        } catch (error) {
            console.error('预览失败:', error);
            return this.showErrorMessage(error.message);
        }
    }
    
    async previewImageCompat(file) {
        if (this.capabilities.createObjectURL) {
            // 使用现代API
            const url = URL.createObjectURL(file);
            return this.showImage(url);
        } else if (this.capabilities.FileReader) {
            // 使用FileReader回退
            return new Promise((resolve, reject) => {
                const reader = new FileReader();
                reader.onload = (e) => resolve(this.showImage(e.target.result));
                reader.onerror = () => reject(new Error('文件读取失败'));
                reader.readAsDataURL(file);
            });
        } else {
            throw new Error('浏览器不支持图片预览');
        }
    }

            // 检查是否支持原生PDF预览
            if (this.supportsNativePDF()) {
                const url = URL.createObjectURL(file);
                return this.showNativePDF(url);

        return new Promise((resolve, reject) => {

        });
    }

        const typeMap = {
            'jpg': 'image', 'jpeg': 'image', 'png': 'image', 'gif': 'image',
```
