## 概述

前端权限设计是现代 Web 应用开发中的核心环节，涉及用户访问控制、功能权限管理、数据安全防护等多个方面。合理的权限设计不仅能保护系统安全，还能提供良好的用户体验。

权限系统通常包含三个核心要素：

- **身份认证**：确认用户身份
- **权限授权**：赋予用户相应权限
- **访问控制**：限制用户访问范围

---

## 核心概念

### RBAC 模型

基于角色的访问控制是目前最主流的权限模型。

```javascript
// 用户-角色-权限关系示例
const users = {
  user1: { name: '张三', roles: ['admin'] },
  user2: { name: '李四', roles: ['editor'] },
  user3: { name: '王五', roles: ['viewer'] }
};

const roles = {
  admin: ['user:create', 'user:update', 'user:delete', 'user:view', 'content:manage'],
  editor: ['user:view', 'content:create', 'content:update', 'content:view'],
  viewer: ['user:view', 'content:view']
};

// 检查用户权限
function hasPermission(userId, permission) {
  const user = users[userId];
  if (!user) return false;

  return user.roles.some(role => roles[role]?.includes(permission));
}
```

### 权限粒度

权限控制通常分为多个粒度层级：

- **页面级**：用户能否访问某个页面
- **功能级**：用户能否使用某个功能
- **数据级**：用户能否操作特定数据
- **字段级**：用户能否查看或编辑特定字段
- **操作级**：用户能否执行特定操作

```javascript
import { createBrowserRouter, Navigate } from 'react-router-dom';

    return <Navigate to="/403" replace />;

  return children;
```

```javascript
// router/index.js
import { createRouter, createWebHistory } from 'vue-router';
import { useUserStore } from '@/stores/user';

const routes = [
  {
    path: '/login',
    name: 'Login',
    component: () => import('@/views/Login.vue'),
    meta: { requiresAuth: false }
  },
  {
    path: '/dashboard',
    name: 'Dashboard',
    component: () => import('@/views/Dashboard.vue'),
    meta: { requiresAuth: true, permissions: ['dashboard:view'] }
  },
  {
    path: '/admin',
    name: 'Admin',
    component: () => import('@/views/Admin.vue'),
    meta: { requiresAuth: true, permissions: ['admin:access'], role: 'admin' }
  }
];

const router = createRouter({
  history: createWebHistory(),
  routes
});

// 全局路由守卫
router.beforeEach(async (to, from, next) => {
  const userStore = useUserStore();

  // 检查是否需要认证
  if (to.meta.requiresAuth && !userStore.isAuthenticated) {
    next('/login');
    return;
  }

  // 检查权限
  if (to.meta.permissions && !userStore.hasPermission(to.meta.permissions)) {
    next('/403');
    return;
  }

  // 检查角色
  if (to.meta.role && userStore.role !== to.meta.role) {
    next('/403');
    return;
  }

  next();
});

export default router;
```

```javascript
    key: 'content',
    label: '内容管理',
    path: '/content',
    icon: 'content',
    permission: 'content:manage'
  }
];

// 过滤菜单权限
function filterMenusByPermission(menus, permissions) {
  return menus
    .filter(menu => {
      // 检查菜单权限
      if (menu.permission && !permissions.includes(menu.permission)) {
        return false;
      }
      return true;
    })
    .map(menu => {
      // 递归处理子菜单
      if (menu.children) {
        return {
          ...menu,
          children: filterMenusByPermission(menu.children, permissions)
        };
      }
      return menu;
    });
}

// 使用示例
const userPermissions = ['dashboard:view', 'user:view', 'content:manage'];
const availableMenus = filterMenusByPermission(menuItems, userPermissions);
```

```javascript
export const useUserStore = defineStore('user', {

      const response = await api.getUserPermissions();

      this.roles = response.roles;

    logout() {
      this.user = null;
      this.permissions = [];
      this.roles = [];
      this.token = null;
      localStorage.removeItem('token');
```

```javascript
// utils/permission.js - 权限工具函数
class PermissionManager {
  constructor(userStore) {
    this.userStore = userStore;
  }

  // 检查单个权限
  checkPermission(permission) {
    return this.userStore.permissions.includes(permission);
  }

  // 检查多个权限（AND）
  checkPermissions(permissions) {
    return permissions.every(p => this.userStore.permissions.includes(p));
  }

  // 检查任意权限（OR）
  checkAnyPermission(permissions) {
    return permissions.some(p => this.userStore.permissions.includes(p));
  }

  // 检查角色
  checkRole(role) {
    return this.userStore.roles.includes(role);
  }

  // 检查任意角色
  checkAnyRole(roles) {
    return roles.some(r => this.userStore.roles.includes(r));
  }

  // 获取可访问的路由
  getAccessibleRoutes(routes) {
    return routes.filter(route => {
      if (route.meta?.permissions) {
        return this.checkAnyPermission(route.meta.permissions);
      }
      if (route.meta?.roles) {
        return this.checkAnyRole(route.meta.roles);
      }
      return true;
    });
  }
}
```

### 2. 数据权限控制

```javascript
// utils/dataPermission.js - 数据级权限控制
class DataPermissionManager {
  constructor(userStore) {
    this.userStore = userStore;
  }

  // 检查数据所属权
  checkOwnership(data, userIdField = 'userId') {
    return data[userIdField] === this.userStore.user?.id;
  }

  // 过滤有权限访问的数据
  filterAccessibleData(dataList, permissionField = 'permission') {
    return dataList.filter(data => {
      const requiredPermission = data[permissionField];
      if (!requiredPermission) return true;
      return this.userStore.permissions.includes(requiredPermission);
    });
  }

  // 获取数据访问范围
  getDataScope() {
    const { roles, permissions } = this.userStore;

    if (roles.includes('admin')) {
      return 'all'; // 全部数据
    }

    if (permissions.includes('data:department')) {
      return 'department'; // 部门数据
    }

    if (permissions.includes('data:self')) {
      return 'self'; // 自己的数据
    }

    return 'none'; // 无权限
  }

  // 构建数据查询条件
  buildDataQueryFilter(userId, departmentId) {
    const scope = this.getDataScope();

    switch (scope) {
      case 'all':
        return {};
      case 'department':
        return { departmentId };
      case 'self':
        return { userId };
      default:
        return { _id: null }; // 返回空结果
    }
  }
}

// 使用示例
function UserService(dataPermissionManager) {
  this.dataPermissionManager = dataPermissionManager;

  this.getUsers = async (userId, departmentId) => {
    const filter = this.dataPermissionManager.buildDataQueryFilter(userId, departmentId);
    return await api.getUsers(filter);
  };
}
```

### 3. API 请求权限控制

```javascript
// utils/request.js - Axios 请求拦截器
import axios from 'axios';
import { useUserStore } from '@/stores/user';

const request = axios.create({
  baseURL: '/api',
  timeout: 10000
});

// 请求拦截器
request.interceptors.request.use(
  (config) => {
    const userStore = useUserStore();
    const token = userStore.token;

    if (token) {
      config.headers.Authorization = `Bearer ${token}`;
    }

    return config;
  },
  (error) => {
    return Promise.reject(error);
  }
);

// 响应拦截器
request.interceptors.response.use(
  (response) => {
    return response.data;
  },
  async (error) => {
    const userStore = useUserStore();

    if (error.response?.status === 401) {
      // Token 过期，刷新 Token
      try {
        const newToken = await api.refreshToken();
        userStore.token = newToken;
        error.config.headers.Authorization = `Bearer ${newToken}`;
        return request(error.config);
      } catch (refreshError) {
        userStore.logout();
        window.location.href = '/login';
      }
    }

    if (error.response?.status === 403) {
      // 权限不足
      console.error('Permission denied');
      // 可以显示权限不足的提示
    }

    return Promise.reject(error);
  }
);

export default request;
```

### 4. 权限缓存策略

```javascript
// stores/permissionStore.js - 权限缓存管理
import { defineStore } from 'pinia';

export const usePermissionStore = defineStore('permission', {
  state: () => ({
    cachedMenus: [],
    cachedRoutes: [],
    lastUpdate: null,
    cacheDuration: 5 * 60 * 1000 // 5分钟缓存
  }),

  getters: {
    isCacheExpired: (state) => {
      if (!state.lastUpdate) return true;
      return Date.now() - state.lastUpdate > state.cacheDuration;
    }
  },

  actions: {
    async loadPermissions(forceRefresh = false) {
      if (!forceRefresh && !this.isCacheExpired) {
        return { menus: this.cachedMenus, routes: this.cachedRoutes };
      }

      const response = await api.getPermissions();
      this.cachedMenus = response.menus;
      this.cachedRoutes = response.routes;
      this.lastUpdate = Date.now();

      return response;
    },

    clearCache() {
      this.cachedMenus = [];
      this.cachedRoutes = [];
      this.lastUpdate = null;
    },

    invalidateCache() {
      this.lastUpdate = 0;
    }
  }
});
```

## 注意事项

### 1. 前后端权限一致性

```javascript
// ❌ 错误示例：只在前端控制权限
function deleteUser(userId) {
  if (!hasPermission('user:delete')) {
    alert('没有权限');
    return;
  }
  // 直接调用删除接口
  return api.deleteUser(userId);
}

// ✅ 正确示例：前后端双重验证
async function deleteUser(userId) {
  // 前端验证
  if (!hasPermission('user:delete')) {
    alert('没有权限');
    return;
  }

  try {
    // 后端也会验证权限
    const result = await api.deleteUser(userId);
    return result;
  } catch (error) {
    if (error.response?.status === 403) {
      alert('权限不足');
    }
    throw error;
  }
}
```

```javascript
// 使用防抖优化权限检查
import { debounce } from 'lodash';

class PermissionChecker {
  constructor() {
    this.cache = new Map();
    this.debouncedCheck = debounce(this.checkPermission.bind(this), 100);
  }

  checkPermission(permission) {
    if (this.cache.has(permission)) {
      return this.cache.get(permission);
    }

    const hasPermission = this._checkPermission(permission);
    this.cache.set(permission, hasPermission);
    return hasPermission;
  }

  _checkPermission(permission) {
    // 实际的权限检查逻辑
    return userStore.permissions.includes(permission);
  }

  clearCache() {
    this.cache.clear();
  }
}
```

## 总结

前端权限设计是一个复杂的系统工程，需要考虑多个方面：

### 核心要点

1. **分层设计**：页面级、功能级、数据级权限控制
2. **前后端协同**：前端验证提供良好体验，后端验证确保安全
3. **性能优化**：合理的缓存策略，避免重复计算
4. **用户体验**：权限变更时提供友好提示
5. **安全防护**：防止 XSS、CSRF 等安全漏洞

### 最佳实践

- 使用 RBAC 模型简化权限管理
- 实现权限缓存机制提高性能
- 前后端双重验证确保安全
- 提供清晰的权限错误提示
- 定期审计和优化权限设计

### 常见场景

- 动态菜单渲染
- 路由守卫控制
- 按钮权限控制
- 数据过滤和查询
- API 请求拦截

通过合理的权限设计，可以构建安全、可靠、易用的前端应用系统。记住权限是一个持续优化和迭代的过程，需要根据实际业务需求不断调整和完善。
