## 概述

前端主题切换是现代 Web 应用的必备功能，尤其是暗黑模式（Dark Mode）已成为提升用户体验的重要特性。本文将详细介绍前端主题切换的实现方案、最佳实践和实际应用。

## 核心概念

### 主题切换的类型

1. **明暗主题切换**：最常见的形式，在浅色和深色主题间切换
2. **多主题支持**：支持多种预设主题，如蓝色、绿色、紫色等
3. **自定义主题**：允许用户自定义颜色和样式
4. **自动切换**：根据系统设置或时间段自动切换

### 实现原理

主题切换的核心是通过 CSS 变量（CSS Custom Properties）来实现动态样式更新，配合 JavaScript 控制主题状态。

### JavaScript 切换函数

```javascript
// 主题管理类
class ThemeManager {
  constructor() {
    this.currentTheme = this.getTheme();
    this.init();
  }

  init() {
    // 初始化主题
    this.applyTheme(this.currentTheme);

    // 监听系统主题变化
    if (window.matchMedia) {
      const darkModeQuery = window.matchMedia('(prefers-color-scheme: dark)');
      darkModeQuery.addEventListener('change', (e) => {
        if (!localStorage.getItem('theme')) {
          this.currentTheme = e.matches ? 'dark' : 'light';
          this.applyTheme(this.currentTheme);
        }
      });
    }
  }

  getTheme() {
    // 优先使用本地存储的主题
    const storedTheme = localStorage.getItem('theme');
    if (storedTheme) {
      return storedTheme;
    }

    // 其次使用系统偏好
    if (window.matchMedia && window.matchMedia('(prefers-color-scheme: dark)').matches) {
      return 'dark';
    }

    // 默认浅色主题
    return 'light';
  }

  setTheme(theme) {
    this.currentTheme = theme;
    localStorage.setItem('theme', theme);
    this.applyTheme(theme);
    this.notifyThemeChange(theme);
  }

  applyTheme(theme) {
    document.documentElement.setAttribute('data-theme', theme);
    document.documentElement.setAttribute('aria-label', `${theme} theme`);
  }

  toggleTheme() {
    const newTheme = this.currentTheme === 'light' ? 'dark' : 'light';
    this.setTheme(newTheme);
  }

  notifyThemeChange(theme) {
    // 触发自定义事件
    const event = new CustomEvent('themeChange', {
      detail: { theme }
    });
    window.dispatchEvent(event);

    // 通知其他标签页
    if ('BroadcastChannel' in window) {
      const channel = new BroadcastChannel('theme');
      channel.postMessage({ theme });
    }
  }
}

// 创建主题管理器实例
const themeManager = new ThemeManager();
```

## 实际应用

### React Hook 实现

```javascript
import { useState, useEffect } from 'react';
import { createGlobalStyle } from 'styled-components';

const GlobalStyle = createGlobalStyle`
  :root {
    --bg-primary: #ffffff;
    --text-primary: #333333;
    /* 更多变量... */
  }

  [data-theme="dark"] {
    --bg-primary: #1a1a1a;
    --text-primary: #f0f0f0;
    /* 更多变量... */
  }

  body {
    background-color: var(--bg-primary);
    color: var(--text-primary);
    transition: background-color 0.3s, color 0.3s;
  }
`;

function useTheme() {
  const [theme, setTheme] = useState('light');

  useEffect(() => {
    // 初始化主题
    const savedTheme = localStorage.getItem('theme') || 'light';
    setTheme(savedTheme);
    document.documentElement.setAttribute('data-theme', savedTheme);

    // 监听其他标签页的主题变化
    const channel = new BroadcastChannel('theme');
    channel.onmessage = (event) => {
      setTheme(event.data.theme);
      document.documentElement.setAttribute('data-theme', event.data.theme);
    };

    return () => {
      channel.close();
    };
  }, []);

  const toggleTheme = () => {
    const newTheme = theme === 'light' ? 'dark' : 'light';
    setTheme(newTheme);
    localStorage.setItem('theme', newTheme);
    document.documentElement.setAttribute('data-theme', newTheme);
    channel.postMessage({ theme: newTheme });
  };

  return { theme, toggleTheme };
}

// 使用示例
function App() {
  const { theme, toggleTheme } = useTheme();

  return (
    <>
      <GlobalStyle />
      <button onClick={toggleTheme}>
        {theme === 'light' ? '🌙' : '☀️'}
      </button>
    </>
  );
}
```

### Vue 3 Composition API 实现

```javascript
import { ref, onMounted, watch } from 'vue';

export function useTheme() {
  const theme = ref('light');

  const applyTheme = (themeName) => {
    document.documentElement.setAttribute('data-theme', themeName);
    localStorage.setItem('theme', themeName);
  };

  const toggleTheme = () => {
    const newTheme = theme.value === 'light' ? 'dark' : 'light';
    theme.value = newTheme;
    applyTheme(newTheme);
  };

  const setTheme = (themeName) => {
    theme.value = themeName;
    applyTheme(themeName);
  };

  onMounted(() => {
    // 初始化主题
    const savedTheme = localStorage.getItem('theme') ||
      (window.matchMedia('(prefers-color-scheme: dark)').matches ? 'dark' : 'light');
    theme.value = savedTheme;
    applyTheme(savedTheme);

    // 监听系统主题变化
    if (window.matchMedia) {
      const darkModeQuery = window.matchMedia('(prefers-color-scheme: dark)');
      darkModeQuery.addEventListener('change', (e) => {
        if (!localStorage.getItem('theme')) {
          setTheme(e.matches ? 'dark' : 'light');
        }
      });
    }

    // 监听其他标签页
    const channel = new BroadcastChannel('theme');
    channel.onmessage = (event) => {
      setTheme(event.data.theme);
    };
  });

  watch(theme, (newTheme) => {
    // 同步到其他标签页
    const channel = new BroadcastChannel('theme');
    channel.postMessage({ theme: newTheme });
  });

  return {
    theme,
    toggleTheme,
    setTheme,
    isDark: computed(() => theme.value === 'dark')
  };
}

// 使用示例
<template>
  <button @click="toggleTheme">
    {{ isDark ? '☀️' : '🌙' }}
  </button>
</template>

<script setup>
import { useTheme } from './composables/useTheme';

const { theme, toggleTheme, isDark } = useTheme();
</script>
```

## 注意事项

### 1. 防止闪烁（FOUC）

主题切换时可能出现闪烁，需要避免：

```javascript
// 在 head 中尽早执行主题设置
<script>
  (function() {
    const theme = localStorage.getItem('theme') ||
      (window.matchMedia('(prefers-color-scheme: dark)').matches ? 'dark' : 'light');
    document.documentElement.setAttribute('data-theme', theme);
  })();
</script>
```

### 2. 可访问性

确保主题切换对所有用户都友好：

```html
<!-- 为按钮添加适当的 ARIA 属性 -->
<button
  id="themeToggle"
  aria-label="切换主题"
  aria-pressed="false"
>
  🌙
</button>

<script>
  const toggleButton = document.getElementById('themeToggle');
  toggleButton.addEventListener('click', () => {
    const currentTheme = document.documentElement.getAttribute('data-theme');
    const newTheme = currentTheme === 'light' ? 'dark' : 'light';
    toggleButton.setAttribute('aria-pressed', newTheme === 'dark' ? 'true' : 'false');
    toggleButton.setAttribute('aria-label', newTheme === 'dark' ? '切换到浅色主题' : '切换到深色主题');
  });
</script>
```

### 3. 图片适配

确保图片在不同主题下都清晰可见：

```css

  content: url('logo-light.png');
}
```

// 使用 requestAnimationFrame 优化主题切换
function smoothThemeChange(newTheme) {
  requestAnimationFrame(() => {
    document.documentElement.setAttribute('data-theme', newTheme);
    localStorage.setItem('theme', newTheme);
  });

// 防抖处理主题切换
function debounce(func, wait) {
  let timeout;
  return function executedFunction(...args) {
    const later = () => {
      clearTimeout(timeout);
      func(...args);
    };
    clearTimeout(timeout);
    timeout = setTimeout(later, wait);
  };
}

// 使用 BroadcastChannel API 实现跨标签页同步
class ThemeSync {
  constructor() {
    this.channel = new BroadcastChannel('theme');
    this.init();
  }

  init() {
    // 监听其他标签页的主题变化
    this.channel.onmessage = (event) => {
      const { theme } = event.data;
      document.documentElement.setAttribute('data-theme', theme);
      localStorage.setItem('theme', theme);
      this.updateUI(theme);
    };
  }

  broadcastThemeChange(theme) {
    this.channel.postMessage({ theme });
  }

  updateUI(theme) {
    // 更新 UI 状态
    const button = document.getElementById('themeToggle');
    button.textContent = theme === 'light' ? '🌙' : '☀️';
  }
}

## 总结

前端主题切换是提升用户体验的重要功能，通过合理的设计和实现，可以为用户提供更好的浏览体验。

主要要点：

1. **CSS 变量**：使用 CSS 变量是实现主题切换的最佳实践
2. **用户体验**：提供自动切换、手动切换和系统跟随多种选择
3. **性能优化**：注意避免闪烁，平滑过渡，确保性能
4. **可访问性**：确保主题切换对所有用户都友好
5. **跨设备同步**：使用 localStorage 和 BroadcastChannel 实现跨设备同步

记住：
- 从用户体验角度设计主题切换
- 考虑不同场景下的主题表现
- 测试各种边界情况和用户交互
- 保持代码的可维护性和扩展性

通过合理的主题切换实现，你的应用将更加现代化和用户友好。
