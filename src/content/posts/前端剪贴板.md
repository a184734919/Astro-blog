## 概述

前端剪贴板功能是Web应用中常用的交互功能，允许用户在浏览器中实现复制、粘贴、剪切等操作。从简单的文本复制到复杂的富文本、图片、文件等数据的剪贴板操作，剪贴板API为开发者提供了强大的能力。本文将详细介绍现代剪贴板API、兼容性处理以及实际应用场景。

## 核心概念

### 剪贴板API演进

**1. 传统方法**
- `document.execCommand('copy')`
- `document.execCommand('paste')`
- 兼容性好但功能有限

**2. 现代Clipboard API**
- `navigator.clipboard.writeText()`
- `navigator.clipboard.readText()`
- `navigator.clipboard.write()`
- `navigator.clipboard.read()`
- 功能强大，但需要HTTPS和用户权限

**3. ClipboardEvent**
- `copy` 事件
- `cut` 事件
- `paste` 事件
- 用于监听和自定义剪贴板操作

### 数据类型支持

**文本数据**
- 纯文本
- HTML富文本
- JSON数据

**二进制数据**
- 图片 (PNG, JPEG, GIF等)
- 音频
- 视频

**其他数据**
- 自定义数据格式
- 文件列表

```javascript
async function copyText(text) {
    try {

        return true;

    }
}
```

```javascript
// 使用现代Clipboard API读取剪贴板文本
async function readClipboardText() {
    try {
        const text = await navigator.clipboard.readText();
        console.log('剪贴板内容:', text);
        return text;
    } catch (error) {
        console.error('读取剪贴板失败:', error);
        return null;
    }
}

// 粘贴到输入框
async function pasteToInput(inputId) {
    try {
        const text = await readClipboardText();
        if (text !== null) {
            const input = document.getElementById(inputId);
            if (input) {
                input.value = text;
                return true;
            }
        }
        return false;
    } catch (error) {
        console.error('粘贴失败:', error);
        return false;
    }
}

// 监听粘贴事件
function setupPasteListener(inputId, callback) {
    const input = document.getElementById(inputId);
    if (input) {
        input.addEventListener('paste', async (event) => {
            event.preventDefault();
            
            try {
                const text = await navigator.clipboard.readText();
                if (callback) {
                    callback(text, event);
                } else {
                    input.value = text;
                }
            } catch (error) {
                console.error('粘贴失败:', error);
            }
        });
    }
}

// 传统方法处理粘贴
function setupPasteListenerFallback(inputId, callback) {
    const input = document.getElementById(inputId);
    if (input) {
        input.addEventListener('paste', (event) => {
            const pastedData = (event.clipboardData || window.clipboardData).getData('text');
            
            if (callback) {
                callback(pastedData, event);
            } else {
                input.value = pastedData;
            }
        });
    }
}

// 智能粘贴处理器
class SmartPasteHandler {
    constructor(options = {}) {
        this.maxLength = options.maxLength || Infinity;
        this.validator = options.validator || (() => true);
        this.formatter = options.formatter || (text => text);
        this.onError = options.onError || (() => {});
    }
    
    async handlePaste(event) {
        event.preventDefault();
        
        try {
            // 尝试使用现代API
            const text = await navigator.clipboard.readText();
            this.processText(text, event);
        } catch (error) {
            // 回退到传统方法
            const text = this.getClipboardDataFallback(event);
            this.processText(text, event);
        }
    }
    
    getClipboardDataFallback(event) {
        const clipboardData = event.clipboardData || window.clipboardData;
        return clipboardData.getData('text');
    }
    
    processText(text, event) {
        try {
            // 验证文本
            if (!this.validator(text)) {
                throw new Error('文本验证失败');
            }
            
            // 格式化文本
            const formattedText = this.formatter(text);
            
            // 长度限制
            if (formattedText.length > this.maxLength) {
                const truncated = formattedText.substring(0, this.maxLength);
                this.onPaste(truncated, event);
                this.onError('文本过长，已截断');
            } else {
                this.onPaste(formattedText, event);
            }
        } catch (error) {
            console.error('粘贴处理失败:', error);
            this.onError(error.message);
        }
    }
    
    onPaste(text, event) {
        // 默认行为：插入到当前光标位置
        const input = event.target;
        if (input && (input.tagName === 'INPUT' || input.tagName === 'TEXTAREA')) {
            const start = input.selectionStart;
            const end = input.selectionEnd;
            const value = input.value;
            
            input.value = value.substring(0, start) + text + value.substring(end);
            input.selectionStart = input.selectionEnd = start + text.length;
        }
    }
}

// 使用示例
const pasteHandler = new SmartPasteHandler({
    maxLength: 1000,
    validator: (text) => {
        // 示例：只允许数字和字母
        return /^[a-zA-Z0-9\s]*$/.test(text);
    },
    formatter: (text) => {
        // 示例：转换为大写并去除空格
        return text.toUpperCase().replace(/\s/g, '');
    },
    onError: (message) => {
        alert(message);
    }
});

const inputElement = document.getElementById('paste-input');
inputElement.addEventListener('paste', pasteHandler.handlePaste.bind(pasteHandler));
```

```javascript
// 复制HTML内容
async function copyHTML(html) {
    try {
        const blob = new Blob([html], { type: 'text/html' });
        const item = new ClipboardItem({
            'text/html': blob,
            'text/plain': new Blob([stripHTML(html)], { type: 'text/plain' })
        });
        
        await navigator.clipboard.write([item]);
        console.log('HTML复制成功');
        return true;
    } catch (error) {
        console.error('HTML复制失败:', error);
        return false;
    }
}

// 去除HTML标签
function stripHTML(html) {
    const div = document.createElement('div');
    div.innerHTML = html;
    return div.textContent || div.innerText || '';
}

// 复制元素内容
async function copyElementContent(selector) {
    const element = document.querySelector(selector);
    if (!element) {
        console.error('元素不存在');
        return false;
    }
    
    const html = element.innerHTML;
    const text = element.textContent;
    
    try {
        const htmlBlob = new Blob([html], { type: 'text/html' });
        const textBlob = new Blob([text], { type: 'text/plain' });
        const item = new ClipboardItem({
            'text/html': htmlBlob,
            'text/plain': textBlob
        });
        
        await navigator.clipboard.write([item]);
        console.log('元素内容复制成功');
        return true;
    } catch (error) {
        console.error('复制失败:', error);
        return false;
    }
}

// 富文本复制组件
class RichTextCopier {
    constructor(options = {}) {
        this.selector = options.selector || '';
        this.includeStyles = options.includeStyles || false;
        this.successCallback = options.onSuccess || (() => {});
        this.errorCallback = options.onError || (() => {});
    }
    
    async copy() {
        const element = document.querySelector(this.selector);
        if (!element) {
            this.errorCallback('元素不存在');
            return false;
        }
        
        try {
            const content = this.prepareContent(element);
            const success = await this.copyToClipboard(content);
            
            if (success) {
                this.successCallback();
            } else {
                this.errorCallback('复制失败');
            }
            
            return success;
        } catch (error) {
            console.error('复制错误:', error);
            this.errorCallback(error.message);
            return false;
        }
    }
    
    prepareContent(element) {
        // 克隆元素以避免修改原始内容
        const clone = element.cloneNode(true);
        
        // 如果需要包含样式
        if (this.includeStyles) {
            this.copyStyles(element, clone);
        }
        
        return clone;
    }
    
    copyStyles(source, target) {
        // 获取计算样式
        const computedStyle = window.getComputedStyle(source);
        
        // 复制关键样式属性
        const styleProperties = [
            'font-family', 'font-size', 'font-weight',
            'color', 'background-color',
            'text-align', 'line-height',
            'padding', 'margin', 'border'
        ];
        
        styleProperties.forEach(property => {
            target.style[property] = computedStyle.getPropertyValue(property);
        });
        
        // 递归处理子元素
        Array.from(source.children).forEach((child, index) => {
            if (target.children[index]) {
                this.copyStyles(child, target.children[index]);
            }
        });
    }
    
    async copyToClipboard(content) {
        try {
            const html = content.outerHTML;
            const text = content.textContent;
            
            const htmlBlob = new Blob([html], { type: 'text/html' });
            const textBlob = new Blob([text], { type: 'text/plain' });
            
            const item = new ClipboardItem({
                'text/html': htmlBlob,
                'text/plain': textBlob
            });
            
            await navigator.clipboard.write([item]);
            return true;
        } catch (error) {
            console.error('剪贴板写入失败:', error);
            
            // 回退到传统方法
            return this.fallbackCopy(content);
        }
    }
    
    fallbackCopy(content) {
        const range = document.createRange();
        range.selectNode(content);
        
        const selection = window.getSelection();
        selection.removeAllRanges();
        selection.addRange(range);
        
        try {
            const successful = document.execCommand('copy');
            selection.removeAllRanges();
            return successful;
        } catch (error) {
            console.error('回退复制失败:', error);
            selection.removeAllRanges();
            return false;
        }
    }
}

// 使用示例
const richTextCopier = new RichTextCopier({
    selector: '#rich-content',
    includeStyles: true,
    onSuccess: () => alert('复制成功！'),
    onError: (message) => alert('复制失败: ' + message)
});

// 绑定到按钮
document.getElementById('copy-rich').addEventListener('click', () => {
    richTextCopier.copy();
});
```

```javascript
// 一键复制管理器
class OneClickCopyManager {
    constructor() {
        this.copyButtons = [];
        this.init();
    }
    
    init() {
        // 自动查找所有带有 data-copy 属性的元素
        this.autoSetup();
        
        // 监听动态添加的元素
        this.observeDOMChanges();
    }
    
    autoSetup() {
        const elements = document.querySelectorAll('[data-copy]');
        elements.forEach(element => {
            this.setupCopyButton(element);
        });
    }
    
    setupCopyButton(element) {
        const text = element.dataset.copy;
        const target = element.dataset.copyTarget;
        
        const button = document.createElement('button');
        button.className = 'copy-icon';
        button.innerHTML = '📋';
        button.title = '点击复制';
        
        button.addEventListener('click', async (e) => {
            e.preventDefault();
            await this.handleCopy(button, text, target);
        });
        
        element.appendChild(button);
        this.copyButtons.push(button);
    }
    
    async handleCopy(button, text, target) {
        let textToCopy = text;
        
        if (target) {
            const targetElement = document.querySelector(target);
            if (targetElement) {
                textToCopy = targetElement.value || targetElement.textContent;
            }
        }
        
        if (!textToCopy) {
            this.showError(button, '没有可复制的内容');
            return;
        }
        
        try {
            const success = await copyText(textToCopy);
            
            if (success) {
                this.showSuccess(button);
            } else {
                this.showError(button, '复制失败');
            }
        } catch (error) {
            console.error('复制错误:', error);
            this.showError(button, '复制出错');
        }
    }
    
    showSuccess(button) {
        const originalHTML = button.innerHTML;
        button.innerHTML = '✅';
        button.classList.add('success');
        
        setTimeout(() => {
            button.innerHTML = originalHTML;
            button.classList.remove('success');
        }, 1500);
    }
    
    showError(button, message) {
        const originalHTML = button.innerHTML;
        button.innerHTML = '❌';
        button.classList.add('error');
        button.title = message;
        
        setTimeout(() => {
            button.innerHTML = originalHTML;
            button.classList.remove('error');
            button.title = '点击复制';
        }, 2000);
    }
    
    observeDOMChanges() {
        const observer = new MutationObserver((mutations) => {
            mutations.forEach(mutation => {
                mutation.addedNodes.forEach(node => {
                    if (node.nodeType === 1) { // Element node
                        const newElements = node.querySelectorAll('[data-copy]');
                        newElements.forEach(element => {
                            this.setupCopyButton(element);
                        });
                    }
                });
            });
        });
        
        observer.observe(document.body, {
            childList: true,
            subtree: true
        });
    }
}
```

```javascript
// 高级代码块复制器
class AdvancedCodeBlockCopier {
    constructor(options = {}) {
        this.options = {
            includeLineNumbers: options.includeLineNumbers || false,
            stripComments: options.stripComments || false,
            formatCode: options.formatCode || false,
            language: options.language || 'plaintext',
            ...options
        };
        
        this.init();
    }
    
    init() {
        const codeBlocks = document.querySelectorAll('pre code');
        codeBlocks.forEach(codeBlock => {
            this.addCopyButton(codeBlock);
        });
    }
    
    addCopyButton(codeBlock) {
        const wrapper = codeBlock.parentElement;
        
        const buttonContainer = document.createElement('div');
        buttonContainer.className = 'code-copy-container';
        buttonContainer.style.cssText = `
            position: absolute;
            top: 10px;
            right: 10px;
            display: flex;
            gap: 5px;
            z-index: 10;
        `;
        
        // 复制按钮
        const copyButton = this.createButton('复制', async () => {
            const code = this.prepareCode(codeBlock);
            const success = await copyText(code);
            this.showButtonFeedback(copyButton, success ? '已复制' : '失败');
        });
        
        // 复制带行号按钮
        if (this.options.includeLineNumbers) {
            const copyWithLinesButton = this.createButton('复制+行号', async () => {
                const code = this.getCodeWithLineNumbers(codeBlock);
                const success = await copyText(code);
                this.showButtonFeedback(copyWithLinesButton, success ? '已复制' : '失败');
            });
            buttonContainer.appendChild(copyWithLinesButton);
        }
        
        buttonContainer.appendChild(copyButton);
        wrapper.appendChild(buttonContainer);
    }
    
    createButton(text, onClick) {
        const button = document.createElement('button');
        button.className = 'code-copy-button';
        button.textContent = text;
        button.style.cssText = `
            padding: 5px 10px;
            background: #4CAF50;
            color: white;
            border: none;
            border-radius: 4px;
            cursor: pointer;
            font-size: 12px;
            transition: background 0.2s;
        `;
        
        button.addEventListener('mouseenter', () => {
            button.style.background = '#45a049';
        });
        
        button.addEventListener('mouseleave', () => {
            button.style.background = '#4CAF50';
        });
        
        button.addEventListener('click', onClick);
        
        return button;
    }
    
    prepareCode(codeBlock) {
        let code = codeBlock.textContent;
        
        // 去除注释
        if (this.options.stripComments) {
            code = this.stripComments(code, this.options.language);
        }
        
        // 格式化代码
        if (this.options.formatCode) {
            code = this.formatCode(code, this.options.language);
        }
        
        return code;
    }
    
    getCodeWithLineNumbers(codeBlock) {
        const lines = codeBlock.textContent.split('\n');
        const numberedLines = lines.map((line, index) => {
            return `${String(index + 1).padStart(3, ' ')} | ${line}`;
        });
        return numberedLines.join('\n');
    }
    
    stripComments(code, language) {
        // 简化的注释去除逻辑
        const commentPatterns = {
            javascript: /(\/\/.*$)|(\/\*[\s\S]*?\*\/)/gm,
            python: /(#.*$)|("""[\s\S]*?"""|'''[\s\S]*?''')/gm,
            css: /(\/\*[\s\S]*?\*\/)/gm,
            html: /(<!--[\s\S]*?-->)/gm
        };
        
        const pattern = commentPatterns[language] || commentPatterns.javascript;
        return code.replace(pattern, '');
    }
    
    formatCode(code, language) {
        // 简化的代码格式化
        // 实际项目中应该使用专业的格式化库
        return code.trim();
    }
    
    showButtonFeedback(button, message) {
        const originalText = button.textContent;
        button.textContent = message;
        button.disabled = true;
        
        setTimeout(() => {
            button.textContent = originalText;
            button.disabled = false;
        }, 1500);
    }
}
```

```javascript
class ImageClipboardHandler {
    constructor(options = {}) {
        this.onImagePaste = options.onImagePaste || (() => {});
        this.onError = options.onError || (() => {});
        this.maxSize = options.maxSize || 10 * 1024 * 1024; // 10MB
        
        this.init();
    }
    
    init() {
        document.addEventListener('paste', this.handlePaste.bind(this));
    }
    
    async handlePaste(event) {
        const items = event.clipboardData?.items;
        
        if (!items) return;
        
        for (const item of items) {
            if (item.type.startsWith('image/')) {
                event.preventDefault();
                
                try {
                    const blob = item.getAsFile();
                    
                    // 检查文件大小
                    if (blob.size > this.maxSize) {
                        throw new Error(`图片大小超过限制 (${this.formatFileSize(this.maxSize)})`);
                    }
                    
                    // 处理图片
                    await this.processImage(blob);
                    
                } catch (error) {
                    console.error('图片粘贴失败:', error);
                    this.onError(error);
                }
                
                break;
            }
        }
    }
    
    async processImage(blob) {
        // 读取图片信息
        const imageInfo = await this.getImageInfo(blob);
        
        // 创建预览
        const preview = await this.createPreview(blob);
        
        // 调用回调
        this.onImagePaste({
            blob,
            preview,
            info: imageInfo
        });
    }
    
    async getImageInfo(blob) {
        return new Promise((resolve, reject) => {
            const img = new Image();
            const url = URL.createObjectURL(blob);
            
            img.onload = () => {
                resolve({
                    width: img.naturalWidth,
                    height: img.naturalHeight,
                    aspectRatio: img.naturalWidth / img.naturalHeight,
                    size: blob.size,
                    type: blob.type
                });
                URL.revokeObjectURL(url);
            };
            
            img.onerror = () => {
                reject(new Error('图片加载失败'));
                URL.revokeObjectURL(url);
            };
            
            img.src = url;
        });
    }
    
    async createPreview(blob) {
        return new Promise((resolve, reject) => {
            const reader = new FileReader();
            
            reader.onload = (e) => {
                resolve(e.target.result);
            };
            
            reader.onerror = () => {
                reject(new Error('预览生成失败'));
            };
            
            reader.readAsDataURL(blob);
        });
    }
    
    formatFileSize(bytes) {
        if (bytes === 0) return '0 B';
        const k = 1024;
        const sizes = ['B', 'KB', 'MB', 'GB'];
        const i = Math.floor(Math.log(bytes) / Math.log(k));
        return Math.round((bytes / Math.pow(k, i)) * 100) / 100 + ' ' + sizes[i];
    }
    
    destroy() {
        document.removeEventListener('paste', this.handlePaste.bind(this));
    }
}
```

```javascript
// 剪贴板API兼容性检测
class ClipboardCompatibility {
    constructor() {
        this.capabilities = this.detectCapabilities();
    }
    
    detectCapabilities() {
        return {
            modernAPI: this.checkModernAPI(),
            execCommand: this.checkExecCommand(),
            clipboardEvents: this.checkClipboardEvents(),
            permissionAPI: this.checkPermissionAPI()
        };
    }
    
    checkModernAPI() {
        return 'clipboard' in navigator && 
               'writeText' in navigator.clipboard &&
               'readText' in navigator.clipboard;
    }
    
    checkExecCommand() {
        return 'execCommand' in document;
    }
    
    checkClipboardEvents() {
        return 'oncopy' in document && 
               'oncut' in document && 
               'onpaste' in document;
    }
    
    checkPermissionAPI() {
        return 'permissions' in navigator;
    }
    
    async requestPermission() {
        if (!this.capabilities.permissionAPI) {
            return false;
        }
        
        try {
            const permission = await navigator.permissions.query({
                name: 'clipboard-write'
            });
            
            if (permission.state === 'granted' || permission.state === 'prompt') {
                return true;
            }
            
            return false;
        } catch (error) {
            console.error('权限检查失败:', error);
            return false;
        }
    }
    
    getBestMethod() {
        if (this.capabilities.modernAPI) {
            return 'modern';
        } else if (this.capabilities.execCommand) {
            return 'fallback';
        } else {
            return 'unsupported';
        }
    }
    
    isSupported() {
        return this.getBestMethod() !== 'unsupported';
    }
}

// 兼容性包装器
class CompatibleClipboard {
    constructor() {
        this.compatibility = new ClipboardCompatibility();
    }
    
    async writeText(text) {
        const method = this.compatibility.getBestMethod();
        
        switch (method) {
            case 'modern':
                return await this.writeTextModern(text);
            case 'fallback':
                return this.writeTextFallback(text);
            default:
                throw new Error('剪贴板功能不支持');
        }
    }
    
    async writeTextModern(text) {
        try {
            await navigator.clipboard.writeText(text);
            return true;
        } catch (error) {
            console.error('现代API写入失败:', error);
            // 回退到传统方法
            return this.writeTextFallback(text);
        }
    }
    
    writeTextFallback(text) {
        const textarea = document.createElement('textarea');
        textarea.value = text;
        textarea.style.position = 'fixed';
        textarea.style.opacity = '0';
        document.body.appendChild(textarea);
        textarea.select();
        
        try {
            const successful = document.execCommand('copy');
            return successful;
        } catch (error) {
            console.error('传统方法写入失败:', error);
            return false;
        } finally {
            document.body.removeChild(textarea);
        }
    }
    
    async readText() {
        if (!this.compatibility.capabilities.modernAPI) {
            throw new Error('剪贴板读取功能不支持');
        }
        
        try {
            return await navigator.clipboard.readText();
        } catch (error) {
            console.error('剪贴板读取失败:', error);
            return null;
        }
    }
}
```

```javascript
// 安全的剪贴板操作
class SecureClipboard {
    constructor() {
        this.maxLength = 10000; // 最大文本长度
        this.allowedDomains = ['example.com', 'trusted-site.com'];
    }
    
    validateText(text) {
        // 长度检查
        if (text.length > this.maxLength) {
            throw new Error(`文本长度超过限制 (${this.maxLength} 字符)`);
        }
        
        // 内容安全检查
        if (this.containsMaliciousContent(text)) {
            throw new Error('文本包含恶意内容');
        }
        
        return true;
    }
    
    containsMaliciousContent(text) {
        // 检查是否包含脚本标签
        if (/<script[^>]*>.*?<\/script>/gi.test(text)) {
            return true;
        }
        
        // 检查是否包含可疑的JavaScript代码
        if (/javascript:/gi.test(text)) {
            return true;
        }
        
        // 检查是否包含数据URI
        if (/data:/gi.test(text)) {
            return true;
        }
        
        return false;
    }
    
    sanitizeText(text) {
        // 移除HTML标签
        const div = document.createElement('div');
        div.textContent = text;
        return div.innerHTML;
    }
    
    async secureWrite(text) {
        try {
            // 验证文本
            this.validateText(text);
            
            // 清理文本
            const sanitizedText = this.sanitizeText(text);
            
            // 写入剪贴板
            await navigator.clipboard.writeText(sanitizedText);
            
            console.log('安全写入成功');
            return true;
        } catch (error) {
            console.error('安全写入失败:', error);
            return false;
        }
    }
    
    async secureRead() {
        try {
            const text = await navigator.clipboard.readText();
            
            // 验证读取的内容
            this.validateText(text);
            
            return text;
        } catch (error) {
            console.error('安全读取失败:', error);
            return null;
        }
    }
    
    async requestWritePermission() {
        try {
            // 检查权限状态
            const permission = await navigator.permissions.query({
                name: 'clipboard-write'
            });
            
            if (permission.state === 'granted') {
                return true;

            if (permission.state === 'prompt') {
                // 尝试写入以触发权限请求
                try {
                    await navigator.clipboard.writeText('');
                    return true;
                } catch (error) {
                    return false;
                }
            }
            
            return false;
        } catch (error) {
            console.error('权限检查失败:', error);
            return false;
        }
    }
}
```

### 3. 性能优化

```javascript
// 剪贴板操作性能优化
class OptimizedClipboard {
    constructor() {
        this.cache = new Map();
        this.debounceTimers = new Map();
        this.throttleTimers = new Map();
    }
    
    // 防抖复制
    debouncedCopy(text, delay = 300) {
        return new Promise((resolve) => {
            const timerId = 'copy_' + text;
            
            if (this.debounceTimers.has(timerId)) {
                clearTimeout(this.debounceTimers.get(timerId));
            }
            
            const timer = setTimeout(async () => {
                const result = await this.copyText(text);
                this.debounceTimers.delete(timerId);
                resolve(result);
            }, delay);
            
            this.debounceTimers.set(timerId, timer);
        });
    }
    
    // 节流复制
    throttledCopy(text, interval = 1000) {
        return new Promise((resolve) => {
            const timerId = 'copy_' + text;
            
            if (this.throttleTimers.has(timerId)) {
                resolve(false);
                return;
            }
            
            this.copyText(text).then(result => {
                resolve(result);
            });
            
            const timer = setTimeout(() => {
                this.throttleTimers.delete(timerId);
            }, interval);
            
            this.throttleTimers.set(timerId, timer);
        });
    }
    
    // 缓存复制内容
    async cachedCopy(text) {
        const cacheKey = this.generateCacheKey(text);
        
        if (this.cache.has(cacheKey)) {
            console.log('使用缓存内容');
            return this.cache.get(cacheKey);
        }
        
        const result = await this.copyText(text);
        
        if (result) {
            this.cache.set(cacheKey, result);
            
            // 设置缓存过期
            setTimeout(() => {
                this.cache.delete(cacheKey);
            }, 60000); // 1分钟过期
        }
        
        return result;
    }
    
    generateCacheKey(text) {
        // 生成缓存键
        return btoa(text.substring(0, 100));
    }
    
    // 批量复制
    async batchCopy(texts, concurrency = 3) {
        const results = [];
        
        for (let i = 0; i < texts.length; i += concurrency) {
            const batch = texts.slice(i, i + concurrency);
            const batchResults = await Promise.all(
                batch.map(text => this.copyText(text))
            );
            results.push(...batchResults);
        }
        
        return results;
    }
    
    async copyText(text) {
        try {
            await navigator.clipboard.writeText(text);
            return true;
        } catch (error) {
            console.error('复制失败:', error);
            return false;
        }
    }
    
    // 清理资源
    cleanup() {
        // 清除所有计时器
        this.debounceTimers.forEach(timer => clearTimeout(timer));
        this.throttleTimers.forEach(timer => clearTimeout(timer));
        
        // 清除缓存
        this.cache.clear();
        
        // 清空计时器集合
        this.debounceTimers.clear();
        this.throttleTimers.clear();
    }
}
```
