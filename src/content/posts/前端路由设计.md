## 概述

前端路由是现代单页应用（SPA）的核心技术之一，它允许在不重新加载页面的情况下，通过改变 URL 来切换不同的视图和组件。

### 路由的作用

- **视图切换**：根据 URL 显示对应的页面内容
- **状态保持**：通过 URL 参数和路由参数传递状态
- **历史记录**：支持浏览器的前进后退功能
- **深度链接**：可以直接访问应用的特定页面
- **代码分割**：配合懒加载提升应用性能

### 路由原理

前端路由主要通过两种方式实现：

1. **Hash 模式**：使用 URL 的 hash 部分（#）实现
2. **History 模式**：使用 HTML5 History API 实现

## 核心概念

### 路由模式对比

```javascript
// Hash 模式
// URL: http://example.com/#/user/123
// 特点：
// - 兼容性好，支持所有浏览器
// - 不需要服务器配置
// - URL 中包含 # 符号
// - 无法利用 SEO

// History 模式
// URL: http://example.com/user/123
// 特点：
// - URL 更美观，更符合传统 URL 格式
// - 支持 SEO
// - 需要服务器配置支持
// - 兼容性要求 HTML5 History API
```

### 路由基础概念

```javascript
// 路由配置的基本概念
const routeConfig = {
  path: '/users/:id',           // 路由路径，支持动态参数
  component: UserDetail,        // 对应的组件
  name: 'UserDetail',           // 路由名称，用于导航
  meta: {                       // 路由元信息
    requiresAuth: true,         // 是否需要认证
    permission: 'user:view',    // 权限要求
    title: '用户详情'           // 页面标题
  },
  redirect: '/users',           // 重定向路径
  alias: ['/user/:id'],         // 别名路径
  props: true                   // 将路由参数作为 props 传递
};
```

### 1. React Router v6 基础使用

```javascript
// App.jsx
import { BrowserRouter, Routes, Route, Link, useNavigate, useParams, useLocation } from 'react-router-dom';
import { lazy, Suspense } from 'react';

// 懒加载组件
const Home = lazy(() => import('./pages/Home'));
const About = lazy(() => import('./pages/About'));
const UserList = lazy(() => import('./pages/UserList'));
const UserDetail = lazy(() => import('./pages/UserDetail'));
const NotFound = lazy(() => import('./pages/NotFound'));

function App() {
  return (
    <BrowserRouter>
      <nav>
        <Link to="/">首页</Link>
        <Link to="/about">关于</Link>
        <Link to="/users">用户列表</Link>
      </nav>

      <Suspense fallback={<div>Loading...</div>}>
        <Routes>
          <Route path="/" element={<Home />} />
          <Route path="/about" element={<About />} />
          <Route path="/users" element={<UserList />} />
          <Route path="/users/:id" element={<UserDetail />} />
          <Route path="*" element={<NotFound />} />
        </Routes>
      </Suspense>
    </BrowserRouter>
  );
}

// UserDetail.jsx - 获取路由参数
function UserDetail() {
  const { id } = useParams();
  const location = useLocation();
  const navigate = useNavigate();

  const handleGoBack = () => {
    navigate(-1); // 返回上一页
  };

  const handleGoHome = () => {
    navigate('/'); // 跳转到首页
  };

  return (
    <div>
      <h1>用户详情</h1>
      <p>用户ID: {id}</p>
      <p>当前路径: {location.pathname}</p>
      <button onClick={handleGoBack}>返回</button>
      <button onClick={handleGoHome}>首页</button>
    </div>
  );
}
```

```javascript
function ProtectedRoute({ children, isAuthenticated, requiredPermission }) {
  const { user } = useAuth();

  if (!isAuthenticated) {
    return <Navigate to="/login" replace />;
  }

  if (requiredPermission && !user?.permissions?.includes(requiredPermission)) {
    return <Navigate to="/403" replace />;
  }

  return children;
}

            <ProtectedRoute
              isAuthenticated={true}
              requiredPermission="admin:access"
            >
              <AdminPanel />
            </ProtectedRoute>
          }
        />
      </Routes>
    </BrowserRouter>
  );
}
```

```javascript
// Vue Router 全局路由守卫
router.beforeEach(async (to, from, next) => {
  const userStore = useUserStore();

  // 设置页面标题
  document.title = to.meta.title || '默认标题';

  // 检查认证状态
  if (to.meta.requiresAuth && !userStore.isAuthenticated) {
    next({
      path: '/login',
      query: { redirect: to.fullPath }
    });
    return;
  }

  // 检查权限
  if (to.meta.permission && !userStore.hasPermission(to.meta.permission)) {
    next('/403');
    return;
  }

  next();
});

router.afterEach((to, from) => {
  // 可以在这里进行页面统计、日志记录等
  console.log(`Navigation from ${from.path} to ${to.path}`);
});

// 组件内守卫
export default {
  beforeRouteEnter(to, from, next) {
    // 在渲染该组件的对应路由被验证前调用
    next(vm => {
      // 通过 `vm` 访问组件实例
    });
  },

  beforeRouteUpdate(to, from, next) {
    // 在当前路由改变，但是该组件被复用时调用
    next();
  },

  beforeRouteLeave(to, from, next) {
    // 导航离开该组件的对应路由时调用
    if (this.hasUnsavedChanges) {
      const answer = window.confirm('确定要离开吗？未保存的更改将丢失。');
      if (answer) {
        next();
      } else {
        next(false);
      }
    } else {
      next();
    }
  }
};
```

### 3. 路由状态管理

```javascript
// 创建路由历史记录管理
class RouterHistory {
  constructor(maxSize = 50) {
    this.history = [];
    this.currentIndex = -1;
    this.maxSize = maxSize;
  }

  push(path) {
    // 删除当前位置之后的历史记录
    this.history = this.history.slice(0, this.currentIndex + 1);
    this.history.push({
      path,
      timestamp: Date.now()
    });
    this.currentIndex++;

    // 限制历史记录大小
    if (this.history.length > this.maxSize) {
      this.history.shift();
      this.currentIndex--;
    }
  }

  canGoBack() {
    return this.currentIndex > 0;
  }

  canGoForward() {
    return this.currentIndex < this.history.length - 1;
  }

  back() {
    if (this.canGoBack()) {
      this.currentIndex--;
      return this.history[this.currentIndex].path;
    }
    return null;
  }

  forward() {
    if (this.canGoForward()) {
      this.currentIndex++;
      return this.history[this.currentIndex].path;
    }
    return null;
  }

  clear() {
    this.history = [];
    this.currentIndex = -1;
  }
}

// 使用示例
const routerHistory = new RouterHistory();

// 监听路由变化
router.afterEach((to) => {
  routerHistory.push(to.path);
});

// 自定义导航函数
function customNavigate(direction) {
  let targetPath;

  switch (direction) {
    case 'back':
      targetPath = routerHistory.back();
      break;
    case 'forward':
      targetPath = routerHistory.forward();
      break;
    default:
      return;
  }

  if (targetPath) {
    router.push(targetPath);
  }
}
```

```javascript
// 路由参数验证中间件
function validateRouteParams(to, from, next) {
  const { id } = to.params;

  // 验证 ID 格式
  if (id && !/^[a-f0-9]{24}$/.test(id)) {
    next('/404');
    return;
  }

  // 验证查询参数
  const { page, limit } = to.query;
  if (page && (isNaN(page) || page < 1)) {
    next({
      path: to.path,
      query: { ...to.query, page: 1 }
    });
    return;
  }

  if (limit && (isNaN(limit) || limit < 1 || limit > 100)) {
    next({
      path: to.path,
      query: { ...to.query, limit: 20 }
    });
    return;
  }

  next();
}

// 应用到路由
router.beforeEach(validateRouteParams);
```

```javascript
// 预加载路由组件
function preloadRouteComponent(routePath) {
  return new Promise((resolve) => {
    const link = document.createElement('link');
    link.rel = 'prefetch';
    link.href = routePath;
    link.onload = resolve;
    link.onerror = resolve;
    document.head.appendChild(link);
  });
}

// 智能预加载
function setupSmartPreload() {
  const preloadRoutes = [
    '/users',
    '/settings',
    '/profile'
  ];

  // 鼠标悬停时预加载
  document.querySelectorAll('a[href^="/"]').forEach(link => {
    link.addEventListener('mouseenter', () => {
      const path = link.getAttribute('href');
      if (preloadRoutes.includes(path)) {
        preloadRouteComponent(path);
      }
    });
  });

  // 空闲时预加载
  if ('requestIdleCallback' in window) {
    requestIdleCallback(() => {
      preloadRoutes.forEach(route => {
        preloadRouteComponent(route);
      });
    });
  }
}

function lazyWithTimeout(importFn, timeout = 5000) {
  return lazy(() => {

      importFn(),
      new Promise((_, reject) =>
        setTimeout(() => reject(new Error('Component load timeout')), timeout)
```

### 6. 路由错误处理

```javascript
// 路由错误边界
class RouterErrorBoundary extends React.Component {
  state = { hasError: false, error: null };

  static getDerivedStateFromError(error) {
    return { hasError: true, error };
  }

  componentDidCatch(error, errorInfo) {
    console.error('Router Error:', error, errorInfo);
    // 上报错误
    logError(error, errorInfo);
  }

  handleRetry = () => {
    this.setState({ hasError: false, error: null });
  };

  render() {
    if (this.state.hasError) {
      return (
        <div className="error-boundary">
          <h1>路由加载失败</h1>
          <p>{this.state.error?.message}</p>
          <button onClick={this.handleRetry}>重试</button>
          <button onClick={() => window.location.href = '/'}>返回首页</button>
        </div>
      );
    }

    return this.props.children;
  }
}

// 在应用中使用
function App() {
  return (
    <RouterErrorBoundary>
      <BrowserRouter>
        {/* 路由配置 */}
      </BrowserRouter>
    </RouterErrorBoundary>
  );
}
```

## 总结

前端路由设计是构建现代 Web 应用的关键环节，需要综合考虑多个方面：

### 核心概念

1. **路由模式**：Hash 模式和 History 模式的选择
2. **路由配置**：路径、组件、参数、元信息的配置
3. **导航控制**：声明式导航和编程式导航
4. **路由守卫**：全局守卫、路由独享守卫、组件内守卫

### 实践要点

1. **性能优化**：懒加载、预加载、代码分割
2. **用户体验**：过渡动画、加载状态、错误处理
3. **安全性**：权限控制、参数验证、XSS 防护
4. **SEO 优化**：服务端渲染、Meta 标签管理
