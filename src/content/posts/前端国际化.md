## 概述

前端国际化（Internationalization，简称 i18n）是指将应用设计为能够适应不同语言和地区的用户需求。随着全球化的发展，支持多语言已成为现代 Web 应用的标配功能。本文将详细介绍前端国际化的实现方案、最佳实践和实际应用。

## 核心概念

### i18n 基本概念

1. **国际化（i18n）**：设计和开发应用的过程，使其能够适应不同语言和地区
2. **本地化（l10n）**：将国际化应用适配到特定语言和地区的过程
3. **区域设置（locale）**：语言和地区的组合，如 `zh-CN`、`en-US`
4. **资源文件**：存储翻译内容的文件，通常按语言分开

### 术语缩写

- i18n = internationalization（首字母 + 18个字母 + 尾字母）
- l10n = localization（首字母 + 10个字母 + 尾字母）
- g11n = globalization（首字母 + 11个字母 + 尾字母）

### 2. 翻译函数实现

```javascript
class I18n {
  constructor(options = {}) {
    this.locale = options.locale || this.detectLocale();
    this.fallbackLocale = options.fallbackLocale || 'en-US';
    this.messages = options.messages || {};
    this.missingTranslationHandler = options.missingTranslationHandler || null;
    this.formatters = {
      number: this.numberFormatter,
      date: this.dateFormatter,
      currency: this.currencyFormatter,
      plural: this.pluralFormatter
    };
  }

  detectLocale() {
    // 1. 检查本地存储
    const stored = localStorage.getItem('locale');
    if (stored) return stored;

    // 2. 检查浏览器语言
    const browserLocale = navigator.language;
    if (browserLocale && this.hasLocale(browserLocale)) {
      return browserLocale;
    }

    // 3. 返回默认语言
    return this.fallbackLocale;
  }

  hasLocale(locale) {
    return Object.keys(this.messages).includes(locale);
  }

  setLocale(locale) {
    if (this.hasLocale(locale)) {
      this.locale = locale;
      localStorage.setItem('locale', locale);
      this.notifyLocaleChange();
    } else {
      console.warn(`Locale ${locale} not found, using ${this.fallbackLocale}`);
      this.locale = this.fallbackLocale;
    }
  }

  t(key, values = {}) {
    let message = this.getMessage(key, this.locale);

    // 如果找不到翻译，尝试回退语言
    if (!message && this.locale !== this.fallbackLocale) {
      message = this.getMessage(key, this.fallbackLocale);
    }

    // 仍然找不到，调用缺失处理器或返回键名
    if (!message) {
      if (this.missingTranslationHandler) {
        return this.missingTranslationHandler(key, this.locale, values);
      }
      return key;
    }

    // 处理插值
    return this.interpolate(message, values);
  }

  getMessage(key, locale) {
    const keys = key.split('.');
    let current = this.messages[locale];

    for (const k of keys) {
      if (current && typeof current === 'object' && k in current) {
        current = current[k];
      } else {
        return null;
      }
    }

    return current;
  }

  interpolate(message, values) {
    return message.replace(/\{\{(\w+)\}\}/g, (match, key) => {
      return values[key] !== undefined ? values[key] : match;
    });
  }

  // 数字格式化
  numberFormatter(value, options = {}) {
    const defaultOptions = {
      style: 'decimal',
      minimumFractionDigits: 0,
      maximumFractionDigits: 2,
      ...options
    };
    return new Intl.NumberFormat(this.locale, defaultOptions).format(value);
  }

  // 日期格式化
  dateFormatter(value, options = {}) {
    const defaultOptions = {
      year: 'numeric',
      month: 'long',
      day: 'numeric',
      ...options
    };
    return new Intl.DateTimeFormat(this.locale, defaultOptions).format(new Date(value));
  }

  // 货币格式化
  currencyFormatter(value, currency = 'USD', options = {}) {
    const defaultOptions = {
      style: 'currency',
      currency: currency,
      minimumFractionDigits: 2,
      ...options
    };
    return new Intl.NumberFormat(this.locale, defaultOptions).format(value);
  }

  // 复数处理
  pluralFormatter(value, key) {
    const rules = new Intl.PluralRules(this.locale);
    const rule = rules.select(value);

    // 获取对应规则的消息
    const message = this.t(`${key}.${rule}`, { count: value });
    return message;
  }

  // 方向检测
  getDirection() {
    const rtlLocales = ['ar', 'he', 'fa', 'ur'];
    const localePrefix = this.locale.split('-')[0];
    return rtlLocales.includes(localePrefix) ? 'rtl' : 'ltr';
  }

  // 语言切换监听
  onLocaleChange(callback) {
    this.localeChangeCallback = callback;
  }

  notifyLocaleChange() {
    if (this.localeChangeCallback) {
      this.localeChangeCallback(this.locale);
    }

    // 触发自定义事件
    const event = new CustomEvent('localechange', {
      detail: { locale: this.locale }
    });
    window.dispatchEvent(event);
  }

  // 加载翻译文件
  async loadMessages(locale) {
    try {
      const response = await fetch(`/locales/${locale}.json`);
      const messages = await response.json();
      this.messages[locale] = messages;
      return messages;
    } catch (error) {
      console.error(`Failed to load locale ${locale}:`, error);
      throw error;
    }
  }

  // 批量加载多个语言包
  async loadMessagesBatch(locales) {
    const promises = locales.map(locale => this.loadMessages(locale));
    return Promise.all(promises);
  }
}

// 创建 i18n 实例
const i18n = new I18n({
  messages: translations,
  fallbackLocale: 'en-US'
});
```

### React Hook 实现

```javascript
import { useState, useEffect, useContext, createContext, useCallback } from 'react';
import { createGlobalStyle } from 'styled-components';

// 创建 i18n 上下文
const I18nContext = createContext();

// 全局样式 - 处理 RTL 支持
const GlobalStyle = createGlobalStyle`
  body {
    direction: ${props => props.direction};
    text-align: ${props => props.direction === 'rtl' ? 'right' : 'left'};
  }

  .rtl-aware {
    direction: ${props => props.direction};
  }
`;

export function I18nProvider({ children, messages, defaultLocale = 'zh-CN' }) {
  const [locale, setLocale] = useState(defaultLocale);
  const [direction, setDirection] = useState('ltr');

  // 改变语言
  const changeLocale = useCallback(async (newLocale) => {
    if (messages[newLocale]) {
      setLocale(newLocale);
      localStorage.setItem('locale', newLocale);

      // 更新文本方向
      const newDirection = newLocale.startsWith('ar') ||
        newLocale.startsWith('he') ||
        newLocale.startsWith('fa') ? 'rtl' : 'ltr';
      setDirection(newDirection);

      // 更新 HTML 属性
      document.documentElement.lang = newLocale;
      document.documentElement.dir = newDirection;
    }
  }, [messages]);

  // 翻译函数
  const t = useCallback((key, values = {}) => {
    const keys = key.split('.');
    let current = messages[locale];

    for (const k of keys) {
      if (current && typeof current === 'object' && k in current) {
        current = current[k];
      } else {
        return key; // 返回键名作为后备
      }
    }

    if (typeof current === 'string') {
      // 处理插值
      return current.replace(/\{\{(\w+)\}\}/g, (match, param) => {
        return values[param] !== undefined ? values[param] : match;
      });
    }

    return current;
  }, [messages, locale]);

  // 格式化函数
  const format = useCallback({
    number: (value, options) => new Intl.NumberFormat(locale, options).format(value),
    date: (value, options) => new Intl.DateTimeFormat(locale, options).format(new Date(value)),
    currency: (value, currency) => new Intl.NumberFormat(locale, {
      style: 'currency',
      currency
    }).format(value),
    plural: (value, key) => {
      const rules = new Intl.PluralRules(locale);
      const rule = rules.select(value);
      return t(`${key}.${rule}`, { count: value });
    }
  }, [locale, t]);

  // 初始化
  useEffect(() => {
    const savedLocale = localStorage.getItem('locale');
    if (savedLocale && messages[savedLocale]) {
      changeLocale(savedLocale);
    }

    // 检测浏览器语言
    const browserLocale = navigator.language;
    if (messages[browserLocale] && !savedLocale) {
      changeLocale(browserLocale);
    }
  }, [messages, changeLocale]);

  const value = {
    locale,
    setLocale: changeLocale,
    t,
    format,
    direction
  };

  return (
    <I18nContext.Provider value={value}>
      <GlobalStyle direction={direction} />
      {children}
    </I18nContext.Provider>
  );
}

// 自定义 Hook
export function useI18n() {
  const context = useContext(I18nContext);
  if (!context) {
    throw new Error('useI18n must be used within I18nProvider');
  }
  return context;
}
```

```javascript
import { ref, computed, watch, provide, inject } from 'vue';

// i18n 插件
const i18nKey = Symbol('i18n');

export function createI18n(options) {
  const { locale, fallbackLocale, messages } = options;

  const currentLocale = ref(locale || 'zh-CN');
  const currentMessages = ref(messages || {});

  // 翻译函数
  const t = (key, values = {}) => {
    const keys = key.split('.');
    let current = currentMessages.value[currentLocale.value];

    for (const k of keys) {
      if (current && typeof current === 'object' && k in current) {
        current = current[k];
      } else {
        // 尝试回退语言
        if (fallbackLocale && currentLocale.value !== fallbackLocale) {
          current = currentMessages.value[fallbackLocale];
          for (const k2 of keys) {
            if (current && typeof current === 'object' && k2 in current) {
              current = current[k2];
            } else {
              return key;
            }
          }
          break;
        }
        return key;
      }
    }

    if (typeof current === 'string') {
      return current.replace(/\{\{(\w+)\}\}/g, (match, param) => {
        return values[param] !== undefined ? values[param] : match;
      });
    }

    return current;
  };

  // 格式化函数
  const format = {
    number: (value, options) => {
      return new Intl.NumberFormat(currentLocale.value, options).format(value);
    },
    date: (value, options) => {
      return new Intl.DateTimeFormat(currentLocale.value, options).format(new Date(value));
    },
    currency: (value, currency, options = {}) => {
      return new Intl.NumberFormat(currentLocale.value, {
        style: 'currency',
        currency,
        ...options
      }).format(value);
    }
  };

  // 改变语言
  const setLocale = (newLocale) => {
    if (currentMessages.value[newLocale]) {
      currentLocale.value = newLocale;
      localStorage.setItem('locale', newLocale);

      // 更新 HTML 属性
      document.documentElement.lang = newLocale;
    }
  };

  // 获取方向
  const direction = computed(() => {
    const rtlLocales = ['ar', 'he', 'fa', 'ur'];
    const localePrefix = currentLocale.value.split('-')[0];
    return rtlLocales.includes(localePrefix) ? 'rtl' : 'ltr';
  });

  // 安装插件
  const install = (app) => {
    app.provide(i18nKey, {
      locale: currentLocale,
      setLocale,
      t,
      format,
      direction
    });

    app.config.globalProperties.$t = t;
    app.config.globalProperties.$i18n = {
      locale: currentLocale,
      setLocale,
      t,
      format
    };

    // 初始化
    const savedLocale = localStorage.getItem('locale');
    if (savedLocale && currentMessages.value[savedLocale]) {
      setLocale(savedLocale);
    }
  };

  return {
    install,
    t,
    setLocale,
    format,
    direction,
    locale: currentLocale
  };
}

// 使用 Hook
export function useI18n() {
  const i18n = inject(i18nKey);
  if (!i18n) {
    throw new Error('useI18n() is called without provider.');
  }
  return i18n;
}
```

```javascript
// 更精确的相对时间
function formatSmartTime(date) {
  const now = new Date();
  const diffInSeconds = (now - new Date(date)) / 1000;

  if (diffInSeconds < 60) {
    return formatRelativeTime(-Math.floor(diffInSeconds), 'second');
  } else if (diffInSeconds < 3600) {
    return formatRelativeTime(-Math.floor(diffInSeconds / 60), 'minute');
  } else if (diffInSeconds < 86400) {
    return formatRelativeTime(-Math.floor(diffInSeconds / 3600), 'hour');
  } else if (diffInSeconds < 2592000) {
    return formatRelativeTime(-Math.floor(diffInSeconds / 86400), 'day');
  } else {
    return formatDate(date, 'short');
  }
}
```

```javascript
// 懒加载翻译文件
async function loadLocale(locale) {
  if (loadedLocales.has(locale)) {
    return;
  }

  try {
    const response = await fetch(`/locales/${locale}.json`);
    const messages = await response.json();

    i18n.setLocaleMessage(locale, messages);
    loadedLocales.add(locale);
  } catch (error) {
    console.error(`Failed to load locale ${locale}:`, error);
  }
}

// 预加载常用语言
async function preloadCommonLocales() {
  const commonLocales = ['zh-CN', 'en-US'];
  await Promise.all(commonLocales.map(loadLocale));
}

function getCachedTranslation(key, values) {
  const cacheKey = `${key}_${JSON.stringify(values)}`;
  if (translationCache.has(cacheKey)) {
    return translationCache.get(cacheKey);
  }

  const translation = t(key, values);
  translationCache.set(cacheKey, translation);
  return translation;
}

  translationCache.clear();
```

```javascript
// 提供缺失翻译处理器
class I18n {
  constructor(options) {
    this.missingHandler = options.missingHandler || this.defaultMissingHandler;
  }

  defaultMissingHandler(key, locale) {
    console.warn(`Translation missing: ${key} in ${locale}`);
    // 可以返回原始键名或回退语言
    return key;
  }

  // 或者返回带有特殊标记的文本
  advancedMissingHandler(key, locale) {
    const fallbackKey = this.getMessage(key, this.fallbackLocale);
    if (fallbackKey) {
      return `[${locale}]${fallbackKey}`;
    }
    return `[MISSING: ${key}]`;
  }
}
```

## 总结

前端国际化是现代应用开发的重要环节，通过合理的架构设计，可以为全球用户提供良好的本地化体验。

核心要点：

1. **模块化设计**：将翻译内容按功能模块组织，便于管理和维护
2. **性能优化**：实现懒加载和缓存机制，优化首屏加载
3. **用户体验**：提供平滑的语言切换和即时反馈
4. **文化适配**：考虑不同地区的文化差异和习惯
5. **技术支持**：利用 Intl API 提供内置的国际化支持

记住：
- 国际化不仅仅是翻译，更是一种用户体验
- 从项目开始就考虑国际化需求
- 使用专业的翻译管理工具
- 定期维护和更新翻译内容
- 考虑用户的语言偏好和地区设置

通过系统化的国际化实现，你的应用将能够服务全球用户，提升产品的国际竞争力。
