## 概述

前端文件下载是Web应用中常见且重要的功能，涵盖从简单的文件下载到复杂的批量下载、断点续传等多种场景。随着浏览器能力的提升，前端文件下载技术也在不断发展，本文将详细介绍各种文件下载实现方案、最佳实践以及性能优化策略。

## 核心概念

### 下载方式分类

**1. 浏览器原生下载**
- `<a>` 标签的 `download` 属性
- window.open() 方法
- location.href / location.assign()

**2. Blob 对象下载**
- Blob API 创建二进制大对象
- URL.createObjectURL() 创建下载链接
- 适合动态生成内容或处理API响应

**3. Fetch API 下载**
- 现代化的异步请求方式
- 支持进度监控和错误处理
- 适合大文件和复杂场景

**4. Service Worker 离线下载**
- 后台下载能力
- 支持断点续传
- 提升下载可靠性

```javascript
// 带进度监控的 Fetch 下载
async function downloadWithProgress(url, filename, onProgress) {
    try {
        const response = await fetch(url);
        
        if (!response.ok) {
            throw new Error(`下载失败: ${response.status}`);
        }
        
        const contentLength = response.headers.get('Content-Length');
        const total = parseInt(contentLength, 10);
        let loaded = 0;
        
        const reader = response.body.getReader();
        const chunks = [];
        
        while (true) {
            const { done, value } = await reader.read();
            
            if (done) break;
            
            chunks.push(value);
            loaded += value.length;
            
            if (onProgress && total > 0) {
                const percentage = Math.round((loaded / total) * 100);
                onProgress(percentage, loaded, total);
            }
        }
        
        const blob = new Blob(chunks);
        downloadBlob(blob, filename);
        
        return { success: true, total, loaded };
    } catch (error) {
        console.error('下载错误:', error);
        throw error;
    }
}

// 使用示例
downloadWithProgress(
    'https://example.com/large-file.zip',
    'download.zip',
    (percentage, loaded, total) => {
        console.log(`下载进度: ${percentage}% (${loaded}/${total} bytes)`);
        // 更新进度条UI
        updateProgressBar(percentage);
    }
);
```

### 1. 批量文件下载

```javascript
class BatchDownloader {
    constructor(options = {}) {
        this.options = {
            maxConcurrent: options.maxConcurrent || 3,
            retryCount: options.retryCount || 3,
            onProgress: options.onProgress || (() => {}),
            onComplete: options.onComplete || (() => {}),
            onError: options.onError || (() => {}),
            ...options
        };
        
        this.queue = [];
        this.activeDownloads = 0;
        this.results = {
            success: [],
            failed: []
        };
    }
    
    async download(files) {
        this.queue = files;
        this.results = { success: [], failed: [] };
        
        await this.processQueue();
        
        this.options.onComplete(this.results);
        return this.results;
    }
    
    async processQueue() {
        while (this.queue.length > 0 || this.activeDownloads > 0) {
            // 启动新的下载任务
            while (this.queue.length > 0 && this.activeDownloads < this.options.maxConcurrent) {
                this.activeDownloads++;
                const file = this.queue.shift();
                this.downloadFile(file);
            }
            
            // 等待一段时间避免CPU占用过高
            await new Promise(resolve => setTimeout(resolve, 100));
        }
    }
    
    async downloadFile(file) {
        let retryCount = 0;
        
        while (retryCount <= this.options.retryCount) {
            try {
                await this.downloadSingleFile(file);
                this.results.success.push(file);
                break;
            } catch (error) {
                retryCount++;
                if (retryCount > this.options.retryCount) {
                    this.results.failed.push({ file, error });
                    this.options.onError(file, error);
                    break;
                }
            }
        }
        
        this.activeDownloads--;
        this.updateProgress();
    }
    
    async downloadSingleFile(file) {
        const response = await fetch(file.url);
        
        if (!response.ok) {
            throw new Error(`下载失败: ${response.status}`);
        }
        
        const blob = await response.blob();
        downloadBlob(blob, file.filename);
    }
    
    updateProgress() {
        const total = this.results.success.length + this.results.failed.length;
        const totalFiles = this.results.success.length + this.results.failed.length + this.queue.length;
        const percentage = totalFiles > 0 ? Math.round((total / totalFiles) * 100) : 0;
        
        this.options.onProgress({
            percentage,
            total,
            totalFiles,
            success: this.results.success.length,
            failed: this.results.failed.length
        });
    }
}

// 使用示例
const batchDownloader = new BatchDownloader({
    maxConcurrent: 3,
    retryCount: 2,
    onProgress: (progress) => {
        console.log(`批量下载进度: ${progress.percentage}%`);
        console.log(`成功: ${progress.success}, 失败: ${progress.failed}`);
    },
    onComplete: (results) => {
        console.log('批量下载完成:', results);
    },
    onError: (file, error) => {
        console.error(`文件下载失败: ${file.filename}`, error);
    }
});

const filesToDownload = [
    { url: 'https://example.com/file1.pdf', filename: 'document1.pdf' },
    { url: 'https://example.com/file2.pdf', filename: 'document2.pdf' },
    { url: 'https://example.com/file3.pdf', filename: 'document3.pdf' }
];

batchDownloader.download(filesToDownload);
```

```javascript
class ResumableDownloader {
    constructor(options = {}) {
        this.options = {
            chunkSize: options.chunkSize || 1024 * 1024, // 1MB
            storageKey: options.storageKey || 'download-progress',
            ...options
        };
        
        this.downloadState = this.loadDownloadState();
    }
    
    loadDownloadState() {
        const saved = localStorage.getItem(this.options.storageKey);
        return saved ? JSON.parse(saved) : {};
    }
    
    saveDownloadState() {
        localStorage.setItem(this.options.storageKey, JSON.stringify(this.downloadState));
    }
    
    async download(url, filename) {
        const fileId = this.getFileId(url, filename);
        
        // 检查是否已有下载进度
        let downloadedChunks = this.downloadState[fileId] || 0;
        
        try {
            const response = await fetch(url);
            const totalSize = parseInt(response.headers.get('Content-Length'), 10);
            
            if (!response.ok) {
                throw new Error(`下载失败: ${response.status}`);
            }
            
            const reader = response.body.getReader();
            const chunks = [];
            let receivedLength = 0;
            
            while (true) {
                const { done, value } = await reader.read();
                
                if (done) break;
                
                chunks.push(value);
                receivedLength += value.length;
                
                // 保存下载进度
                this.downloadState[fileId] = receivedLength;
                this.saveDownloadState();
                
                // 报告进度
                if (this.options.onProgress) {
                    const percentage = Math.round((receivedLength / totalSize) * 100);
                    this.options.onProgress(percentage, receivedLength, totalSize);
                }
            }
            
            const blob = new Blob(chunks);
            downloadBlob(blob, filename);
            
            // 下载完成，清除进度
            delete this.downloadState[fileId];
            this.saveDownloadState();
            
            return { success: true, totalSize, receivedLength };
        } catch (error) {
            console.error('下载错误:', error);
            throw error;
        }
    }
    
    getFileId(url, filename) {
        return btoa(url + filename);
    }
    
    cancelDownload(url, filename) {
        const fileId = this.getFileId(url, filename);
        delete this.downloadState[fileId];
        this.saveDownloadState();
    }
    
    getProgress(url, filename) {
        const fileId = this.getFileId(url, filename);
        return this.downloadState[fileId] || 0;
    }
}

// 使用示例
const resumableDownloader = new ResumableDownloader({
    chunkSize: 1024 * 1024, // 1MB
    storageKey: 'my-download-progress',
    onProgress: (percentage, loaded, total) => {
        console.log(`下载进度: ${percentage}% (${loaded}/${total} bytes)`);
    }
});

// 开始下载
async function startDownload() {
    try {
        await resumableDownloader.download(
            'https://example.com/large-file.zip',
            'large-file.zip'
        );
        console.log('下载完成');
    } catch (error) {
        console.error('下载失败:', error);
    }
}

// 检查下载进度
function checkProgress() {
    const progress = resumableDownloader.getProgress(
        'https://example.com/large-file.zip',
        'large-file.zip'
    );
    console.log(`已下载: ${progress} bytes`);
}
```

```javascript
            // 创建ZIP文件
            const zip = new JSZip();
```

```javascript
// 使用 FileReader 读取大文件时的内存管理
function readLargeFileInChunks(file, chunkSize = 1024 * 1024) {
    return new Promise((resolve, reject) => {
        const reader = new FileReader();
        let offset = 0;
        const chunks = [];
        
        reader.onload = function(e) {
            if (e.target.error) {
                reject(e.target.error);
                return;
            }
            
            chunks.push(e.target.result);
            offset += chunkSize;
            
            if (offset >= file.size) {
                resolve(chunks);
            } else {
                readNextChunk();
            }
        };
        
        function readNextChunk() {
            const slice = file.slice(offset, offset + chunkSize);
            reader.readAsArrayBuffer(slice);
        }
        
        readNextChunk();
    });
}
```

```javascript
// 完善的错误处理
class RobustDownloader {
    async download(url, filename, options = {}) {
        const { timeout = 30000, retries = 3 } = options;
        
        for (let attempt = 1; attempt <= retries; attempt++) {
            try {
                return await this.downloadWithTimeout(url, filename, timeout);
            } catch (error) {
                console.error(`下载尝试 ${attempt} 失败:`, error);
                
                if (attempt === retries) {
                    throw new Error(`下载失败，已重试 ${retries} 次: ${error.message}`);
                }
                
                // 指数退避
                await new Promise(resolve => 
                    setTimeout(resolve, Math.pow(2, attempt) * 1000)
                );
            }
        }
    }
    
    async downloadWithTimeout(url, filename, timeout) {
        const controller = new AbortController();
        const timeoutId = setTimeout(() => controller.abort(), timeout);
        
        try {
            const response = await fetch(url, { 
                signal: controller.signal 
            });
            
            if (!response.ok) {
                throw new Error(`HTTP错误: ${response.status} ${response.statusText}`);
            }
            
            // 检查内容类型
            const contentType = response.headers.get('Content-Type');
            if (!this.isValidContentType(contentType)) {
                throw new Error(`无效的内容类型: ${contentType}`);
            }
            
            const blob = await response.blob();
            downloadBlob(blob, filename);
            
            return { success: true };
        } catch (error) {
            if (error.name === 'AbortError') {
                throw new Error('下载超时');
            }
            throw error;
        } finally {
            clearTimeout(timeoutId);
        }
    }
    
    isValidContentType(contentType) {
        // 根据实际需求验证内容类型
        const validTypes = [
            'application/pdf',
            'application/json',
            'text/plain',
            'image/jpeg',
            'image/png'
        ];
        
        return validTypes.some(type => contentType?.includes(type));
    }
}
```

```javascript
    }

        if (bytes === 0) return '0 Bytes';
        
        const k = 1024;
        const sizes = ['Bytes', 'KB', 'MB', 'GB'];
        const i = Math.floor(Math.log(bytes) / Math.log(k));
        
        return Math.round((bytes / Math.pow(k, i)) * 100) / 100 + ' ' + sizes[i];
    }
}

// 下载队列管理
class DownloadQueue {
    constructor(maxConcurrent = 3) {
        this.queue = [];
        this.active = [];
        this.maxConcurrent = maxConcurrent;
        this.progressContainer = null;
    }
    
    add(downloadTask) {
        this.queue.push(downloadTask);
        this.process();
    }
    
    async process() {
        while (this.active.length < this.maxConcurrent && this.queue.length > 0) {
            const task = this.queue.shift();
            this.active.push(task);
            
            task.execute()
                .finally(() => {
                    this.active = this.active.filter(t => t !== task);
                    this.process();
                });
        }
    }
    
    cancelAll() {
        this.queue.forEach(task => task.cancel());
        this.active.forEach(task => task.cancel());
        this.queue = [];
    }
}

// 下载任务
class DownloadTask {
    constructor(url, filename, progressContainer) {
        this.url = url;
        this.filename = filename;
        this.progressContainer = progressContainer;
        this.cancelled = false;
        this.controller = new AbortController();
    }
    
    async execute() {
        const progress = new DownloadProgress(this.progressContainer);
        
        try {
            await downloadWithProgress(
                this.url,
                this.filename,
                (percentage, loaded, total) => {
                    if (!this.cancelled) {
                        progress.updateProgress(percentage, loaded, total);
                    }
                },
                this.controller.signal
            );
            
            if (!this.cancelled) {
                progress.setComplete();
            }
        } catch (error) {
            if (!this.cancelled) {
                progress.setError(error.message);
            }
        }
    }
    
    cancel() {
        this.cancelled = true;
        this.controller.abort();
    }
}
```

## 总结

前端文件下载技术已经非常成熟，从简单的基础下载到复杂的批量下载、断点续传等功能都能很好地实现。选择合适的下载方案需要考虑文件大小、网络环境、用户体验等多个因素。
